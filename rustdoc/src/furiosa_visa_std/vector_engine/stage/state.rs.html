12/// VE state that tracks stash and ALU usage.
13///
14/// The type parameter `D` ties the stash's scalar type to the pipeline's current scalar type,
15/// ensuring at compile time that stash reads match the pipeline's `D`.
16///
17/// - `VeState<D, NoTensor>` — stash is empty.
18/// - `VeState<D, HasTensor<D, Mapping>>` — stash holds a `Tensor<D, Mapping>`.
19#[derive(Debug)]
20pub struct VeState<D: VeScalar, Stash: TensorState<D>> {
21    /// Stash state — either [`NoTensor`] or [`HasTensor`] holding the actual tensor.
22    pub(crate) stash: Stash,
23    /// Set of ALUs that have been used.
24    pub(crate) used_alus: HashSet<RngdAlu>,
25    /// Marker for the scalar type `D`.
26    _marker: PhantomData<D>,
27}
28
29impl<D: VeScalar, Stash: TensorState<D>> VeState<D, Stash> {
30    /// Checks if ALU is available and marks it as used.
31    /// Panics if ALU is already in use.
32    pub fn use_alu(&mut self, alu: RngdAlu) {
33        assert!(!self.used_alus.contains(&alu), "{alu:?} is already in use");
34        self.used_alus.insert(alu);
35    }
36
37    /// Clones the stash data, transposing to target mapping if needed.
38    pub fn clone_stash_as<TargetMapping: M>(&self) -> Option<Tensor<D, TargetMapping>> {
39        self.stash.clone_tensor_as()
40    }
41
42    /// Clones the stash data, transposing to target mapping if needed.
43    ///
44    /// TODO: This should checked at compile-time, or replaced with clone_stash_as. (PROG-155)
45    ///
46    /// # Panics
47    /// Panics if the target type D2 does not match the stashed type D.
48    pub fn force_clone_stash_as<D2: VeScalar, TargetMapping: M>(&self) -> Option<Tensor<D2, TargetMapping>> {
49        assert!(
50            D::KIND == D2::KIND,
51            "stash type mismatch: stashed as {:?}, requested as {:?}",
52            D::KIND,
53            D2::KIND
54        );
55
56        let cloned = self.stash.clone_tensor_as::<TargetMapping>();
57
58        // SAFETY: Runtime assert above ensures D and D2 are the same concrete type
59        // (both i32 or both f32), so Tensor<D, _> and Tensor<D2, _> have identical layout.
60        cloned.map(|t| unsafe {
61            // We cannot use transmute because compiler doesn't know that D and D2 have same size.
62            // transmute_copy + forget does the same thing.
63            let converted = std::mem::transmute_copy(&t);
64            std::mem::forget(t);
65            converted
66        })
67    }
68}

70impl<D: VeScalar> VeState<D, NoTensor> {
71    /// Creates a new empty VeState.
72    pub fn new() -> Self {
73        Self {
74            stash: NoTensor,
75            used_alus: HashSet::new(),
76            _marker: PhantomData,
77        }
78    }
79
80    /// Creates a new VeState with a single ALU already used.
81    pub fn with_alu(alu: RngdAlu) -> Self {
82        Self {
83            stash: NoTensor,
84            used_alus: HashSet::from_iter([alu]),
85            _marker: PhantomData,
86        }
87    }
88
89    /// Stores the tensor data in the stash, returning a new VeState with the stash's mapping.
90    /// Consumes self to change the `TensorState` parameter.
91    pub fn stash<NewMapping: M>(self, data: &Tensor<D, NewMapping>) -> VeState<D, HasTensor<D, NewMapping>> {
92        VeState {
93            stash: HasTensor::new(data.clone()),
94            used_alus: self.used_alus,
95            _marker: PhantomData,
96        }
97    }
98
99    /// Consumes another VeState and merges its ALU usage into this one.
100    /// Used when combining two groups in binary operations.
101    /// Since both groups share the same ALU state (global resource),
102    /// this simply performs a union of used ALUs.
103    pub fn merge(&mut self, other: VeState<D, NoTensor>) {
104        self.used_alus.extend(other.used_alus);
105    }
106
107    /// Converts VeState to different scalar type parameters, preserving ALU tracking.
108    pub fn retype<NewD: VeScalar>(self) -> VeState<NewD, NoTensor> {
109        VeState {
110            stash: NoTensor,
111            used_alus: self.used_alus,
112            _marker: PhantomData,
113        }
114    }
115}
