8fn to_buffer<D: Scalar, Chip: M, Element: M>(addr: u64) -> Buffer {
9    let len = <Pair<Chip, Element> as M>::SIZE * std::mem::size_of::<D>();
10    Buffer::from_raw(unsafe { lib().furiosa_buffer_from_npu(rt(), addr, len) })
11}
