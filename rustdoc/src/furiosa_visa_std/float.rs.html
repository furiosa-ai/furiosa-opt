13pub(crate) fn f8_e4_from_f32(v: f32) -> u8 {
14    if v.is_nan() {
15        return 0x7f; // qNaN
16    }
17    let n = v.to_bits();
18    let exp = (n >> 23) & 0xff;
19    let man = n & 0x7fffff;
20    let abs_value = if exp > 127 - 7 {
21        let valid = (n >> 20) & 0x7ff;
22        let val = if (n & 0xfffff) + (valid & 1) > 0x80000 {
23            valid + 1
24        } else {
25            valid
26        };
27        if val >= (((127 + 8) << 3) | 0x6) {
28            0x7e_u32 // MAX
29        } else {
30            val - ((127 - 7) << 3)
31        }
32    } else if exp == 127 - 7 {
33        if man >= 0x700000 {
34            8
35        } else if man > 0x500000 {
36            7
37        } else if man >= 0x300000 {
38            6
39        } else if man > 0x100000 {
40            5
41        } else {
42            4
43        }
44    } else if exp == 127 - 8 {
45        if man >= 0x600000 {
46            4
47        } else if man > 0x200000 {
48            3
49        } else {
50            2
51        }
52    } else if exp == 127 - 9 {
53        if man >= 0x400000 { 2 } else { 1 }
54    } else if exp == 127 - 10 && man != 0 {
55        1
56    } else {
57        0
58    };
59    (((n >> 31) << 7) | abs_value) as u8
60}

62pub(crate) fn f8_e4_is_zero(bits: u8) -> bool {
63    bits == F8E4_ZERO || bits == F8E4_NEG_ZERO
64}

106const MANTISSA_MASK: u32 = (1 << 23) - 1;
107const EXPONENT_MASK: u32 = (1 << 8) - 1;
108const EXPONENT_SHIFT: i32 = 23;
109const SIGN_SHIFT: i32 = 31;
110const EXPONENT_OFFSET: u32 = (1 << 7) - 1;
111
112pub(crate) fn float_to_fixedpoint(input: f32, integer_width: u32) -> i32 {
113    assert!(integer_width < 32);
114    if input.is_nan() {
115        return if input.is_sign_positive() { i32::MAX } else { i32::MIN };
116    }
117
118    let fraction_width = 31 - integer_width;
119    let f64_result: f64 = f64::from(input)
120        .mul(2_f64.powi(fraction_width as i32))
121        .round()
122        .clamp(i32::MIN as f64, i32::MAX as f64);
123    f64_result as i32
124}

126pub(crate) fn fixedpoint_to_float(input: i32, integer_width: u32) -> f32 {
127    assert!(integer_width < 32);
128    let abs_value: u32 = input.unsigned_abs();
129
130    let mask = (1 << 31) - 1;
131    let leading_zeros = (abs_value & mask).leading_zeros() - 1;
132
133    let aligned_abs_value = abs_value << leading_zeros;
134    let rounded = round_to_nearest_even(aligned_abs_value, 31, 24);
135    let mantissa = rounded & MANTISSA_MASK;
136
137    let sign = if input < 0 { 1 } else { 0 };
138
139    let exponent = if input == i32::MIN {
140        integer_width + EXPONENT_OFFSET
141    } else if leading_zeros == 31 {
142        0
143    } else {
144        let mut result = integer_width + (EXPONENT_OFFSET - 1) - leading_zeros;
145        if (rounded >> 24) != 0 {
146            result += 1;
147        }
148        result
149    };
150    let float_as_bits = (sign << SIGN_SHIFT) | ((exponent & EXPONENT_MASK) << EXPONENT_SHIFT) | mantissa;
151    f32::from_bits(float_as_bits)
152}

154fn round_to_nearest_even(input: u32, iw: u32, ow: u32) -> u32 {
155    let in_width_mask = (1u32 << iw) - 1;
156    let input_valid = input & in_width_mask;
157    let shift = (iw - ow) - 2;
158
159    let shift_mask = (1u32 << shift) - 1;
160    let discarded = input_valid & shift_mask;
161    let sticky = discarded != 0;
162
163    let interest = input_valid >> shift;
164    let round_factor = ((interest & 0x7) << 1) | u32::from(sticky);
165
166    let round_up = matches!(
167        round_factor,
168        0b0110 | 0b1110 | 0b0101 | 0b1101 | 0b0111 | 0b1111 | 0b1100
169    );
170
171    (interest >> 2) + u32::from(round_up)
