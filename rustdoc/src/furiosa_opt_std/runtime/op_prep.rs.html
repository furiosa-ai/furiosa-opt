1//! Per-op preparation step shared between Simulation and Typecheck.
2//!
3//! Each helper here is the "prep" that runs before an op's real work — it derives the mapping
4//! data the op needs (broadcast residue, reshape quotient, scatter payload, …) and panics on
5//! ill-formed mappings along the way. Called from exactly two places per op: the Simulation
6//! impl on `MathRawTensor` consumes the returned mapping to drive its iteration loop, and the
7//! Typecheck impl on `PhantomRawTensor` discards the value (`let _ = …`) but still triggers
8//! the panic side, so type errors surface without the value-level loop ever running. Npu /
9//! Emulation never call them — their `BufRawTensor` op bodies are `todo!()`.
10//!
11//! Most helpers are noun-style and return a derived `FMapping`; `assert_zip` is the one pure
12//! check (returns `()`). The shared file lets Simulation and Typecheck stay in lockstep on
13//! mapping algebra without duplicating the prep step.
14//!
15//! [`Backend`]: crate::runtime::backend::Backend
16//! [`RawTensor`]: crate::tensor::raw::RawTensor
17

21/// Broadcast residue for `RawTensor::write_transpose`. Returns the broadcast portion of the
22/// dst-by-src division; `MathRawTensor`'s write loop iterates over it. Panics on ill-formed
23/// mappings.
24pub(crate) fn transpose_broadcast<Src: M, Dst: M>(allow_broadcast: bool) -> FMapping {
25    let src_mapping = Src::to_value();
26    let dst_mapping = Dst::to_value();
27    let broadcast = dst_mapping
28        .divide(&src_mapping)
29        .exact_checked()
30        .unwrap_or_else(|e| {
31            panic!(
32                "[write_transpose] failed to divide mapping expressions for transpose: {e:?}\n\
33                 src_mapping: {src_mapping:?}\n\
34                 dst_mapping: {dst_mapping:?}"
35            )
36        })
37        .relaxed_residues()
38        .dividend_residue;
39    if !allow_broadcast {
40        assert!(broadcast.is_padding());
41    }
42    broadcast
43}

45/// Axes-equality check for `RawTensor::zip_with`.
46pub(crate) fn assert_zip(lhs_axes: &[Term], rhs_axes: &[Term]) {
47    assert_eq!(
48        lhs_axes, rhs_axes,
49        "Tensors must have the same axes for element-wise binary operations"
50    );
51}

53/// Broadcast residue for `RawTensor::reduce_then_broadcast`. Panics on ill-formed mappings.
54pub(crate) fn reduce_broadcast(src_axes: &[Term], dst_axes: &[Term]) -> FMapping {
55    let src_fmapping = FMapping::from_axes(RSlice::from(src_axes));
56    let dst_fmapping = FMapping::from_axes(RSlice::from(dst_axes));
57    dst_fmapping
58        .divide(src_fmapping)
59        .exact_checked()
60        .expect("Failed to calculate broadcast factor")
61        .relaxed_residues()
62        .dividend_residue
63}

65/// Scatter payload mapping and destination axis term. Used by every backend whose
66/// `RawTensor::write_scatter` body iterates the host. Panics on ill-formed scatters.
67pub(crate) fn scatter_params(src_fmapping: &FMapping, dst_fmapping: &FMapping, key: &FMapping) -> (FMapping, Term) {
68    let payload = src_fmapping
69        .clone()
70        .divide(key.clone())
71        .exact_checked()
72        .unwrap_or_else(|e| panic!("Scatter key `{key:?}` not found in source `{src_fmapping:?}`: {e:?}"))
73        .relaxed_residues()
74        .dividend_residue;
75    let dst_residue = dst_fmapping
76        .clone()
77        .divide(payload.clone())
78        .exact_checked()
79        .unwrap_or_else(|e| panic!("Destination `{dst_fmapping:?}` missing payload axes `{payload:?}`: {e:?}"))
80        .relaxed_residues()
81        .dividend_residue;
82    let (dst_term, _) = dst_residue
83        .into_inner()
84        .into_iter()
85        .find_map(|factor| match factor {
86            Factor::Term { inner, resize } => Some((inner, resize)),
87            Factor::Padding { .. } => None,
88        })
89        .expect("Destination has no scatter target axis after removing payload");
90    (payload, dst_term)
91}

93/// Derived parameters for `RawTensor::write_gather`.
94///
95/// Inverse of [`scatter_params`]:
96/// - scatter: `src ÷ key = payload`, `dst ÷ payload = dst_term` (the indexed axis on dst).
97/// - gather:  `dst ÷ idx_axes = payload`, `src ÷ payload = src_term` (the indexed axis on src/table).
98pub(crate) struct GatherParams {
99    /// Axes shared by `src` (the table) and `dst` (the gather output), iterated identically
100    /// on both sides. Derived as `dst ÷ idx_axes`.
101    pub payload: FMapping,
102    /// Axes in `dst` that replace the indexed axis on the source side. The `write_gather` loop
103    /// iterates these positions and writes each output slot sequentially. Derived as
104    /// `dst ÷ payload`.
105    pub idx_residue: FMapping,
106    /// Single-term locator on the `src` side identifying the indexed (lookup) axis. The runtime
107    /// indices tensor's values index into this axis. Derived as the surviving term in
108    /// `src ÷ payload`.
109    pub src_term: Term,
110}

112/// Compute [`GatherParams`] for a gather op. Used by every backend whose
113/// `RawTensor::write_gather` body iterates the host. Panics on ill-formed gathers.
114///
115/// `idx_fmapping` is the full mapping of the indices tensor; the "key-axes-replacement"
116/// inside `dst_fmapping` is derived as `dst ÷ payload`.
117pub(crate) fn gather_params(src_fmapping: &FMapping, dst_fmapping: &FMapping, idx_fmapping: &FMapping) -> GatherParams {
118    let payload = dst_fmapping
119        .clone()
120        .divide(idx_fmapping.clone())
121        .exact_checked()
122        .unwrap_or_else(|e| {
123            panic!("Gather indices `{idx_fmapping:?}` not contained in destination `{dst_fmapping:?}`: {e:?}")
124        })
125        .relaxed_residues()
126        .dividend_residue;
127    let idx_residue = dst_fmapping
128        .clone()
129        .divide(payload.clone())
130        .exact_checked()
131        .unwrap_or_else(|e| panic!("Destination `{dst_fmapping:?}` missing payload axes `{payload:?}`: {e:?}"))
132        .relaxed_residues()
133        .dividend_residue;
134    let src_residue = src_fmapping
135        .clone()
136        .divide(payload.clone())
137        .exact_checked()
138        .unwrap_or_else(|e| panic!("Source `{src_fmapping:?}` missing payload axes `{payload:?}`: {e:?}"))
139        .relaxed_residues()
140        .dividend_residue;
141    let (src_term, _) = src_residue
142        .into_inner()
143        .into_iter()
144        .find_map(|factor| match factor {
145            Factor::Term { inner, resize } => Some((inner, resize)),
146            Factor::Padding { .. } => None,
147        })
148        .expect("Source has no gather target axis after removing payload");
149    GatherParams {
150        payload,
151        idx_residue,
152        src_term,
153    }
154}
