10/// NPU backend.
11///
12/// Host-side code only prepares native `Vec<D>` staging buffers and moves them between host
13/// memory and NPU HBM through `to_hbm` / `from_hbm`. Tensor math is not interpreted on the host
14/// CPU for this backend — value-producing operations live on `BufRawTensor`'s `RawTensor` impl
15/// as `todo!()` placeholders shared with the Emulation backend. Under Npu the device kernel
16/// produces values and the host never reaches those bodies at runtime.
17#[derive(Debug, Clone, Copy)]
18pub struct Npu;
19
20impl Backend for Npu {
21    type RawTensor<D: Scalar> = BufRawTensor<D>;
22
23    async fn to_hbm<D: Scalar, Element: M, Chip: M, Element2: M>(
24        host: &HostTensor<D, Element, Self>,
25        address: Address,
26    ) -> HbmTensor<D, Chip, Element2, Self> {
27        Kernel::write(host, address).await
28    }
29
30    async fn from_hbm<D: Scalar, Chip: M, Element: M, Element2: M>(
31        hbm: &HbmTensor<D, Chip, Element, Self>,
32    ) -> HostTensor<D, Element2, Self> {
33        Kernel::read(hbm).await
34    }
35}
