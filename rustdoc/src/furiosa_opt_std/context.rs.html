54/// Device context.
55#[primitive(Context)]
56#[derive(Debug)]
57pub struct Context {
58    /// Tensor unit for the main context.
59    pub main: TuContext<{ Tu::Main }>,
60    /// Tensor unit for the sub context.
61    pub sub: TuContext<{ Tu::Sub }>,
62    /// Tensor DMA context.
63    pub tdma: DmaContext<{ Dma::Tensor }>,
64    /// PCIe DMA context.
65    pub pdma: DmaContext<{ Dma::Pcie }>,
66}
67
68impl crate::runtime::DeviceSend for &mut Context {}

70impl Context {
71    /// Acquire the tensor units.
72    pub fn acquire() -> impl DerefMut<Target = Context> {
73        static SINGLETON: LazyLock<Mutex<Context>> = LazyLock::new(|| {
74            Mutex::new(Context {
75                main: TuContext::<{ Tu::Main }> { _marker: PhantomData },
76                sub: TuContext::<{ Tu::Sub }> { _marker: PhantomData },
77                tdma: DmaContext::<{ Dma::Tensor }> { _marker: PhantomData },
78                pdma: DmaContext::<{ Dma::Pcie }> { _marker: PhantomData },
79            })
80        });
81        SINGLETON.lock().unwrap_or_else(|poisoned| poisoned.into_inner())
82    }
83}

85impl<const T: Tu> TuContext<{ T }> {
86    /// Begin a tensor unit operation in this context.
87    #[primitive(TuContext::begin)]
88    pub fn begin<'l, D: Scalar, Chip: M, Cluster: M, Slice: M, Element: M>(
89        &'l mut self,
90        tensor: DmTensorView<'l, D, Chip, Cluster, Slice, Element>,
91    ) -> BeginTensor<'l, { T }, D, Chip, Cluster, Slice, Identity, Element> {
92        // SAFETY: the mappings differ only by `Identity`.
93        BeginTensor::new(self, unsafe { tensor.inner.read().transmute() })
94    }
95
96    /// Begin a tensor unit operation in this context with interleaved tensors.
97    #[primitive(TuContext::begin_interleaved)]
98    pub fn begin_interleaved<'l, I: AxisName, D: Scalar, Chip: M, Cluster: M, Slice: M, Element: M>(
99        &'l mut self,
100        lhs: DmTensorView<'l, D, Chip, Cluster, Slice, Element>,
101        rhs: DmTensorView<'l, D, Chip, Cluster, Slice, Element>,
102    ) -> BeginTensor<'l, { T }, D, Chip, Cluster, Slice, Symbol<I>, Element> {
103        let mut output = Tensor::<D, m![{ Chip }, { Cluster }, { Slice }, { Symbol<I> }, { Element }]>::uninit();
104
105        for (i, input) in [lhs, rhs].into_iter().enumerate() {
106            output
107                .view_mut()
108                .tile::<Symbol<I>, m![{ Chip }, { Cluster }, { Slice }, 1 # 2, { Element }], 1>(i)
109                .write_transpose(input.inner, false);
110        }
111
112        BeginTensor::new(self, output)
113    }
114}

116impl TuContext<{ Tu::Sub }> {
117    /// Perform asymmetric cluster slice operation using ParallelCopy (stos command).
118    /// This operation allows different clusters to select different slice positions.
119    ///
120    /// # Arguments
121    /// * `tensor` - Input tensor with cluster dimension
122    /// * `slice_indices` - Array of slice indices, one per cluster
123    ///
124    /// # Example
125    /// For Cluster=2 with slice_indices=\[1,0\]:
126    /// - Cluster 0 selects slice position 1
127    /// - Cluster 1 selects slice position 0
128    ///
129    /// # Restrictions
130    /// The `AxisToSlice` should be the outermost axis in `Element`.
131    #[primitive(TuContext::parallel_copy_cluster_slice)]
132    pub fn parallel_copy_cluster_slice<
133        'l,
134        const CLUSTER_DIM: usize,
135        AxisToSlice: M,
136        AxisSlicedElement: M,
137        Element2: M,
138        D: Scalar,
139        Chip: M,
140        Cluster: M,
141        Slice: M,
142        Element: M,
143    >(
144        &mut self,
145        tensor: DmTensorView<'l, D, Chip, Cluster, Slice, Element>,
146        slice_indices: &[usize; CLUSTER_DIM],
147    ) -> super::tensor::memory::DmTensor<D, Chip, Cluster, Slice, Element2> {
148        let mut sliced = unsafe { DmTensor::from_addr(0) };
149
150        for (cluster_idx, slice_idx) in slice_indices.iter().enumerate() {
151            let cluster_slice = tensor.cluster_tile::<Cluster, 1, Padding<Identity, CLUSTER_DIM>>(cluster_idx);
152            cluster_slice
153                .tile::<AxisToSlice, 1, AxisSlicedElement>(*slice_idx)
154                .to_dm_view_pcopy(
155                    self,
156                    sliced
157                        .view_mut()
158                        .cluster_tile::<Cluster, 1, Padding<Identity, CLUSTER_DIM>>(cluster_idx),
159                );
160        }
161
162        sliced
163    }
164
165    /// Perform asymmetric chip slice operation using ParallelCopy (stos command).
166    /// This operation allows different chips to select different slice positions.
167    ///
168    /// # Arguments
169    /// * `tensor` - Input tensor with chip dimension
170    /// * `slice_indices` - Array of slice indices, one per chip
171    ///
172    /// # Example
173    /// For Chip=4 with slice_indices=\[3,0,1,2\]:
174    /// - Chip 0 selects slice position 3
175    /// - Chip 1 selects slice position 0
176    /// - Chip 2 selects slice position 1
177    /// - Chip 3 selects slice position 2
178    #[primitive(TuContext::parallel_copy_chip_slice)]
179    pub fn parallel_copy_chip_slice<
180        'l,
181        const CHIP_DIM: usize,
182        AxisToSlice: M,
183        AxisSlicedElement: M,
184        Element2: M,
185        D: Scalar,
186        Chip: M,
187        Cluster: M,
188        Slice: M,
189        Element: M,
190    >(
191        &mut self,
192        tensor: DmTensorView<'l, D, Chip, Cluster, Slice, Element>,
193        slice_indices: &[usize; CHIP_DIM],
194    ) -> DmTensor<D, Chip, Cluster, Slice, Element2> {
195        let mut sliced = unsafe { DmTensor::from_addr(0) };
196
197        for (chip_idx, slice_idx) in slice_indices.iter().enumerate() {
198            let chip_slice = tensor.chip_tile::<Chip, 1, Padding<Identity, CHIP_DIM>>(chip_idx);
199            chip_slice
200                .tile::<AxisToSlice, 1, AxisSlicedElement>(*slice_idx)
201                .to_dm_view_pcopy(
202                    self,
203                    sliced
204                        .view_mut()
205                        .chip_tile::<Chip, 1, Padding<Identity, CHIP_DIM>>(chip_idx),
206                );
207        }
208
209        sliced
210    }
