1//! Outer stage -- broadcasts both operands and multiplies them elementwise.
2//!
3//! The Outer stage runs three sub-stages in series:
4//!
5//! - [`stream_adapter`]: validates the stream (`CollectTensor`) side and
6//!   transposes it into the joint computation mapping.
7//! - [`trf_sequencer`]: transposes the TRF into the joint computation mapping.
8//! - Multiplier: elementwise multiplies the two aligned operands. Inputs widen
9//!   to the contraction output type (`i4`/`i8` -> `i32`, `f8`/`bf16` -> `f32`)
10//!   before multiplication, matching the DPE accumulator type.
11//!
12//! The resulting [`ContractOuterTensor`] carries a *single* multiplied tensor in
13//! the common computation mapping `[Chip, Cluster, Slice, Lane, Time, Packet]`,
14//! ready for the [`super::packet`] reduce-add.
15
16pub(super) mod stream_adapter;
17pub(super) mod trf_sequencer;

31/// Output of the Outer stage: the multiplied product (Stream Adapter * TRF Sequencer), ready for the Packet Reducer.
32///
33/// The product is already cast to the contraction output type ([`ContractionCast::Output`]),
34/// so the downstream Packet Reducer only has to run reduce-add.
35#[derive(Debug)]
36pub struct ContractOuterTensor<
37    'l,
38    const T: Tu,
39    D: Scalar + ContractionCast,
40    Chip: M,
41    Cluster: M,
42    Slice: M,
43    Lane: M,
44    Time: M,
45    Packet: M,
46    B: Backend = CurrentBackend,
47> {
48    pub(crate) ctx: &'l mut TuContext<{ T }>,
49    pub(crate) inner: Tensor<
50        <D as ContractionCast>::Output,
51        Pair<Chip, Pair<Cluster, Pair<Slice, Pair<Lane, Pair<Time, Packet>>>>>,
52        B,
53    >,
54}

56// ANCHOR: contract_outer_def
57impl<
58    'l,
59    const T: Tu,
60    P: CanApplyContractOuter,
61    D: Scalar + ContractionCast,
62    Chip: M,
63    Cluster: M,
64    Slice: M,
65    Time: M,
66    Packet: M,
67    B: Backend,
68> TuTensor<'l, T, P, D, Chip, Cluster, Slice, Time, Packet, B>
69{
70    /// Runs the Outer stage: Stream Adapter + TRF Sequencer broadcast both operands to a matching
71    /// computation shape, then the Multiplier widens to the contraction output type and multiplies
72    /// them elementwise. The result is a single multiplied tensor ready for the Packet Reducer.
73    #[primitive(TuTensor::contract_outer)]
74    pub fn contract_outer<OutTime: M, OutPacket: M, Lane: M, TrfElement: M>(
75        self,
76        trf_tensor: &TrfTensor<D, Chip, Cluster, Slice, Lane, TrfElement, B>,
77    ) -> ContractOuterTensor<'l, T, D, Chip, Cluster, Slice, Lane, OutTime, OutPacket, B>
78    where
79        D: Cast<<D as ContractionCast>::Output>,
80    {
81        verify_contract_outer::<D, Lane, Time, Packet, OutTime, OutPacket>();
82
83        // Broadcast both operands into the joint computation mapping.
84        let lhs = stream_adapter::contract_outer::<D, Chip, Cluster, Slice, Lane, Time, Packet, OutTime, OutPacket, B>(
85            self.inner,
86        );
87        let trf = trf_sequencer::contract_outer::<D, Chip, Cluster, Slice, Lane, TrfElement, OutTime, OutPacket, B>(
88            trf_tensor,
89        );
90
91        // Multiplier: widen to D::Output, then elementwise multiply.
92        let lhs = lhs.map(|v| v.map(|v| v.cast()));
93        let trf = trf.map(|v| v.map(|v| v.cast()));
94        let inner = lhs.zip_with(&trf, |a, b| a * b);
95
96        ContractOuterTensor { ctx: self.ctx, inner }
97    }
98}
99// ANCHOR_END: contract_outer_def

101/// Top-level validation for `.contract_outer`: runs the Stream Adapter checks and
102/// ensures the input `Time` divides the expected joint-mapping `Time`.
103fn verify_contract_outer<D: Scalar, Lane: M, Time: M, Packet: M, OutTime: M, OutPacket: M>() {
104    stream_adapter::verify_stream_adapter::<D, Lane, Time, Packet, OutTime, OutPacket>();
105
106    let expected_time = OutTime::to_value()
107        .factorize()
108        .mul(
109            OutPacket::to_value()
110                .factorize()
111                .split_at(D::length_from_bytes(crate::engine::FLIT_BYTES))
112                .0
113                .remove_padding(),
114        )
115        .normalize();
116    let input_time = Time::to_value().factorize();
117    expected_time
118        .divide(input_time.clone())
119        .exact_checked()
120        .expect("`contract_outer`: Time does not divide OutTime");
121}
