1//! Stream Adapter: stream-side validation and transpose for [`super::contract_outer`].
2//!
3//! - [`verify_stream_adapter`]: checks that the input `Time`/`Packet` can be
4//!   expanded into the joint computation mapping (inner-flit packet match,
5//!   Time-divides-OutTime, broadcast axes confined to innermost positions).
6//! - [`contract_outer`]: transposes the stream tensor into the joint shape.
7
8use abi_stable::std_types::Tuple2;
9use furiosa_mapping::*;
10
11use crate::engine::FLIT_BYTES;
12use crate::runtime::Backend;
13use crate::scalar::Scalar;
14use crate::tensor::Tensor;
15
16pub(super) fn contract_outer<
17    D: Scalar,
18    Chip: M,
19    Cluster: M,
20    Slice: M,
21    Lane: M,
22    Time: M,
23    Packet: M,
24    OutTime: M,
25    OutPacket: M,
26    B: Backend,
27>(
28    inner: Tensor<D, m![{ Chip }, { Cluster }, { Slice }, { Time }, { Packet }], B>,
29) -> Tensor<D, m![{ Chip }, { Cluster }, { Slice }, { Lane }, { OutTime }, { OutPacket }], B> {
30    verify_stream_adapter::<D, Lane, Time, Packet, OutTime, OutPacket>();
31    inner.transpose(true)
32}
33
34/// Validates that the input `Time`/`Packet` can be expanded into the joint
35/// computation mapping:
36/// - The inner flit of `OutPacket` matches the input `Packet`.
37/// - `Time × outer_flit(OutPacket)` divides `OutTime`, with broadcast axes
38///   confined to the innermost positions of `OutTime` (same relative order as
39///   in `Time`).
40pub(super) fn verify_stream_adapter<D: Scalar, Lane: M, Time: M, Packet: M, OutTime: M, OutPacket: M>() {
41    // Inner flit of OutPacket must match input Packet.
42    assert!(
43        [1, 2, 4, 8].contains(&Lane::SIZE),
44        "Lane::SIZE must be 1, 2, 4, or 8, got {}",
45        Lane::SIZE
46    );
47
48    let out_packet_size = D::size_in_bytes_from_length(OutPacket::SIZE);
49    assert!(
50        [32, 64].contains(&out_packet_size),
51        "OutPacket must be 32 or 64 bytes (matching PackSize ∈ {{1, 2}}), got {out_packet_size} bytes"
52    );
53
54    let flit_elements = D::length_from_bytes(FLIT_BYTES);
55    let Tuple2(out_packet_outer, out_packet_inner) = OutPacket::to_value().factorize().split_at(flit_elements);
56    let out_packet_inner = out_packet_inner.normalize();
57    let expected_packet = Packet::to_value().factorize();
58    assert_eq!(
59        out_packet_inner, expected_packet,
60        "`contract_outer` packet mismatch: inner flit of OutPacket != Packet: {out_packet_inner} != {expected_packet}",
61    );
62
63    // Time must equal OutTime * outer flit portion of OutPacket.
64    // Padding is stripped for the collect_flits = 1 case.
65    let expected_time = OutTime::to_value()
66        .factorize()
67        .mul(out_packet_outer.remove_padding())
68        .normalize();
69    let input_time = Time::to_value().factorize();
70
71    let tiling_size = expected_time.size() / input_time.size();
72    let division_terms = expected_time
73        .divide(input_time.clone())
74        .exact_checked()
75        .expect("`contract_outer`: Time does not divide OutTime")
76        .division_terms;
77
78    // Non-broadcast axes must follow the same order in both mappings.
79    assert!(
80        division_terms
81            .windows(2)
82            .all(|w| w[0].divisor_stride > w[1].divisor_stride),
83        "`contract_outer`: Time axes are reordered in OutTime"
84    );
85
86    // Broadcast axes are the innermost in `OutTime`.
87    assert!(
88        division_terms.iter().all(|d| d.dividend_stride >= tiling_size),
89        "`contract_outer`: tiling axes must be innermost in OutTime"
90    );
91}
