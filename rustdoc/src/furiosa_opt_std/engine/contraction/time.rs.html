14// ANCHOR: contract_time_def
15impl<'l, const T: Tu, D: Scalar, Chip: M, Cluster: M, Slice: M, Lane: M, Time: M, Packet: M, B: Backend>
16    ContractPacketTensor<'l, T, D, Chip, Cluster, Slice, Lane, Time, Packet, B>
17{
18    /// Accumulates per-cycle contractions over the `Time` dimension via the shared
19    /// accumulator buffer, shrinking input `Time` to `OutTime`. The axes present in
20    /// `Time` but absent from `OutTime` are reduce-added.
21    #[primitive(ContractPacketTensor::contract_time)]
22    pub fn contract_time<OutTime: M>(
23        self,
24    ) -> ContractTimeTensor<'l, T, D, Chip, Cluster, Slice, Lane, OutTime, Packet, B> {
25        verify_contract_time::<Time, OutTime>();
26        ContractTimeTensor {
27            ctx: self.ctx,
28            inner: self.inner.reduce_add(),
29            pre_reduce_time: Time::to_value().factorize(),
30        }
31    }
32}

35/// Validates the `OutTime ⊆ Time` constraint for `.contract_time()`.
36///
37/// The axes in `OutTime` must be a subset of the axes in `Time`, with their
38/// relative order preserved. Axes present in `Time` but absent from `OutTime`
39/// are summed away by `reduce_add`. Retained axes must also preserve their
40/// padding from `Time`.
41pub(crate) fn verify_contract_time<Time: M, OutTime: M>() {
42    let time = Time::to_value().factorize();
43    let out_time = OutTime::to_value().factorize();
44
45    // The outer portion of `Time` should divide `Time`.
46    // Some axes can be reduced in temporal accumulation.
47    let division = time.clone().divide(out_time.clone()).exact_checked().unwrap_or_else(|_| {
48        panic!(
49            "contract_time: OutTime mismatch. Some axes present in Time are not present in OutTime: {time}, {out_time}"
50        )
51    });
52    let division_terms = &division.division_terms;
53    // Non-reduced axes must have their order preserved in `OutTime`.
54    assert!(
55        division_terms
56            .windows(2)
57            .all(|w| w[0].divisor_stride > w[1].divisor_stride),
58        "contract_time: OutTime axes must follow the same order as the Time axes"
59    );
60
61    // Each retained axis in `out_time` must preserve its padding from `time`.
62    // We store `padding_size / stride` per term (always exact since padding
63    // aligns to stride boundaries) and verify that the stride boundaries between
64    // consecutive retained axes in `out_time` match the gaps produced by
65    // padding in `time`.
66    let mut time_padding_per_stride: HashMap<usize, usize> = HashMap::new();
67    let factors = time.factors();
68    let mut stride = 1;
69    for (i, factor) in factors.iter().enumerate() {
70        match factor {
71            Factor::Term { resize, .. } => {
72                time_padding_per_stride.insert(
73                    stride,
74                    if let Some(Factor::Padding { size, .. }) = factors.get(i + 1) {
75                        size / stride
76                    } else {
77                        *resize
78                    },
79                );
80                stride *= resize;
81            }
82            Factor::Padding { size, .. } => {
83                stride = *size;
84            }
85        }
86    }
87
88    // Sort retained axes inner-to-outer by their position in `out_time` (divisor).
89    let mut sorted_divisions: Vec<&DivisionTerm> = division_terms.iter().collect();
90    sorted_divisions.sort_by_key(|d| d.divisor_stride);
91
92    // The first divisor should have a stride of 1.
93    if let Some(first) = sorted_divisions.first() {
94        assert_eq!(
95            first.divisor_stride, 1,
96            "contract_time: Padding mismatch. \
97             OutTime {out_time} has unexpected leading padding not present in Time {time}"
98        );
99    }
100
101    // For each retained axis, its padding end must equal the start of the next retained axis.
102    for (pos, d) in sorted_divisions.iter().enumerate() {
103        let expected_end = d.divisor_stride
104            * time_padding_per_stride
105                .get(&d.dividend_stride)
106                .copied()
107                .unwrap_or(d.resize);
108        let end = sorted_divisions
109            .get(pos + 1)
110            .map_or(out_time.size(), |next| next.divisor_stride);
111        assert_eq!(
112            expected_end, end,
113            "contract_time: Padding mismatch. \
114             Non-reduced axes in OutTime {out_time} do not preserve padding from Time {time}"
115        );
116    }
117}
