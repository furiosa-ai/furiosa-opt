1//! Collect Engine: packet normalization to flit-sized chunks.
2//!
3//! Normalizes a `FetchTensor` or `SwitchTensor` packet to exactly one flit
4//! (`FLIT_BYTES`). Pads to flit-aligned boundary, then splits: inner 32 bytes
5//! become `Packet2`, outer flit portion is absorbed into `Time2`.
6//!
7//! Also home to the terminating moves from `CollectTensor`:
8//! - `to_trf`: store to the Tensor Register File.
9//! - `to_vrf`: store to the Vector Register File.
10

29/// Tensor after collect engine: packet is exactly 32 bytes (one flit).
30pub type CollectTensor<'l, const T: Tu, D, Chip, Cluster, Slice, Time, Packet, B = CurrentBackend> =

33// ANCHOR: collect_impl
34impl<'l, const T: Tu, P: CanApplyCollect, D: Scalar, Chip: M, Cluster: M, Slice: M, Time: M, Packet: M, B: Backend>
35    TuTensor<'l, T, P, D, Chip, Cluster, Slice, Time, Packet, B>
36{
37    /// Normalizes packet to exactly 32 bytes (one flit).
38    ///
39    /// Pads to flit-aligned boundary, then splits: inner 32 bytes become
40    /// `Packet2`, outer flit portion is absorbed into `Time2`. For packets
41    /// already ≤ 32 bytes, only padding is added.
42    #[primitive(TuTensor::collect)]
43    pub fn collect<Time2: M, Packet2: M>(self) -> CollectTensor<'l, T, D, Chip, Cluster, Slice, Time2, Packet2, B> {
44        verify_collect::<D, Time, Packet, Time2, Packet2>();
45        CollectTensor::new(self.ctx, self.inner.transpose(false))
46    }
47}

52    TuTensor<'l, T, P, D, Chip, Cluster, Slice, Time, Packet, B>
53{
54    /// Stores to the tensor register file.
55    #[primitive(TuTensor::to_trf)]
56    pub fn to_trf<Lane: M, Element: M>(
57        self,
58        address: TrfAddress,
59    ) -> TrfTensor<D, Chip, Cluster, Slice, Lane, Element, B> {
60        verify_to_trf::<D, Lane, Time, Packet, Element>(&address);
61        TrfTensor::new(self.inner.transpose(false), address)
62    }
63}

78/// Validates collect engine constraints: normalizes packet to exactly one flit (32 bytes).
79///
80/// Pads the input packet to flit-aligned boundary, then splits:
81/// - Inner 32 bytes → Packet2 (one flit)
82/// - Outer flit portion → absorbed into Time2
83///
84/// For packets already ≤ 32 bytes, only padding is added.
85pub(crate) fn verify_collect<D: Scalar, Time: M, Packet: M, Time2: M, Packet2: M>() {
86    let in_packet_bytes = D::size_in_bytes_from_length(Packet::SIZE);
87    let aligned_bytes = align_up(in_packet_bytes, FLIT_BYTES);
88    let flit_elements = D::length_from_bytes(FLIT_BYTES);
89
90    // Output packet must be exactly one flit.
91    assert_eq!(
92        D::size_in_bytes_from_length(Packet2::SIZE),
93        FLIT_BYTES,
94        "Collect output packet must be exactly {FLIT_BYTES} bytes (one flit)."
95    );
96
97    // Pad input packet to flit-aligned boundary, then split at flit boundary.
98    let in_factorized = Packet::to_value().factorize();
99    let padded = in_factorized.pad(D::length_from_bytes(aligned_bytes));
100    let Tuple2(in_outer, in_flit) = padded.split_at(flit_elements);
101
102    // Output packet = inner flit.
103    let expected_packet = in_flit.normalize();
104    let out_packet = Packet2::to_value().factorize();
105    assert_eq!(
106        expected_packet, out_packet,
107        "Collect packet mismatch. Expected: {expected_packet}, got: {out_packet}"
108    );
109
110    // Time2 = Time × outer flit portion.
111    let expected_time = Time::to_value().factorize().mul(in_outer).normalize();
112    let out_time = Time2::to_value().factorize();
113    assert_eq!(
114        expected_time, out_time,
115        "Collect time mismatch. Expected: {expected_time}, got: {out_time}"
116    );
117}

119/// Validates `to_trf` constraints: reshapes `[Time, Packet]` into `[Lane, Element]`.
120///
121/// - `Lane::SIZE` must be 1, 2, 4, or 8.
122/// - Total bytes `Lane::SIZE * Element::SIZE * sizeof(D)` must fit in the chosen TRF region.
123/// - `Lane::SIZE` must divide `Time::SIZE`, with the outer factors of `Time` equal to `Lane`.
124/// - The remaining inner factors of `Time` concatenated with `Packet` must equal `Element`.
125pub(crate) fn verify_to_trf<D: Scalar, Lane: M, Time: M, Packet: M, Element: M>(address: &TrfAddress) {
126    assert!(
127        [1, 2, 4, 8].contains(&Lane::SIZE),
128        "Lane::SIZE must be 1, 2, 4, or 8, got {}",
129        Lane::SIZE
130    );
131
132    // Trf data should fit in the register file.
133    let capacity = address.capacity();
134    let total_trf_bytes = D::size_in_bytes_from_length(Lane::SIZE * Element::SIZE);
135    assert!(
136        total_trf_bytes <= capacity,
137        "TRF data ({} bytes = {} lanes x {} bytes) exceeds register file capacity ({} bytes for {})",
138        total_trf_bytes,
139        Lane::SIZE,
140        D::size_in_bytes_from_length(Element::SIZE),
141        capacity,
142        address,
143    );
144
145    // [time_outer] = [Lane]
146    let time = Time::to_value().factorize();
147    let Tuple2(time_outer, time_inner) = time.split_at(exact_div(Time::SIZE, Lane::SIZE).unwrap_or_else(|| {
148        panic!(
149            "Lane::SIZE ({}) does not divide Time::SIZE ({})",
150            Lane::SIZE,
151            Time::SIZE
152        )
153    }));
154    let lane = Lane::to_value().factorize();
155    assert_eq!(
156        time_outer, lane,
157        "`to_trf` lane mismatch: time_outer != Lane: {time_outer} != {lane}",
158    );
159
160    // [time_inner, Packet] = [Element]
161    let expected_element = time_inner.mul(Packet::to_value().factorize()).normalize();
162    let element = Element::to_value().factorize();
163    assert_eq!(
164        expected_element, element,
165        "`to_trf` element mismatch: [time_inner, Packet] != Element: {expected_element} != {element}",
166    );
167}
