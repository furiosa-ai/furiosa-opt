58/// Verify vector_narrow_clip: Packet (size 8) → Packet2 (size 4, front half only).
59///
60/// Checks that the back 4 of Packet are dummy,
61/// and that Packet2 matches the front 4 of Packet.
62pub(crate) fn verify_vector_narrow_clip<Packet: M, Packet2: M>() {
63    assert_eq!(
64        Packet::SIZE,
65        8,
66        "vector_narrow_clip: input Packet must have 8 elements (one flit), got {}. \
67         vector_narrow_clip is used to strip the back-4 dummy lanes before float operations. \
68         If Packet is already 4, you don't need vector_narrow_clip.",
69        Packet::SIZE,
70    );
71    let Tuple2(packet_outer, packet_inner) = Packet::to_value().factorize().split_at(4);
72    // Back 4 must be dummy — i.e., the outer half must be [1 # 2] (uninit padding).
73    // If this fails, the back 4 lanes contain real data and you need vector_narrow_split() instead.
74    assert_eq!(
75        packet_outer.clone().normalize(),
76        <m![1 # 2]>::to_value().factorize(),
77        "vector_narrow_clip: the back 4 lanes of the packet must be dummy (padding), \
78         but got: {packet_outer}. \
79         If the back 4 lanes contain real data, use vector_narrow_split() instead of vector_narrow_clip()."
80    );
81    // Output must be the front 4
82    assert_eq!(
83        Packet2::SIZE,
84        4,
85        "vector_narrow_clip: output Packet2 must have 4 elements, got {}.",
86        Packet2::SIZE,
87    );
88    assert_eq!(
89        packet_inner,
90        Packet2::to_value().factorize(),
91        "vector_narrow_clip: Packet2 must match the front 4 of Packet. \
92         Expected: {packet_inner}, got: {}.",
93        Packet2::to_value().factorize(),
94    );
95}
