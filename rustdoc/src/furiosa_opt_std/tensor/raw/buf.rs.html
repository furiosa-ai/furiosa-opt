8/// Buf raw tensor: native `Vec<D>` host staging buffer, shared by `Npu` and `Emulation`.
9///
10/// Holds a flat `Vec<D>` that serves as the host staging area for DMA to/from device memory.
11/// Under Npu the device kernel produces the values, so `BufRawTensor`'s value-touching
12/// [`RawTensor`] methods are not reached at runtime; their bodies are `todo!()` placeholders
13/// labelled "buffer semantics not implemented yet" — the Emulation backend (host-side
14/// buffer interpreter) will eventually supply real implementations.
15///
16/// TODO(buffer-axes-consistency): `axes` here is derived via `gen_axes::<Mapping>()` which
17/// returns canonical-order axes regardless of `Mapping`'s syntactic order. `data` is stored in
18/// caller-supplied buffer order (= `Mapping`-order). The two fields therefore do not describe
19/// the same layout when `Mapping`'s order differs from canonical. The inconsistency is dormant
20/// today (`read_index` / `write_index` are `todo!()`, so no axis-vs-data lookup actually
21/// happens), but a real Emulation interpreter will need this resolved — likely by replacing
22/// `axes` with `mapping: furiosa_mapping::Mapping` so the storage carries its own buffer
23/// shape. `MathRawTensor` is unaffected because its storage is in canonical math layout
24/// regardless of the boundary `Mapping`.
25#[derive(Debug, Clone, PartialEq, Eq)]
26#[doc(hidden)]
27pub struct BufRawTensor<D: Scalar> {
28    pub(crate) axes: Vec<Term>,
29    pub(crate) data: Vec<D>,
30}
31
32impl<D: Scalar> RawTensor<D> for BufRawTensor<D> {
33    fn axes(&self) -> &[Term] {
34        &self.axes
35    }
36
37    fn uninit_from_axes(axes: Vec<Term>) -> Self {
38        let len = shape_from_axes(&axes).iter().product::<usize>();
39        // SAFETY: zero-init may be incorrect for some `D`, but every concrete scalar in this crate
40        // is plain-old-data; the buffer is only ever read after a DMA write fills it, never while
41        // it still contains the placeholder bytes.
42        let mut data: Vec<D> = Vec::with_capacity(len);
43        // Use `resize_with` so we don't require `D: Default`.
44        data.resize_with(len, || unsafe { std::mem::zeroed() });
45        Self { axes, data }
46    }
47
48    fn read_index(&self, index: Index) -> Opt<D> {
49        match self.linear_index(index) {
50            Some(linear) => Opt::Init(self.data[linear]),
51            None => Opt::Uninit,
52        }
53    }
54
55    fn write_index(&mut self, index: Index, value: Opt<D>) {
56        let Opt::Init(v) = value else { return };
57        let Some(linear) = self.linear_index(index) else {
58            return;
59        };
60        self.data[linear] = v;
61    }
62
63    fn from_buf<Mapping: M>(data: impl IntoIterator<Item = D>) -> Self {
64        let axes = gen_axes::<Mapping>();
65        let data: Vec<D> = data.into_iter().collect();
66        let expected = shape_from_axes(&axes).iter().product::<usize>();
67        assert_eq!(expected, data.len(), "shape mismatch");
68        Self { axes, data }
69    }
70
71    fn to_buf<Mapping: M>(&self) -> Vec<D> {
72        self.data.clone()
73    }

163impl<D: Scalar> BufRawTensor<D> {
164    /// Resolve `index` to a row-major offset into `self.data`, or `None` if the
165    /// index falls outside the tensor's axes.
166    fn linear_index(&self, index: Index) -> Option<usize> {
167        let coords = finalize_coords(&self.axes, index)?;
168        let shape = shape_from_axes(&self.axes);
169        Some(coords.iter().zip(shape.iter()).fold(0usize, |acc, (c, &s)| acc * s + c))
170    }
171}
