13/// Phantom raw tensor (Typecheck): metadata only (axes), no host buffer. All reads return
14/// `Opt::Uninit` and all writes are silently dropped.
15///
16/// Each [`RawTensor`] method runs the relevant `op_prep` helper for its panic side
17/// (so mapping errors still surface under Typecheck) and then returns a fresh phantom tensor —
18/// no per-element iteration. The Backend trait already routes value-producing ops through its
19/// own typecheck overrides; the no-op writes only matter when user code reaches `RawTensor`
20/// directly through `Tensor` inherent methods (e.g. `Tensor::reshape`, `view.write_transpose`).
21#[derive(Debug, Clone, PartialEq, Eq)]
22#[doc(hidden)]
23pub struct PhantomRawTensor<D: Scalar> {
24    axes: Vec<Term>,
25    _phantom: PhantomData<D>,
26}
27
28impl<D: Scalar> RawTensor<D> for PhantomRawTensor<D> {
29    fn axes(&self) -> &[Term] {
30        &self.axes
31    }
32
33    fn uninit_from_axes(axes: Vec<Term>) -> Self {
34        Self {
35            axes,
36            _phantom: PhantomData,
37        }
38    }
39
40    fn read_index(&self, _index: Index) -> Opt<D> {
41        Opt::Uninit
42    }
43
44    fn write_index(&mut self, _index: Index, _value: Opt<D>) {}
45
46    fn from_buf<Mapping: M>(_data: impl IntoIterator<Item = D>) -> Self {
47        Self::uninit_from_axes(gen_axes::<Mapping>())
48    }
49
50    /// Typecheck has no values — the input buffer is dropped on the floor regardless of its
51    /// length, so skip the trait's default length check (which would fail for any caller writing
52    /// `Tensor::<_, M, Typecheck>::try_from_buf(vec![])` to construct a phantom tensor without
53    /// caring about size). Mapping errors still surface elsewhere via the Typecheck Backend
54    /// overrides; this method's job is only buffer ingestion.
55    fn try_from_buf<Mapping: M>(data: impl IntoIterator<Item = D>) -> Result<Self, BufferConvertError> {
56        Ok(Self::from_buf::<Mapping>(data))
57    }
58
59    fn to_buf<Mapping: M>(&self) -> Vec<D> {
60        Vec::new()
61    }
62
63    fn from_fn<F>(axes: Vec<Term>, _f: F) -> Self
64    where
65        F: FnMut(&Vec<Term>, &IxDyn) -> Opt<D>,
66    {
67        Self::uninit_from_axes(axes)
68    }
69
70    fn map<D2: Scalar, Output: RawTensor<D2>, F>(&self, _f: F) -> Output
71    where
72        F: FnMut(&Opt<D>) -> Opt<D2>,
73    {
74        Output::uninit_from_axes(self.axes.clone())
75    }
76
77    fn reduce<Src: M, Dst: M, Reduce>(&self, _reduce_fn: Reduce, _identity: Opt<D>) -> Self
78    where
79        Reduce: Fn(Opt<D>, Opt<D>) -> Opt<D>,
80    {
81        // Run the structural check for parity with `MathRawTensor`, then return the Dst-shaped
82        // axes (no values to compute).
83        let _ = Src::to_value()
84            .divide(&Dst::to_value())
85            .exact_checked()
86            .unwrap_or_else(|e| {
87                panic!(
88                    "[reduce] Dst is not a factor of Src: {e:?}\n\
89                     Src: {:?}\n\
90                     Dst: {:?}",
91                    Src::to_value(),
92                    Dst::to_value()
93                )
94            });
95        Self::uninit_from_axes(gen_axes::<Dst>())
96    }
97
98    fn reduce_then_broadcast<Src: M, Dst: M, Reduce>(&self, _reduce_fn: Reduce, _identity: Opt<D>) -> Self
99    where
100        Reduce: Fn(Opt<D>, Opt<D>) -> Opt<D>,
101    {
102        let dst_axes = gen_axes::<Dst>();
103        // Filter own axes against Dst (mirrors `reduce`'s axis projection) so the broadcast
104        // residue is computed against the same shape Math would see.
105        let reduced_axes: Vec<Term> = self
106            .axes
107            .iter()
108            .filter(|axis| dst_axes.contains(axis))
109            .cloned()
110            .collect();
111        let _ = reduce_broadcast(&reduced_axes, &dst_axes);
112        Self::uninit_from_axes(dst_axes)
113    }
114
115    fn reshape<Mapping: M, Mapping2: M>(self) -> Self {
116        assert_eq!(Mapping::SIZE, Mapping2::SIZE);
117        if Mapping::to_value() == Mapping2::to_value() {
118            self
119        } else {
120            Self::uninit_from_axes(gen_axes::<Mapping2>())
121        }
122    }
123
124    fn write_transpose<Src: M, Dst: M>(
125        &mut self,
126        _src: &Self,
127        _src_offset: &Index,
128        _dst_offset: &Index,
129        allow_broadcast: bool,
130    ) {
131        let _ = transpose_broadcast::<Src, Dst>(allow_broadcast);
132    }
133
134    fn zip_with<D2, D3, Other, Output, F>(&self, rhs: &Other, _f: F) -> Output
135    where
136        D2: Scalar,
137        D3: Scalar,
138        Other: RawTensor<D2>,
139        Output: RawTensor<D3>,
140        F: Fn(Opt<D>, Opt<D2>) -> Opt<D3>,
141    {
142        assert_zip(self.axes(), rhs.axes());
143        Output::uninit_from_axes(self.axes.clone())
144    }
145
146    fn write_scatter<Src, Key, Dst, Idx, IdxRaw>(&self, _dst: &mut Self, _index: &IdxRaw, _scaled: bool)
147    where
148        Src: M,
149        Key: M,
150        Dst: M,
151        Idx: M,
152        IdxRaw: RawTensor<i32>,
153    {
154        let src_fmapping = Src::to_value().factorize();
155        let dst_fmapping = Dst::to_value().factorize();
156        let key = Key::to_value().factorize();
157        let _ = scatter_params(&src_fmapping, &dst_fmapping, &key);
158    }
159
160    fn write_gather<Src, Dst, Idx, IdxRaw>(&self, _dst: &mut Self, _index: &IdxRaw, _scaled: bool)
161    where
162        Src: M,
163        Dst: M,
164        Idx: M,
165        IdxRaw: RawTensor<i32>,
166    {
167        let src_fmapping = Src::to_value().factorize();
168        let dst_fmapping = Dst::to_value().factorize();
169        let idx_fmapping = Idx::to_value().factorize();
170        let _ = gather_params(&src_fmapping, &dst_fmapping, &idx_fmapping);
171    }
172
173    /// Phantom skips the per-position iteration entirely — Typecheck has no values to update,
174    /// and the closure body's `read_index` would only ever return `Opt::Uninit`. Returns a
175    /// fresh phantom tensor.
176    fn apply_branch_operands<Mapping, Operand, TagRaw, F>(
177        &self,
178        _tag: &TagRaw,
179        _operands: &[Operand],
180        _update: F,
181    ) -> Self
182    where
183        D: VeScalar,
184        Mapping: M,
185        TagRaw: RawTensor<u8>,
186        Operand: OperandTag<D, Mapping>,
187        F: FnMut(&Index, &Operand, &mut Self),
188    {
189        self.clone()
190    }
191}

193impl<D: Scalar> RawTensorOpt<D> for PhantomRawTensor<D> {
194    fn from_opt_buf<Mapping: M>(_data: impl IntoIterator<Item = Opt<D>>) -> Self {
195        Self::uninit_from_axes(gen_axes::<Mapping>())
196    }
197
198    /// Like [`RawTensor::try_from_buf`], skip the length check: Typecheck callers may
199    /// intentionally pass a buffer of any size to construct a phantom tensor.
200    fn try_from_opt_buf<Mapping: M>(data: impl IntoIterator<Item = Opt<D>>) -> Result<Self, BufferConvertError> {
201        Ok(Self::from_opt_buf::<Mapping>(data))
202    }
203
204    fn to_opt_buf<Mapping: M>(&self) -> Vec<Opt<D>> {
205        Vec::new()
206    }
