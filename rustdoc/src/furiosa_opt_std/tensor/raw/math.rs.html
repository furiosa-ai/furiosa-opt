12/// Math raw tensor (Simulation): `ArrayD<Opt<D>>` host buffer with full ALU semantics.
13///
14/// `data` only stores the math (non-padding) values: its shape is the product of the axes'
15/// moduli. Buffers in physical layout order (length = `Mapping::SIZE`, includes padding) are
16/// loaded position-by-position via [`Self::write_index`], which silently no-ops on padding
17/// positions. Constructors that want bulk-load from such a buffer should iterate
18/// `Index::new().gen_indexes(Mapping::to_value().factorize())` and call `write_index`.
19#[derive(Debug, Clone, PartialEq, Eq)]
20#[doc(hidden)]
21pub struct MathRawTensor<D: Scalar> {
22    axes: Vec<Term>,
23    data: ArrayD<Opt<D>>,
24}
25
26impl<D: Scalar> RawTensor<D> for MathRawTensor<D> {
27    fn axes(&self) -> &[Term] {
28        &self.axes
29    }
30
31    fn uninit_from_axes(axes: Vec<Term>) -> Self {
32        let shape = shape_from_axes(&axes);
33        let data = ArrayD::from_elem(IxDyn(&shape), Opt::Uninit);
34        Self { axes, data }
35    }
36
37    fn read_index(&self, index: Index) -> Opt<D> {
38        let Some(coords) = finalize_coords(&self.axes, index) else {
39            return Opt::Uninit;
40        };
41        *self.data.get(coords.as_slice()).expect("Index out of bounds.")
42    }
43
44    fn write_index(&mut self, index: Index, value: Opt<D>) {
45        let Some(coords) = finalize_coords(&self.axes, index) else {
46            return;
47        };
48        *self.data.get_mut(coords.as_slice()).expect("Index out of bounds.") = value;
49    }
50
51    fn from_buf<Mapping: M>(data: impl IntoIterator<Item = D>) -> Self {
52        let fmapping = Mapping::to_value().factorize();
53        let mut tensor = Self::uninit_from_axes(gen_axes::<Mapping>());
54        for (i, value) in data.into_iter().enumerate() {
55            tensor.write_index(fmapping.eval(i), Opt::Init(value));
56        }
57        tensor
58    }
59
60    fn to_buf<Mapping: M>(&self) -> Vec<D> {
61        let fmapping = Mapping::to_value().factorize();
62        (0..Mapping::SIZE)
63            .map(|i| match self.read_index(fmapping.eval(i)) {
64                Opt::Init(value) => value,
65                Opt::Uninit => panic!(
66                    "MathRawTensor::to_buf called on a tensor containing Opt::Uninit slots; \
67                     use the logical Opt-buffer view instead."
68                ),
69            })
70            .collect()
71    }
72
73    fn map<D2: Scalar, Output: RawTensor<D2>, F>(&self, mut f: F) -> Output
74    where
75        F: FnMut(&Opt<D>) -> Opt<D2>,
76    {
77        let axes = self.axes.to_vec();
78        let mut output = Output::uninit_from_axes(axes.clone());
79        for index in Index::new().gen_indexes(FMapping::from_axes(RSlice::from(axes.as_slice()))) {
80            let value = self.read_index(index.clone());
81            output.write_index(index, f(&value));
82        }
83        output
84    }
85
86    fn reduce<Src: M, Dst: M, Reduce>(&self, reduce_fn: Reduce, identity: Opt<D>) -> Self
87    where
88        Reduce: Fn(Opt<D>, Opt<D>) -> Opt<D>,
89    {
90        // Reduce residue = Src - Dst (per-factor algebra), derived structurally from the source
91        // mapping so partial-axis reductions survive `gen_axes` consolidation.
92        let reduce_residue = Src::to_value()
93            .divide(&Dst::to_value())
94            .exact_checked()
95            .unwrap_or_else(|e| {
96                panic!(
97                    "[reduce] Dst is not a factor of Src: {e:?}\n\
98                     Src: {:?}\n\
99                     Dst: {:?}",
100                    Src::to_value(),
101                    Dst::to_value()
102                )
103            })
104            .relaxed_residues()
105            .dividend_residue;
106        let mut output = Self::uninit_from_axes(gen_axes::<Dst>());
107        for dst_index in Index::new().gen_indexes(Dst::to_value().factorize()) {
108            let mut acc = identity;
109            for src_index in dst_index.clone().gen_indexes(reduce_residue.clone()) {
110                acc = reduce_fn(acc, self.read_index(src_index));
111            }
112            output.write_index(dst_index, acc);
113        }
114        output
115    }
116
117    fn reduce_then_broadcast<Src: M, Dst: M, Reduce>(&self, reduce_fn: Reduce, identity: Opt<D>) -> Self
118    where
119        Reduce: Fn(Opt<D>, Opt<D>) -> Opt<D>,
120    {
121        // `Src.divide(Dst)` decomposes the algebra; everything we need falls out as `remainder`s:
122        //   * remainder(Dividend) → factors to reduce  (Src − matched),
123        //   * remainder(Divisor)  → factors to broadcast (Dst − matched),
124        //   * Inter = Src.factorize() − reduce_residue, again via `remainder`.
125        // Going through `divide` rather than a `Term`-equality filter is what lets a single
126        // symbol's sub-factors line up correctly (e.g., `Src=K(1,128)` and `Dst=K(16,8)` where
127        // the K(1,16) sub-factor must be reduced and the K(16,8) sub-factor kept).
128        //
129        // No intermediate tensor: for each Inter position we accumulate from `self` over the
130        // reduce residue, then write the accumulator to every Dst position that shares this
131        // Inter (= broadcast residue × Inter).
132        let src_mapping = Src::to_value();
133        let division = src_mapping.divide(&Dst::to_value());
134        let reduce_residue = division
135            .remainder(furiosa_mapping::DivisionSide::Dividend)
136            .expect("[reduce_then_broadcast] reduce residue must be well-formed");
137        let broadcast_residue = division
138            .remainder(furiosa_mapping::DivisionSide::Divisor)
139            .expect("[reduce_then_broadcast] broadcast residue must be well-formed");
140        let inter_fmapping = src_mapping
141            .factorize()
142            .divide(reduce_residue.clone())
143            .remainder(furiosa_mapping::DivisionSide::Dividend)
144            .expect("[reduce_then_broadcast] inter = Src − reduce_residue must be well-formed");
145
146        let mut dst = Self::uninit_from_axes(gen_axes::<Dst>());
147        for inter_index in Index::new().gen_indexes(inter_fmapping) {
148            let mut acc = identity;
149            for src_index in inter_index.clone().gen_indexes(reduce_residue.clone()) {
150                acc = reduce_fn(acc, self.read_index(src_index));
151            }
152            for dst_index in inter_index.gen_indexes(broadcast_residue.clone()) {
153                dst.write_index(dst_index, acc);
154            }
155        }
156        dst
157    }
158
159    fn reshape<Mapping: M, Mapping2: M>(self) -> Self {
160        assert_eq!(Mapping::SIZE, Mapping2::SIZE);
161        Self::from_opt_buf::<Mapping2>(self.to_opt_buf::<Mapping>())
162    }

164    fn write_transpose<Src: M, Dst: M>(
165        &mut self,
166        src: &Self,
167        src_offset: &Index,
168        dst_offset: &Index,
169        allow_broadcast: bool,
170    ) {
171        let broadcast = transpose_broadcast::<Src, Dst>(allow_broadcast);
172        for index in Index::new().gen_indexes(Src::to_value().factorize()) {
173            let mut src_index = index.clone();
174            src_index.add(src_offset.clone());
175            let value = src.read_index(src_index);
176
177            let mut dst_index_base = index;
178            dst_index_base.add(dst_offset.clone());
179            for broadcast_index in dst_index_base.gen_indexes(broadcast.clone()) {
180                self.write_index(broadcast_index, value);
181            }
182        }
183    }

185    fn zip_with<D2, D3, Other, Output, F>(&self, rhs: &Other, f: F) -> Output
186    where
187        D2: Scalar,
188        D3: Scalar,
189        Other: RawTensor<D2>,
190        Output: RawTensor<D3>,
191        F: Fn(Opt<D>, Opt<D2>) -> Opt<D3>,
192    {
193        assert_zip(self.axes(), rhs.axes());
194        let axes = self.axes().to_vec();
195        let mut output = Output::uninit_from_axes(axes.clone());
196        for index in Index::new().gen_indexes(FMapping::from_axes(RSlice::from(axes.as_slice()))) {
197            let l = self.read_index(index.clone());
198            let r = rhs.read_index(index.clone());
199            output.write_index(index, f(l, r));
200        }
201        output
202    }

204    fn write_scatter<Src, Key, Dst, Idx, IdxRaw>(&self, dst: &mut Self, index: &IdxRaw, scaled: bool)
205    where
206        Src: M,
207        Key: M,
208        Dst: M,
209        Idx: M,
210        IdxRaw: RawTensor<i32>,
211    {
212        let src_fmapping = Src::to_value().factorize();
213        let dst_fmapping = Dst::to_value().factorize();
214        let key = Key::to_value().factorize();
215
216        let index_stride = if scaled {
217            let payload = src_fmapping
218                .clone()
219                .divide(key.clone())
220                .exact_checked()
221                .expect("Src must contain scatter key")
222                .relaxed_residues()
223                .dividend_residue;
224            payload.remove_padding().size() * std::mem::size_of::<D>()
225        } else {
226            1
227        };
228
229        let indices: Vec<usize> = (0..Idx::SIZE)
230            .map(|i| {
231                let mut idx = Index::new();
232                idx.add_mapping::<Idx>(i);
233                let opt = index.read_index(idx);
234                let Opt::Init(v) = opt else {
235                    panic!("Scatter index must be initialized")
236                };
237                usize::try_from(v).expect("Scatter index must be non-negative") / index_stride
238            })
239            .collect();
240
241        let (payload, dst_term) = scatter_params(&src_fmapping, &dst_fmapping, &key);
242        for payload_index in Index::new().gen_indexes(payload) {
243            for (key_pos, key_index) in Index::new().gen_indexes(key.clone()).into_iter().enumerate() {
244                let mut src_index = payload_index.clone();
245                src_index.add(key_index);
246                let value = self.read_index(src_index);
247
248                let mut dst_index = payload_index.clone();
249                dst_index.add_term(dst_term.clone(), indices[key_pos]);
250                dst.write_index(dst_index, value);
251            }
252        }
253    }

255    fn write_gather<Src, Dst, Idx, IdxRaw>(&self, dst: &mut Self, index: &IdxRaw, scaled: bool)
256    where
257        Src: M,
258        Dst: M,
259        Idx: M,
260        IdxRaw: RawTensor<i32>,
261    {
262        let src_fmapping = Src::to_value().factorize();
263        let dst_fmapping = Dst::to_value().factorize();
264        let idx_fmapping = Idx::to_value().factorize();
265
266        let params = gather_params(&src_fmapping, &dst_fmapping, &idx_fmapping);
267
268        let index_stride = if scaled {
269            params.payload.clone().remove_padding().size() * std::mem::size_of::<D>()
270        } else {
271            1
272        };
273
274        let indices: Vec<usize> = (0..Idx::SIZE)
275            .map(|i| {
276                let mut idx = Index::new();
277                idx.add_mapping::<Idx>(i);
278                let opt = index.read_index(idx);
279                let Opt::Init(v) = opt else {
280                    panic!("Gather index must be initialized")
281                };
282                usize::try_from(v).expect("Gather index must be non-negative") / index_stride
283            })
284            .collect();
285
286        for payload_index in Index::new().gen_indexes(params.payload) {
287            for (idx_pos, dst_iter_index) in Index::new()
288                .gen_indexes(params.idx_residue.clone())
289                .into_iter()
290                .enumerate()
291            {
292                let mut src_index = payload_index.clone();
293                src_index.add_term(params.src_term.clone(), indices[idx_pos]);
294                let value = self.read_index(src_index);
295
296                let mut dst_index = payload_index.clone();
297                dst_index.add(dst_iter_index);
298                dst.write_index(dst_index, value);
299            }
300        }
301    }

303    fn apply_branch_operands<Mapping, Operand, TagRaw, F>(
304        &self,
305        tag: &TagRaw,
306        operands: &[Operand],
307        mut update: F,
308    ) -> Self
309    where
310        D: VeScalar,
311        Mapping: M,
312        TagRaw: RawTensor<u8>,
313        Operand: OperandTag<D, Mapping>,
314        F: FnMut(&Index, &Operand, &mut Self),
315    {
316        let mut output = self.clone();
317        for index in Index::new().gen_indexes(Mapping::to_value().factorize()) {
318            let eid = tag.read_index(index.clone());
319            let Opt::Init(_) = eid else {
320                continue;
321            };
322            for operand in operands {
323                if !operand.tag_filter().matches(eid) {
324                    continue;
325                }
326                update(&index, operand, &mut output);
327            }
328        }
329        output
330    }

333impl<D: Scalar> RawTensorOpt<D> for MathRawTensor<D> {
334    fn from_opt_buf<Mapping: M>(data: impl IntoIterator<Item = Opt<D>>) -> Self {
335        // Physical-layout iteration: padding positions land at an index whose `finalize()`
336        // returns `RErr`, so `write_index` silently no-ops there. The corresponding `data[i]`
337        // value (typically `Opt::Uninit` for padding slots) is dropped, matching
338        // `MathRawTensor`'s padding-stripped `ArrayD` representation.
339        let fmapping = Mapping::to_value().factorize();
340        let mut tensor = Self::uninit_from_axes(gen_axes::<Mapping>());
341        for (i, value) in data.into_iter().enumerate() {
342            tensor.write_index(fmapping.eval(i), value);
343        }
344        tensor
345    }
346
347    fn to_opt_buf<Mapping: M>(&self) -> Vec<Opt<D>> {
348        let fmapping = Mapping::to_value().factorize();
349        (0..Mapping::SIZE).map(|i| self.read_index(fmapping.eval(i))).collect()
350    }
