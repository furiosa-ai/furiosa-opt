1//! Transpose examples.
2
3use furiosa_visa_std::prelude::*;
4
5type Chip = m![1];
6axes![A = 8, B = 16, C = 32];
7
8/// Device function that transposes a tensor from shape [A, B, C] to [C, A, B].
9/// it is divided into two steps for demonstration purposes.
10#[device(chip = 1)]
11pub fn transpose_simple(
12    ctx: &mut Context,
13    input: &HbmTensor<f32, Chip, m![A, B, C]>,
14) -> HbmTensor<f32, Chip, m![C, A, B]> {
15    // transpose: [A, B, C] -> [A, C, B]
16    let intermediate: HbmTensor<f32, Chip, m![A, C, B]> = input.to_hbm(&mut ctx.tdma, 0);
17
18    // transpose: [A, C, B] -> [C, A, B]
19    intermediate.to_hbm(&mut ctx.tdma, 0x1000)
20}
