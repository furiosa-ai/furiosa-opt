1pub mod simpl {
2    use furiosa_visa_std::prelude::*;
3
4    axes![A = 512, B = 4];
5
6    #[device(chip = 1)]
7    pub fn view_simpl(
8        ctx: &mut Context,
9        input_hbm: &HbmTensor<i32, m![1], m![A, B]>,
10    ) -> HbmTensor<i32, m![1], m![A, B]> {
11        // Transfer input tensor to DM with shape:
12        // - (A=512/256) on the cluster dimension,
13        // - (A=512%256) on the slice dimension,
14        // - (B=4) on the element dimension.
15        let input = input_hbm.to_dm::<m![A / 256], m![A % 256], m![B]>(&mut ctx.tdma, 0x2000);
16
17        // Allocate output tensor in DM with the same shape as input.
18        let mut output = unsafe { DmTensor::<i32, m![1], m![A / 256], m![A % 256], m![B]>::from_addr(0x3000) };
19
20        // Create views on input (B=0,1,2) and output (B=1,2,3), and copy data from input to output.
21        let input012 = input.view().tile::<m![B], 3, m![B = 3 # 4]>(0);
22        let output123 = output.view_mut().tile::<m![B], 3, m![B = 3 # 4]>(1);
23        input012.to_dm_view(&mut ctx.tdma, output123);
24
25        // Create views on input (B=3) and output (B=0), and copy data from input to output.
26        let input3 = input.view().tile::<m![B], 1, m![1 # 4]>(3);
27        let output0 = output.view_mut().tile::<m![B], 1, m![1 # 4]>(0);
28        input3.to_dm_view(&mut ctx.tdma, output0);
29
30        // Transfer output tensor back to HBM.
31        output.to_hbm(&mut ctx.tdma, 0x4000)
32    }
33}

35pub mod padding {
36    use furiosa_visa_std::prelude::*;
37
38    axes![A = 9, B = 7];
39
40    #[device(chip = 1)]
41    pub fn view_padding(
42        ctx: &mut Context,
43        input_hbm: &HbmTensor<i32, m![1], m![[A, B] # 64]>,
44    ) -> HbmTensor<i32, m![1], m![[A, B] # 64]> {
45        // Transfer input tensor to DM with shape:
46        // - 1 on the cluster dimension,
47        // - (A=9)(B=7)=64/16 on the slice dimension,
48        // - ((A=9)(B=7)=64%8) * ((A=9)(B=7)=64/8%2) on the element dimension.
49        let input = input_hbm
50            .to_dm::<m![1], m![[A, B] # 64 / 16], m![[A, B] # 64 % 8, [A, B] # 64 / 8 % 2]>(&mut ctx.tdma, 0x2000);
51
52        // Allocate output tensor in DM with the same shape as input.
53        let mut output = unsafe {
54            DmTensor::<i32, m![1], m![1], m![[A, B] # 64 / 16], m![[A, B] # 64 % 8, [A, B] # 64 / 8 % 2]>::from_addr(
55                0x3000,
56            )
57        };
58
59        // Create views on input (((A=9)(B=7)=64%4)=0,1,2) and output (((A=9)(B=7)=64%4)=1,2,3), and copy data from input to output.
60        let input012 = input
61            .view()
62            .tile::<m![[A, B] # 64 % 4], 3, m![[A, B] # 64 / 4 % 2, ([A, B] # 64 % 4) = 3 # 4, [A, B] # 64 / 8 % 2]>(0);
63        let output123 = output
64            .view_mut()
65            .tile::<m![[A, B] # 64 % 4], 3, m![[A, B] # 64 / 4 % 2, ([A, B] # 64 % 4) = 3 # 4, [A, B] # 64 / 8 % 2]>(1);
66        input012.to_dm_view(&mut ctx.tdma, output123);
67
68        // Create views on input (((A=9)(B=7)=64%4)=3) and output (((A=9)(B=7)=64%4)=0), and copy data from input to output.
69        let input3 = input
70            .view()
71            .tile::<m![[A, B] # 64 % 4], 1, m![[A, B] # 64 / 4 % 2, 1 # 4, [A, B] # 64 / 8 % 2]>(3);
72        let output0 = output
73            .view_mut()
74            .tile::<m![[A, B] # 64 % 4], 1, m![[A, B] # 64 / 4 % 2, 1 # 4, [A, B] # 64 / 8 % 2]>(0);
75        input3.to_dm_view(&mut ctx.tdma, output0);
76
77        // Transfer output tensor back to HBM.
78        output.to_hbm(&mut ctx.tdma, 0x4000)
79    }
