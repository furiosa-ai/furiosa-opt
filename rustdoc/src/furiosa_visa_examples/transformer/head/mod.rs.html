1//! Composable kernel: `transformer_-1:output_projection..output`
2//!
3//! Pipeline: O-Projection → Residual Add → RMS Norm → MLP (gate+up+SiLU+down)
4//!           → Residual Add → RMS Norm → LM Head (logits)
5//!
6//! Qwen 2.5 0.5B, 4 PE, W16A16KV16 (bf16), tokenwise seq_len=128.
7
8mod lm_head;
9
10use crate::transformer::common::{mlp, norm, o_proj};
11
12use crate::transformer::axes::{
13    H,             // = 896, hidden_size
14    M,             // = 4864, mlp_intermediate
15    S_decode as S, // = 128, sequence length
16    W_vocab as W,  // = 151936, vocab_size
17};
18use furiosa_visa_std::prelude::*;
19
20type Chip = m![1];
21
22#[device(chip = 1)]
23#[allow(clippy::too_many_arguments)]
24pub fn forward(
25    ctx: &mut Context,
26    // Inputs (arg0..arg8)
27    matmul_score_v: &HbmTensor<bf16, Chip, m![S, H]>,  // arg0
28    o_proj_weight: &HbmTensor<bf16, Chip, m![H, H]>,   // arg1
29    hidden_states: &HbmTensor<bf16, Chip, m![S, H]>,   // arg2
30    norm_weight_0: &HbmTensor<bf16, Chip, m![H]>,      // arg3
31    gate_weight: &HbmTensor<bf16, Chip, m![M, H]>,     // arg4
32    up_weight: &HbmTensor<bf16, Chip, m![M, H]>,       // arg5
33    down_weight: &HbmTensor<bf16, Chip, m![H, M]>,     // arg6
34    norm_weight_1: &HbmTensor<bf16, Chip, m![H]>,      // arg7
35    embedding_table: &HbmTensor<bf16, Chip, m![W, H]>, // arg8
36    // Output (out#0, pre-allocated by caller)
37    out_logits: &mut HbmTensor<bf16, Chip, m![S, W]>, // out#0
38) {
39    // PyTorch Qwen2ForCausalLM last layer + LM head:
40    //   (finish last Qwen2Decoder: o_proj → residual → MLP → residual)
41    //   hidden_states = norm(hidden_states)          # final RMSNorm
42    //   logits = lm_head(hidden_states)               # [S, W=151936]
43    //   (lm_head weight = embed_tokens weight when tie_word_embeddings=True)
44
45    // Phase 1: O-projection — output = o_proj(attn_output)  [bias=False]
46    let o_proj_out = o_proj::o_proj(ctx, matmul_score_v, o_proj_weight);
47    // Phase 2: residual + post_attention_layernorm (fused)
48    let (normalized_0, residual_0) = norm::residual_norm(ctx, hidden_states, &o_proj_out, norm_weight_0);
49    // Phase 3: MLP — silu(gate_proj(x)) * up_proj(x), then down_proj
50    let mlp_out = mlp::mlp(ctx, &normalized_0, gate_weight, up_weight, down_weight);
51    // Phase 4: residual + final RMSNorm (Qwen2Model.norm)
52    let normalized_1 = norm::final_norm(ctx, &residual_0, &mlp_out, norm_weight_1);
53    // Phase 5: LM Head — lm_head(hidden_states) → [S=128, W=151936]
54    //   Uses embedding_table as weight (tie_word_embeddings=True)
55    lm_head::lm_head(ctx, &normalized_1, embedding_table, out_logits);
56}
