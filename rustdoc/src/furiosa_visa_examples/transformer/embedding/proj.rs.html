1//! Q/K/V projection kernels used by transformer attention.
2//!
3//! This module implements the PyTorch linear projections:
4//! `q = x @ Wq + bq`, `k = x @ Wk + bk`, and `v = x @ Wv + bv`.
5//!
6//! The kernels load projection weights from HBM, reshape them for DPE contraction,
7//! run matmul with bias add, and write `[S, Q/K/V]` outputs back to HBM.
8//! V and K use the same projection pattern, while Q uses a reversed placement
9//! so the activation tile is loaded into TRF.
10
11use crate::transformer::axes::{
12    H,             // = 896, hidden_size = 14 * 64
13    K,             // = 128, kv_proj_size = N * D
14    Q,             // = 896, q_proj_size
15    S_decode as S, // = 128, sequence length
16    V,             // = 128, v_proj_size
17    Y,             // = 4, broadcast
18};
19use furiosa_visa_std::prelude::*;
20
21type Chip = m![1];
22type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
23
24/// V projection: normalized input × V_weight → out_v.
25pub(crate) fn v_proj(
26    ctx: &mut Context,
27    input: &DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]>,
28    weight: &HbmTensor<bf16, Chip, m![V, H]>,
29    out: &mut HbmTensor<bf16, Chip, m![S, V]>,
30) {
31    let weight_dm: DmTensor<bf16, Chip, Cluster, m![V / 8, H / 224, V / 2 % 4], m![V % 2, H % 224]> =
32        weight.to_dm(&mut ctx.tdma, 0x400);
33
34    // Reorder the weight tile so H blocks align with DPE contraction.
35    let weight_it: DmTensor<bf16, Chip, Cluster, m![V / 8, H / 224, H / 56 % 4], m![V % 8, H % 56 # 64]> = ctx
36        .main
37        .begin(weight_dm.view())
38        .fetch::<bf16, m![V % 2, H / 56 % 4], m![H % 56]>()
39        .switch::<m![V / 8, H / 224, H / 56 % 4], m![V / 2 % 4, V % 2]>(SwitchConfig::InterTranspose {
40            slice1: 4,
41            slice0: 1,
42            time0: 1,
43        })
44        .collect::<m![V / 2 % 4, V % 2], m![H % 56 # 64]>()
45        .commit(0x800);
46
47    // Load reordered V weight into TRF for the DPE matmul.
48    let weight_trf: TrfTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![V % 8], m![V / 8 % 4, H % 56 # 64]> = ctx
49        .sub
50        .begin(weight_it.view())
51        .fetch::<bf16, m![V % 8], m![H % 56]>()
52        .switch::<m![Y, S / 32, H / 56], m![V % 8]>(SwitchConfig::Broadcast1 { slice1: 4, slice0: 16 })
53        .collect::<m![V % 8], m![V / 8 % 4, H % 56 # 64]>()
54        .to_trf(TrfAddress::FirstHalf);
55
56    // Load the pre-tiled V bias buffer and stage it in VRF for fused bias add.
57    let v_bias_hbm: HbmTensor<f32, Chip, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2, V % 32]> =
58        unsafe { HbmTensor::from_addr(0xc000) };
59    let v_bias_dm: DmTensor<f32, Chip, Cluster, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![V % 32]> =
60        v_bias_hbm.to_dm(&mut ctx.tdma, 0x400);
61    let v_bias_vrf: VrfTensor<f32, Chip, Cluster, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![V % 32]> = ctx
62        .sub
63        .begin(v_bias_dm.view())
64        .fetch::<f32, m![1], m![V % 32]>()
65        .collect::<m![1], m![V % 32]>()
66        .to_vrf(0x0);
67
68    // Compute V = input @ V_weight and apply bias in the VE path.
69    let result: DmTensor<bf16, Chip, Cluster, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![S % 2, V % 32]> = ctx
70        .main
71        .begin(input.view())
72        .fetch::<bf16, m![S / 4 % 8, S / 2 % 2, S % 2], m![H / 8 % 7, H % 8]>()
73        .collect::<m![S / 4 % 8, S / 2 % 2, S % 2], m![H / 8 % 7, H % 8]>()
74        .align::<m![S / 4 % 8, S / 2 % 2, S % 2], m![H / 8 % 7, H % 8], _, _>(&weight_trf)
75        .contract::<m![1]>()
76        .accumulate::<m![S / 4 % 8, V / 8 % 4, S % 4], m![V % 8]>(AccumulationKind::Interleaved)
77        .vector_init()
78        .vector_intra_slice_branch(BranchMode::Unconditional)
79        .vector_clip(ClipBinaryOpF32::Add, &v_bias_vrf)
80        .vector_inter_slice_reduce::<m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![V / 8 % 4, S % 2]>(
81            InterSliceReduceOpF32::Add,
82        )
83        .vector_final()
84        .cast::<bf16, m![V % 8 # 16]>()
85        .commit(0x600);
86
87    let v_reassembled: DmTensor<bf16, Chip, Cluster, m![S / 2, Y], m![S % 2, V]> = ctx
88        .main
89        .begin(result.view())
90        .fetch::<bf16, m![S % 2], m![V % 32]>()
91        .switch::<m![S / 2, Y], m![S % 2]>(SwitchConfig::CustomBroadcast { ring_size: 256 })
92        .collect::<m![S % 2], m![V % 32]>()
93        .commit(0x800);
94
95    v_reassembled.view().to_hbm_view(&mut ctx.tdma, out.view_mut());
96}

98/// Q projection: normalized input × Q_weight + bias.
99///
100/// The Q path stages activations in TRF and streams Q weights from SRAM.
101pub(crate) fn q_proj(
102    ctx: &mut Context,
103    input: &DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]>,
104    weight: &HbmTensor<bf16, Chip, m![Q, H]>,
105    bias: &HbmTensor<bf16, Chip, m![Q]>,
106) -> HbmTensor<bf16, Chip, m![S, Q]> {
107    // Reshape Q bias from HBM layout into a VRF layout for fused add.
108
109    let q_bias_dm: DmTensor<bf16, Chip, Cluster, m![Q / 32 % 2, Q / 4 % 8, 1 # 16], m![Q / 64, Q % 4]> =
110        bias.to_dm(&mut ctx.tdma, 0x6600);
111
112    let q_bias_te: DmTensor<bf16, Chip, Cluster, m![Q / 32 % 2, Q / 4 % 8, 1 # 16], m![Q % 4, Q / 64 # 16]> = ctx
113        .main
114        .begin(q_bias_dm.view())
115        .fetch::<bf16, m![Q / 64], m![Q % 4]>()
116        .collect::<m![Q / 64], m![Q % 4]>()
117        .commit(0x500);
118
119    let q_bias_it: DmTensor<bf16, Chip, Cluster, m![Q / 32 % 2, Q / 4 % 8, Q % 4, 1 # 4], m![1 # 4, Q / 64 # 16]> = ctx
120        .main
121        .begin(q_bias_te.view())
122        .fetch::<bf16, m![Q % 4], m![Q / 64 # 16]>()
123        .switch::<m![Q / 32 % 2, Q / 4 % 8, Q % 4, 1 # 4], m![1 # 4]>(SwitchConfig::InterTranspose {
124            slice1: 4,
125            slice0: 4,
126            time0: 1,
127        })
128        .collect::<m![1 # 4], m![Q / 64 # 16]>()
129        .commit(0x600);
130
131    let q_bias_reassembled: DmTensor<bf16, Chip, Cluster, m![Q / 32 % 2, Q / 4 % 8, Q % 4, Y], m![Q / 64 # 16]> = ctx
132        .main
133        .begin(q_bias_it.view())
134        .fetch::<bf16, m![1], m![Q / 64 # 16]>()
135        .switch::<m![Q / 32 % 2, Q / 4 % 8, Q % 4, Y], m![1]>(SwitchConfig::CustomBroadcast { ring_size: 4 })
136        .collect::<m![1], m![Q / 64 # 16]>()
137        .commit(0x500);
138
139    // Reshape bias Slice to match the Q matmul pipeline's Slice.
140    let q_bias_reassembled: DmTensor<bf16, Chip, Cluster, m![Q % 2, Q / 2 % 2, Y, H / 56], m![Q / 64 # 16]> =
141        unsafe { q_bias_reassembled.reshape() };
142
143    let q_bias_vrf: VrfTensor<f32, Chip, Cluster, m![Q % 2, Q / 2 % 2, Y, H / 56], m![Q / 64 # 16]> = ctx
144        .sub
145        .begin(q_bias_reassembled.view())
146        .fetch::<f32, m![1], m![Q / 64 # 16]>()
147        .collect::<m![1], m![Q / 64 # 16]>()
148        .to_vrf(0x0);
149
150    // Pad the input packet axis for TRF-friendly alignment.
151    let input_padded: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56 # 64]> = ctx
152        .main
153        .begin(input.view())
154        .fetch::<bf16, m![S % 32], m![H % 56]>()
155        .collect::<m![S % 32], m![H % 56 # 64]>()
156        .commit(0x2100);
157
158    // Stage padded activations in TRF for Q projection.
159    let input_trf: TrfTensor<
160        bf16,
161        Chip,
162        Cluster,
163        m![Q % 2, Q / 2 % 2, Y, H / 56],
164        m![S % 8],
165        m![H / 8 % 7, S / 8 % 4, S % 8 # 32],
166    > = ctx
167        .sub
168        .begin(input_padded.view())
169        .fetch::<bf16, m![S % 8], m![H % 56]>()
170        .switch::<m![Q % 2, Q / 2 % 2, Y, H / 56], m![S % 8]>(SwitchConfig::Broadcast1 { slice1: 4, slice0: 16 })
171        .collect::<m![S % 8], m![H / 8 % 7, S / 8 % 4, S % 8 # 32]>()
172        .to_trf(TrfAddress::SecondHalf);
173
174    let weight_dm: DmTensor<
175        bf16,
176        Chip,
177        Cluster,
178        m![Q / 4 % 16, H / 224, Q / 448, Q / 2 % 2],
179        m![Q / 64, Q % 2, H % 224],
180    > = weight.to_dm(&mut ctx.tdma, 0x3100);
181
182    // Reorder Q weight blocks so H fragments align with contraction groups.
183    let weight_it: DmTensor<
184        bf16,
185        Chip,
186        Cluster,
187        m![Q / 4 % 16, H / 224, H / 112 % 2, H / 56 % 2],
188        m![Q / 64, Q % 2, Q / 448, Q / 2 % 2, H % 56 # 64],
189    > = ctx
190        .main
191        .begin(weight_dm.view())
192        .fetch::<bf16, m![Q % 2, Q / 448, Q / 2 % 2, H / 56 % 4], m![H % 56]>()
193        .switch::<m![Q / 4 % 16, H / 224, H / 112 % 2, H / 56 % 2], m![Q / 64, Q % 2, Q / 448, Q / 2 % 2]>(
194            SwitchConfig::InterTranspose {
195                slice1: 4,
196                slice0: 1,
197                time0: 1,
198            },
199        )
200        .collect::<m![Q / 64, Q % 2, Q / 448, Q / 2 % 2], m![H % 56 # 64]>()
201        .commit(0x4a00);
202
203    // Strip packet padding before DPE alignment.
204    let weight_stripped: DmTensor<
205        bf16,
206        Chip,
207        Cluster,
208        m![Q / 4 % 16, H / 224, H / 112 % 2, H / 56 % 2],
209        m![Q / 448, Q / 64, Q / 2 % 2, Q % 2, H % 56],
210    > = ctx
211        .main
212        .begin(weight_it.view())
213        .fetch::<bf16, m![Q / 448, Q / 64, Q / 2 % 2, Q % 2], m![H % 56]>()
214        .collect::<m![Q / 448, Q / 64, Q / 2 % 2, Q % 2], m![H % 56]>()
215        .commit(0x3100);
216
217    // Reshape weight Slice to match TRF Slice for alignment.
218    let weight_stripped: DmTensor<
219        bf16,
220        Chip,
221        Cluster,
222        m![Q % 2, Q / 2 % 2, Y, H / 56],
223        m![Q / 448, Q / 64, Q / 2 % 2, Q % 2, H % 56],
224    > = unsafe { weight_stripped.reshape() };
225
226    // Compute Q = input @ Q_weight and apply Q bias in the VE path.
227    let result: DmTensor<bf16, Chip, Cluster, m![Q % 2, Q / 2 % 32, S / 32], m![Q / 64, S % 32]> = ctx
228        .main
229        .begin(weight_stripped.view())
230        .fetch::<bf16, m![Q / 64, Q / 448, Q / 2 % 2, Q % 2], m![H / 8 % 7, H % 8]>()
231        .collect::<m![Q / 64, Q / 448, Q / 2 % 2, Q % 2], m![H / 8 % 7, H % 8]>()
232        .align::<m![Q / 64, Q / 448, Q / 2 % 2, Q % 2], m![H / 8 % 7, H % 8], _, _>(&input_trf)
233        .contract::<m![1]>()
234        .accumulate::<m![Q / 64, S / 8 % 4, Q / 448], m![S % 8]>(AccumulationKind::Interleaved)
235        .vector_init()
236        .vector_intra_slice_branch(BranchMode::Unconditional)
237        .vector_clip(ClipBinaryOpF32::Add, &q_bias_vrf)
238        .vector_inter_slice_reduce::<m![Q % 2, Q / 2 % 32, S / 32], m![Q / 64, S % 32]>(InterSliceReduceOpF32::Add)
239        .vector_final()
240        .cast::<bf16, m![S % 32]>()
241        .commit(0x0A00);
242
243    let result_d0b: DmTensor<bf16, Chip, Cluster, m![Q % 2, Q / 2 % 32, 1 # 4], m![Q / 64, S]> = ctx
244        .main
245        .begin(result.view())
246        .fetch::<bf16, m![Q / 64], m![S % 32]>()
247        .switch::<m![Q % 2, Q / 2 % 32, 1 # 4], m![Q / 64]>(SwitchConfig::Broadcast01 {
248            slice1: 4,
249            slice0: 1,
250            time0: 1,
251        })
252        .collect::<m![Q / 64], m![S % 32]>()
253        .commit(0x2100);
254
255    result_d0b.to_hbm(&mut ctx.tdma, 0x10e36000)
256}

258/// K projection: normalized input × K_weight → K in HBM.
259pub(crate) fn k_proj(
260    ctx: &mut Context,
261    input: &DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]>,
262    weight: &HbmTensor<bf16, Chip, m![K, H]>,
263) -> HbmTensor<bf16, Chip, m![S, K]> {
264    // Load the pre-tiled K bias buffer and stage it in VRF for fused bias add.
265    let k_bias_hbm: HbmTensor<f32, Chip, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2, K % 32]> =
266        unsafe { HbmTensor::from_addr(0x4000) };
267    let k_bias_dm: DmTensor<f32, Chip, Cluster, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![K % 32]> =
268        k_bias_hbm.to_dm(&mut ctx.tdma, 0x400);
269    let k_bias_vrf: VrfTensor<f32, Chip, Cluster, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![K % 32]> = ctx
270        .sub
271        .begin(k_bias_dm.view())
272        .fetch::<f32, m![1], m![K % 32]>()
273        .collect::<m![1], m![K % 32]>()
274        .to_vrf(0x0);
275
276    let weight_dm: DmTensor<bf16, Chip, Cluster, m![K / 8, H / 224, K / 2 % 4], m![K % 2, H % 224]> =
277        weight.to_dm(&mut ctx.tdma, 0x0);
278
279    // Reorder the weight tile so H blocks align with DPE contraction.
280    let weight_it: DmTensor<bf16, Chip, Cluster, m![K / 8, H / 224, H / 56 % 4], m![K % 8, H % 56 # 64]> = ctx
281        .main
282        .begin(weight_dm.view())
283        .fetch::<bf16, m![K % 2, H / 56 % 4], m![H % 56]>()
284        .switch::<m![K / 8, H / 224, H / 56 % 4], m![K / 2 % 4, K % 2]>(SwitchConfig::InterTranspose {
285            slice1: 4,
286            slice0: 1,
287            time0: 1,
288        })
289        .collect::<m![K / 2 % 4, K % 2], m![H % 56 # 64]>()
290        .commit(0x0E00);
291
292    // Load reordered K weight into TRF for the DPE matmul.
293    let weight_trf: TrfTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![K % 8], m![K / 8 % 4, H % 56 # 64]> = ctx
294        .sub
295        .begin(weight_it.view())
296        .fetch::<bf16, m![K % 8], m![H % 56]>()
297        .switch::<m![Y, S / 32, H / 56], m![K % 8]>(SwitchConfig::Broadcast1 { slice1: 4, slice0: 16 })
298        .collect::<m![K % 8], m![K / 8 % 4, H % 56 # 64]>()
299        .to_trf(TrfAddress::FirstHalf);
300
301    // Compute K = input @ K_weight and apply bias in the VE path.
302    let result: DmTensor<bf16, Chip, Cluster, m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![S % 2, K % 32]> = ctx
303        .main
304        .begin(input.view())
305        .fetch::<bf16, m![S / 4 % 8, S / 2 % 2, S % 2], m![H / 8 % 7, H % 8]>()
306        .collect::<m![S / 4 % 8, S / 2 % 2, S % 2], m![H / 8 % 7, H % 8]>()
307        .align::<m![S / 4 % 8, S / 2 % 2, S % 2], m![H / 8 % 7, H % 8], _, _>(&weight_trf)
308        .contract::<m![1]>()
309        .accumulate::<m![S / 4 % 8, K / 8 % 4, S % 4], m![K % 8]>(AccumulationKind::Interleaved)
310        .vector_init()
311        .vector_intra_slice_branch(BranchMode::Unconditional)
312        .vector_clip(ClipBinaryOpF32::Add, &k_bias_vrf)
313        .vector_inter_slice_reduce::<m![1 # 2, 1 # 2, S / 32, S / 4 % 8, S / 2 % 2], m![K / 8 % 4, S % 2]>(
314            InterSliceReduceOpF32::Add,
315        )
316        .vector_final()
317        .cast::<bf16, m![K % 8 # 16]>()
318        .commit(0x0);
319
320    // Reorder K output slices into the final `[S, K]` layout.
321    let k_xpose: DmTensor<bf16, Chip, Cluster, m![S / 2, Y], m![S % 2, K % 32]> = ctx
322        .main
323        .begin(result.view())
324        .fetch::<bf16, m![S % 2], m![K % 32]>()
325        .switch::<m![S / 2, Y], m![S % 2]>(SwitchConfig::Transpose { slice1: 4, slice0: 64 })
326        .collect::<m![S % 2], m![K % 32]>()
327        .commit(0x100);
328
329    // DMA K result → DRAM (for RoPE)
330    k_xpose.to_hbm(&mut ctx.tdma, 0x8c800)
331}
