1//! Composable kernel: `embedding..transformer_0:qkv_projection`
2//!
3//! Pipeline: Embedding lookup -> RMS Norm -> Q/K/V Projection -> RoPE
4//!
5//! Qwen 2.5 0.5B, 4 PE, W16A16KV16 (bf16), tokenwise seq_len=128.
6
7#[allow(clippy::module_inception)]
8mod embedding;
9pub(crate) mod proj;
10mod rms_norm;
11use crate::transformer::common::rope;
12
13use crate::transformer::axes::{
14    D,             // = 64, head_dim
15    H,             // = 896, hidden_size
16    K,             // = 128, k_proj
17    P,             // = 32768, max_position
18    Q,             // = 896, q_proj
19    R,             // = 2, rope_rot
20    S_decode as S, // = 128, sequence length
21    V,             // = 128, v_proj
22    W_vocab as W,  // = 151936, vocab_size
23};
24use furiosa_visa_std::prelude::*;
25
26type Chip = m![1];
27
28#[device(chip = 1)]
29#[allow(clippy::too_many_arguments)]
30pub fn forward(
31    ctx: &mut Context,
32    // Inputs (arg0..arg8)
33    embedding_table: &HbmTensor<bf16, Chip, m![W, H]>,      // arg0
34    input_ids: &HbmTensor<i32, Chip, m![S]>,                // arg1
35    norm_weight: &HbmTensor<bf16, Chip, m![H]>,             // arg2
36    q_weight: &HbmTensor<bf16, Chip, m![Q, H]>,             // arg3
37    q_bias: &HbmTensor<bf16, Chip, m![Q]>,                  // arg4
38    k_weight: &HbmTensor<bf16, Chip, m![K, H]>,             // arg5
39    v_weight: &HbmTensor<bf16, Chip, m![V, H]>,             // arg6
40    rope_table: &HbmTensor<bf16, Chip, m![P, D / 2, R, R]>, // arg7
41    position_ids: &HbmTensor<i32, Chip, m![S]>,             // arg8
42    // Outputs (out#0..out#3, pre-allocated by caller)
43    out_q: &mut HbmTensor<bf16, Chip, m![S, Q]>,      // out#0
44    out_k: &mut HbmTensor<bf16, Chip, m![S, K]>,      // out#1
45    out_v: &mut HbmTensor<bf16, Chip, m![S, V]>,      // out#2
46    out_hidden: &mut HbmTensor<bf16, Chip, m![S, H]>, // out#3
47) {
48    // PyTorch Qwen2Model first layer:
49    //   hidden_states = embed_tokens(input_ids)
50    //   hidden_states = input_layernorm(hidden_states)
51    //   q, k, v = q/k/v_proj(hidden_states)
52    //   q, k = rotary_emb(position_ids, q, k)
53
54    // Embedding lookup — embed_tokens(input_ids) → [S=128, H=896]
55    let embeddings_dm = embedding::embed(ctx, input_ids, embedding_table);
56    // Write embedding output to caller-provided hidden state buffer.
57    embeddings_dm.view().to_hbm_view(&mut ctx.tdma, out_hidden.view_mut());
58    // Apply input RMSNorm.
59    let normalized = rms_norm::rms_norm(ctx, out_hidden, norm_weight);
60    // QKV projections
61    //   PyTorch: q = q_proj(x) [bias=True], k = k_proj(x) [bias=True], v = v_proj(x) [bias=True]
62    //   K/V bias is omitted in this kernel path
63    proj::v_proj(ctx, &normalized, v_weight, out_v);
64    let q = proj::q_proj(ctx, &normalized, q_weight, q_bias);
65    let k = proj::k_proj(ctx, &normalized, k_weight);
66    // q, k = rotary_emb(position_ids, q, k)
67    rope::rope(ctx, &q, &k, rope_table, position_ids, out_q, out_k);
68}
