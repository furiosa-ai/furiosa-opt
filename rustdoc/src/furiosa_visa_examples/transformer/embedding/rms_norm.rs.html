1//! RMS normalization for embedding output.
2//!
3//! `y = x * (1 / sqrt(sum(x^2) / H + eps)) * weight`
4
5use crate::transformer::axes::{
6    H,             // = 896, hidden_size = 14 * 64
7    S_decode as S, // = 128, sequence length
8    X,             // = 16, broadcast
9    Y,             // = 4, broadcast
10    Z,             // = 2, broadcast
11};
12use furiosa_visa_std::prelude::*;
13
14type Chip = m![1];
15type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
16
17/// RMS Norm: hidden_states [S, H] → normalized [S, H]
18///
19/// Returns a DmTensor in SRAM for downstream QKV projections.
20pub(super) fn rms_norm(
21    ctx: &mut Context,
22    hidden_hbm: &HbmTensor<bf16, Chip, m![S, H]>,
23    norm_weight: &HbmTensor<bf16, Chip, m![H]>,
24) -> DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]> {
25    // Reload hidden states into a tiled SRAM layout for RMSNorm.
26    let hidden_dm_0: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 224, S / 8 % 4], m![S % 8, H % 224]> =
27        hidden_hbm.to_dm(&mut ctx.tdma, 0x500);
28
29    // Reorder tiles so hidden channels map to 56-wide vector lanes.
30    let hidden_dm_1: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 224, S / 8 % 4], m![S % 32, H % 56 # 64]> = ctx
31        .main
32        .begin(hidden_dm_0.view())
33        .fetch::<bf16, m![S % 8, H / 56 % 4], m![H % 56]>()
34        .collect::<m![S % 32], m![H % 56 # 64]>()
35        .commit(0x1300);
36
37    // Strip padded lanes and keep contiguous 56-wide channel blocks.
38    let hidden_dm: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 224, S / 8 % 4], m![S % 32, H % 56]> = ctx
39        .main
40        .begin(hidden_dm_1.view())
41        .fetch::<bf16, m![S % 32], m![H % 56]>()
42        .collect::<m![S % 32], m![H % 56]>()
43        .commit(0x500);
44
45    // Load norm weights.
46    let weight_dm_0: DmTensor<bf16, Chip, Cluster, m![X, H / 112, 1 # 2], m![H % 112]> =
47        norm_weight.to_dm(&mut ctx.tdma, 0x0);
48
49    // Reorder weight tiles into 56-wide channel groups.
50    let weight_dm_1: DmTensor<bf16, Chip, Cluster, m![X, H / 112, H / 56 % 2], m![1 # 2, H % 56 # 64]> = ctx
51        .main
52        .begin(weight_dm_0.view())
53        .fetch::<bf16, m![H / 56 % 2], m![H % 56]>()
54        .switch::<m![X, H / 112, H / 56 % 2], m![Z]>(SwitchConfig::InterTranspose {
55            slice1: 2,
56            slice0: 1,
57            time0: 1,
58        })
59        .collect::<m![Z], m![H % 56 # 64]>()
60        .commit(0x100);
61
62    // Channel padding strip pass.
63    let weight_dm_2: DmTensor<bf16, Chip, Cluster, m![X, H / 112, H / 56 % 2], m![1 # 2, H % 56]> = ctx
64        .main
65        .begin(weight_dm_1.view())
66        .fetch::<bf16, m![Z], m![H % 56]>()
67        .collect::<m![Z], m![H % 56]>()
68        .commit(0x0);
69
70    // Dummy-lane strip pass for compact per-channel weights.
71    let weight_dm_3: DmTensor<bf16, Chip, Cluster, m![X, H / 112, H / 56 % 2], m![H % 56]> = ctx
72        .main
73        .begin(weight_dm_2.view())
74        .fetch::<bf16, m![1], m![H % 56]>()
75        .collect::<m![1], m![H % 56]>()
76        .commit(0x100);
77
78    // Reshape weight tiles to match the normalization pipeline's Slice decomposition.
79    let weight_dm_3: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![H % 56]> =
80        unsafe { weight_dm_3.reshape() };
81
82    // Load weights into VRF as f32 for vector multiply.
83    let weight_vrf: VrfTensor<f32, Chip, Cluster, m![Y, S / 32, H / 56], m![H % 56 # 64]> = ctx
84        .sub
85        .begin(weight_dm_3.view())
86        .fetch::<f32, m![H / 8 % 7 # 8], m![H % 8]>()
87        .collect::<m![H / 8 % 7 # 8], m![H % 8]>()
88        .to_vrf(0);
89
90    // Duplicate hidden states so variance and final scaling can run independently.
91    let mut hidden_copy: DmTensor<
92        bf16,
93        Chip,
94        Cluster,
95        m![Y, S / 32, H / 224, S / 8 % 4],
96        m![S / 8 % 4, S % 8, H % 56],
97    > = unsafe { DmTensor::from_addr(0x1300) };
98
99    hidden_dm.to_dm_pcopy(&mut ctx.sub, &mut hidden_copy);
100
101    // Compute per-token variance term in f32.
102    let variance: DmTensor<f32, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32]> = ctx
103        .main
104        .begin(hidden_dm.view())
105        .fetch::<f32, m![S % 32, H / 8 % 7], m![H % 8]>()
106        .collect::<m![S % 32, H / 8 % 7], m![H % 8]>()
107        .vector_init()
108        .vector_intra_slice_branch(BranchMode::Unconditional)
109        .vector_split::<m![S % 32, H / 8 % 7, H / 4 % 2], m![H % 4]>()
110        .vector_stash()
111        .vector_fp_binary(FpBinaryOp::MulF(FpMulAlu::Mul0), Stash)
112        .vector_intra_slice_reduce::<H, m![S % 32], m![1 # 4]>(IntraSliceReduceOpF32::Add)
113        .vector_fp_div(
114            FpDivBinaryOp::DivF.with_mode(BinaryArgMode::Mode01),
115            ArrayVec::new([VeBranchOperand::always(VeRhs::constant(896.0f32))]),
116        )
117        .vector_pad_way8::<m![1 # 8]>()
118        .vector_clip(ClipBinaryOpF32::Add, 6.25e-8f32) // TODO: use filter to move Time -> Packet
119        .vector_inter_slice_reduce::<m![Y, S / 32, H / 56], m![1]>(InterSliceReduceOpF32::Add)
120        .vector_final()
121        .commit(0x2400);
122
123    // Compute reciprocal RMS scale and keep it in VRF.
124    let inv_rms_vrf: VrfTensor<f32, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32]> = ctx
125        .main
126        .begin(variance.view())
127        .fetch::<f32, m![S / 8 % 4], m![S % 8]>()
128        .collect::<m![S / 8 % 4], m![S % 8]>()
129        .vector_init()
130        .vector_intra_slice_branch(BranchMode::Unconditional)
131        .vector_split::<m![S / 8 % 4, S / 4 % 2], m![S % 4]>()
132        .vector_fp_unary(FpUnaryOp::Sqrt)
133        .vector_fp_div(
134            FpDivBinaryOp::DivF.with_mode(BinaryArgMode::Mode10),
135            ArrayVec::new([VeBranchOperand::always(VeRhs::constant(1.0f32))]),
136        )
137        .vector_concat::<m![S / 8 % 4], m![S % 8]>()
138        .vector_final()
139        .to_vrf(0);
140
141    // Reshape hidden copy to match VRF Slice decomposition for vector operations.
142    let hidden_copy: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S / 8 % 4, S % 8, H % 56]> =
143        unsafe { hidden_copy.reshape() };
144
145    // Apply normalization: x * inv_rms * weight, then cast back to bf16.
146    let result_dm_y: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]> = ctx
147        .main
148        .begin(hidden_copy.view())
149        .fetch::<f32, m![S % 32, H / 8 % 7], m![H % 8]>()
150        .collect::<m![S % 32, H / 8 % 7], m![H % 8]>()
151        .vector_init()
152        .vector_intra_slice_branch(BranchMode::Unconditional)
153        .vector_split::<m![S % 32, H / 8 % 7, H / 4 % 2], m![H % 4]>()
154        .vector_fp_binary(FpBinaryOp::MulF(FpMulAlu::Mul0), &inv_rms_vrf)
155        .vector_fp_binary(FpBinaryOp::MulF(FpMulAlu::Mul1), &weight_vrf)
156        .vector_concat::<m![S % 32, H / 8 % 7], m![H % 8]>()
157        .vector_final()
158        .cast::<bf16, m![H % 8 # 16]>()
159        .commit(0x1300);
160
161    let result_dm: DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]> = ctx
162        .main
163        .begin(result_dm_y.view())
164        .fetch::<bf16, m![S % 32], m![H % 56]>()
165        .collect::<m![S % 32], m![H % 56]>()
166        .commit(0x1300);
167
168    result_dm
169}
