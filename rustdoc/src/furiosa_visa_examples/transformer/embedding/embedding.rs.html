1//! Embedding lookup.
2//!
3//! Scales `input_ids` to byte offsets, then gathers rows from the embedding table.
4
5use crate::transformer::axes::{
6    H,             // = 896, hidden_size = 14 * 64
7    S_decode as S, // = 128, sequence length
8    W_vocab as W,  // = 151936, vocab_size
9};
10use furiosa_visa_std::prelude::*;
11
12type Chip = m![1];
13type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
14
15/// Embedding lookup: input_ids → DmTensor in SRAM.
16///
17/// Returns the gather result in SRAM. Caller is responsible for spilling
18/// to DRAM for downstream normalization and output paths.
19pub(super) fn embed(
20    ctx: &mut Context,
21    input_ids: &HbmTensor<i32, Chip, m![S]>,
22    embedding_table: &HbmTensor<bf16, Chip, m![W, H]>,
23) -> DmTensor<bf16, Chip, Cluster, m![H / 128 # 8, S / 4], m![S % 4, H % 128]> {
24    // Load token ids to SRAM.
25    let ids_dm = input_ids.to_dm::<Cluster, m![1 # 256], m![S]>(&mut ctx.tdma, 0x200);
26
27    // Convert token ids into byte offsets for table gather.
28    let ids_scaled: DmTensor<i32, Chip, Cluster, m![1 # 256], m![S]> = ctx
29        .main
30        .begin(ids_dm.view())
31        .fetch::<i32, m![S / 8], m![S % 8]>()
32        .collect::<m![S / 8], m![S % 8]>()
33        .vector_init()
34        .vector_intra_slice_branch(BranchMode::Unconditional)
35        .vector_fxp(FxpBinaryOp::MulInt, 1792)
36        .vector_final()
37        .commit::<m![S]>(0x200);
38
39    // Reshape indices for gather layout.
40    let ids_reshaped: DmTensor<i32, Chip, Cluster, m![1 # 256], m![S % 4, S / 4]> = ctx
41        .main
42        .begin(ids_scaled.view())
43        .fetch::<i32, m![S / 4], m![S % 4]>()
44        .collect::<m![S / 4], m![S % 4 # 8]>()
45        .commit::<m![S % 4, S / 4]>(0x0);
46
47    // Move reshaped indices to HBM for gather input.
48    let ids_hbm: HbmTensor<i32, Chip, m![S % 4, S / 4]> = ids_reshaped.to_hbm(&mut ctx.tdma, 0x10e36000);
49
50    // Gather embedding rows for each token id.
51    embedding_table.dma_gather(&ids_hbm, 0x0)
52}
