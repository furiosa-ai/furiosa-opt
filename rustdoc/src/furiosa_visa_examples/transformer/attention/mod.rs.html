1//! Composable kernel: `transformer_0:attention`
2//!
3//! Pipeline: KV Cache Scatter → Q*K^T Matmul → Softmax → Score*V Matmul
4//!
5//! Qwen 2.5 0.5B, 4 PE, W16A16KV16 (bf16), prefill batch=1 seq_len=1024.
6//!
7//! ## Model Constants (Qwen 2.5 0.5B)
8//!
9//! - hidden_size = 896, num_q_heads = 14, num_kv_heads = 2, head_dim = 64
10//! - gqa_ratio = 7 (14 Q heads / 2 KV heads)
11//! - prefill seq_len = 1024, kv_cache_len = 1124 (1024 + 100 padding)
12
13mod attn_output;
14mod attn_weight;
15mod kv_cache;
16mod softmax;
17
18use crate::transformer::axes::{
19    C_kvcache as C, // = 1124, kv_cache_len = seq_len + 100 padding
20    D,              // = 64, head_dim
21    H,              // = 896, hidden_size = num_q_heads * head_dim = 14 * 64
22    K,              // = 128, kv_proj_size = N * D = 2 * 64
23    N,              // = 2, num_kv_heads
24    S_prefill as S, // = 1024, query sequence length (prefill)
25    T,              // = 1024, key/value sequence length
26};
27use furiosa_visa_std::prelude::*;
28
29type Chip = m![1];
30
31#[device(chip = 1)]
32#[expect(clippy::too_many_arguments)]
33pub fn forward(
34    ctx: &mut Context,
35    attn_q: &HbmTensor<bf16, Chip, m![S, H]>,
36    attn_k: &HbmTensor<bf16, Chip, m![T, K]>,
37    attn_v: &HbmTensor<bf16, Chip, m![T, K]>,
38    k_scatter_index: &HbmTensor<i32, Chip, m![1, T]>,
39    v_scatter_index: &HbmTensor<i32, Chip, m![1, T]>,
40    attention_mask: &HbmTensor<i32, Chip, m![1, S, T]>,
41    out_attn: &mut HbmTensor<bf16, Chip, m![S, H]>,
42    out_k_cache: &mut HbmTensor<bf16, Chip, m![C, 1, N, D]>,
43    out_v_cache: &mut HbmTensor<bf16, Chip, m![C, 1, N, D]>,
44) {
45    // PyTorch Qwen2Attention.forward():
46    //   q, k, v = ...  (already projected + RoPE'd by embedding/decoder kernel)
47    //   attn_output = attn(q, k, v, kv_cache, attention_metadata, attention_mask)
48    //
49    // Decomposed here into: KV cache write → Q×K → softmax → Score×V
50    //   scale = head_dim**-0.5 = 64**-0.5 = 0.125 (applied in attn_weight)
51    //   GQA: 14 Q-heads / 2 KV-heads = 7 groups (G=7, softmax per group)
52
53    // KV cache scatter — kv_cache.write(k, v) at indexed positions
54    kv_cache::cache_kv(
55        ctx,
56        attn_k,
57        attn_v,
58        k_scatter_index,
59        v_scatter_index,
60        out_k_cache,
61        out_v_cache,
62    );
63
64    // score = softmax(Q @ K^T * 0.125 + mask) — prefill causal attention
65    let qk_scores = attn_weight::attn_weight(ctx, attn_q, attn_k);
66    let softmax_out = softmax::softmax(ctx, &qk_scores, attention_mask);
67    // attn_output = score @ V
68    attn_output::attn_output(ctx, &softmax_out, attn_v, out_attn);
69}
