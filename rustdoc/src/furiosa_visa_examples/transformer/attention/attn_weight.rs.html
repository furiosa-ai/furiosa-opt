1use crate::transformer::axes::{
2    D,              // = 64, head_dim
3    G,              // = 7, gqa_ratio = 14 / 2
4    H,              // = 896, hidden_size = 14 * 64
5    K,              // = 128, kv_proj_size = N * D
6    N,              // = 2, num_kv_heads
7    S_prefill as S, // = 1024, query sequence length (prefill)
8    T,              // = 1024, key/value sequence length
9    W_kbcast as W,  // = 32, broadcast (K replication across S positions)
10    Y,              // = 4, broadcast
11};
12use furiosa_visa_std::prelude::*;
13
14type Chip = m![1];
15type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
16
17pub(super) fn attn_weight(
18    ctx: &mut Context,
19    attn_q: &HbmTensor<bf16, Chip, m![S, H]>,
20    attn_k: &HbmTensor<bf16, Chip, m![T, K]>,
21) -> DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]> {
22    // Load key tensor from HBM to DM.
23    let k_dm: DmTensor<bf16, Chip, Cluster, m![Y, T / 16 % 32, T / 512], m![T % 16, K]> =
24        attn_k.to_dm(&mut ctx.tdma, 0x200);
25
26    // Reorder key tiles to split K into head chunks for TRF loading.
27    let k_it: DmTensor<bf16, Chip, Cluster, m![Y, T / 16 % 32, K / 64], m![T / 512, T % 16, K % 64]> = ctx
28        .main
29        .begin(k_dm.view())
30        .fetch::<bf16, m![T % 16, K / 64], m![K % 64]>()
31        .switch::<m![Y, T / 16 % 32, K / 64], m![T % 16, T / 512]>(SwitchConfig::InterTranspose {
32            slice1: 2,
33            slice0: 1,
34            time0: 1,
35        })
36        .collect::<m![T / 512, T % 16], m![K % 64]>()
37        .commit(0x1200);
38
39    // Copy the first T half for matmul.
40    let mut k_half_0: DmTensor<bf16, Chip, Cluster, m![Y, T / 16 % 32, K / 64], m![T % 16, K % 64]> =
41        unsafe { DmTensor::from_addr(0x1200) };
42    k_it.view()
43        .tile::<m![T / 512], 2, m![T % 16, K % 64]>(0)
44        .to_dm_view_pcopy(&mut ctx.sub, k_half_0.view_mut());
45
46    // Copy the second T half for matmul.
47    let mut k_half_1: DmTensor<bf16, Chip, Cluster, m![Y, T / 16 % 32, K / 64], m![T % 16, K % 64]> =
48        unsafe { DmTensor::from_addr(0x1a00) };
49    k_it.view()
50        .tile::<m![T / 512], 2, m![T % 16, K % 64]>(1)
51        .to_dm_view_pcopy(&mut ctx.sub, k_half_1.view_mut());
52
53    // Load first key half into TRF first half.
54    let k_trf_0: TrfTensor<bf16, Chip, Cluster, m![S / 256, W, N], m![T % 8], m![K % 64]> = ctx
55        .sub
56        .begin(k_half_0.view())
57        .fetch::<bf16, m![T % 8], m![T / 8 % 2, K % 64]>()
58        .switch::<m![S / 256, W, N], m![T % 8]>(SwitchConfig::Broadcast1 { slice1: 32, slice0: 2 })
59        .collect::<m![T % 8], m![K % 64]>()
60        .to_trf(TrfAddress::FirstHalf);
61
62    // Load first key half into TRF second half for cascade accumulation.
63    let k_trf_0s: TrfTensor<bf16, Chip, Cluster, m![S / 256, W, N], m![T % 8], m![K % 64]> = ctx
64        .sub
65        .begin(k_half_0.view())
66        .fetch::<bf16, m![T % 8], m![T / 8 % 2, K % 64]>()
67        .switch::<m![S / 256, W, N], m![T % 8]>(SwitchConfig::Broadcast1 { slice1: 32, slice0: 2 })
68        .collect::<m![T % 8], m![K % 64]>()
69        .to_trf(TrfAddress::SecondHalf);
70
71    // Load query tensor from HBM to DM.
72    let q_dm: DmTensor<bf16, Chip, Cluster, m![S / 256, W, N], m![S % 8, G, D]> = attn_q.to_dm(&mut ctx.tdma, 0x2200);
73
74    // Compute Q x K^T using TRF first half and write partial scores.
75    ctx.main
76        .begin(q_dm.view())
77        .fetch::<bf16, m![S % 8, G], m![D]>()
78        .collect::<m![S % 8, G], m![D]>()
79        .align::<m![S % 8, G], m![D], _, _>(&k_trf_0)
80        .contract::<m![1]>()
81        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
82        .vector_init()
83        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
84        .vector_final()
85        .cast::<bf16, m![T % 512]>()
86        .commit::<m![S % 8, G, T % 512]>(0x8000);
87
88    // Accumulate Q x K^T with TRF second half into the same score buffer.
89    ctx.main
90        .begin(q_dm.view())
91        .fetch::<bf16, m![S % 8, G], m![D]>()
92        .collect::<m![S % 8, G], m![D]>()
93        .align::<m![S % 8, G], m![D], _, _>(&k_trf_0s)
94        .contract::<m![1]>()
95        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
96        .vector_init()
97        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
98        .vector_final()
99        .cast::<bf16, m![T % 512]>()
100        .commit::<m![S % 8, G, T % 512]>(0x8000);
101
102    // Load second key half into TRF first half.
103    let k_trf_1: TrfTensor<bf16, Chip, Cluster, m![S / 256, W, N], m![T % 8], m![K % 64]> = ctx
104        .sub
105        .begin(k_half_1.view())
106        .fetch::<bf16, m![T % 8], m![T / 8 % 2, K % 64]>()
107        .switch::<m![S / 256, W, N], m![T % 8]>(SwitchConfig::Broadcast1 { slice1: 32, slice0: 2 })
108        .collect::<m![T % 8], m![K % 64]>()
109        .to_trf(TrfAddress::FirstHalf);
110
111    // Load second key half into TRF second half for cascade accumulation.
112    let k_trf_1s: TrfTensor<bf16, Chip, Cluster, m![S / 256, W, N], m![T % 8], m![K % 64]> = ctx
113        .sub
114        .begin(k_half_1.view())
115        .fetch::<bf16, m![T % 8], m![T / 8 % 2, K % 64]>()
116        .switch::<m![S / 256, W, N], m![T % 8]>(SwitchConfig::Broadcast1 { slice1: 32, slice0: 2 })
117        .collect::<m![T % 8], m![K % 64]>()
118        .to_trf(TrfAddress::SecondHalf);
119
120    // Compute Q x K^T for the second T half using TRF first half.
121    ctx.main
122        .begin(q_dm.view())
123        .fetch::<bf16, m![S % 8, G], m![D]>()
124        .collect::<m![S % 8, G], m![D]>()
125        .align::<m![S % 8, G], m![D], _, _>(&k_trf_1)
126        .contract::<m![1]>()
127        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
128        .vector_init()
129        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
130        .vector_final()
131        .cast::<bf16, m![T % 512]>()
132        .commit::<m![S % 8, G, T % 512]>(0x16000);
133
134    // Accumulate the second T-half scores with TRF second half.
135    ctx.main
136        .begin(q_dm.view())
137        .fetch::<bf16, m![S % 8, G], m![D]>()
138        .collect::<m![S % 8, G], m![D]>()
139        .align::<m![S % 8, G], m![D], _, _>(&k_trf_1s)
140        .contract::<m![1]>()
141        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
142        .vector_init()
143        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
144        .vector_final()
145        .cast::<bf16, m![T % 512]>()
146        .commit::<m![S % 8, G, T % 512]>(0x16000);
147
148    // View both T-half score blocks as one concatenated tensor.
149    let qk_concat: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![T / 512, S % 8, G, T % 512]> =
150        unsafe { DmTensor::from_addr(0x8000) };
151
152    // Merge the two T chunks into a contiguous T axis.
153    let qk_scores: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]> = ctx
154        .main
155        .begin(qk_concat.view())
156        .fetch::<bf16, m![T / 512, S % 8, G], m![T % 512]>()
157        .collect::<m![T / 512, S % 8, G], m![T % 512]>()
158        .commit(0x28000);
159
160    qk_scores
161}
