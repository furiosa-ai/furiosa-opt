1//! Masked softmax with 7-way GQA head grouping.
2//!
3//! PyTorch: softmax(scores + mask, dim=-1) per attention head.
4//! GQA groups G=7 heads share the same KV, so softmax runs independently per group.
5//!
6//! Numerically stable: max-subtract → exp → sum → div.
7
8use crate::transformer::axes::{
9    G,              // = 7, gqa_ratio = 14 / 2
10    N,              // = 2, num_kv_heads
11    S_prefill as S, // = 1024, query sequence length (prefill)
12    T,              // = 1024, key/value sequence length
13};
14use furiosa_visa_std::prelude::*;
15
16type Chip = m![1];
17type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
18
19pub(super) fn softmax(
20    ctx: &mut Context,
21    qk_scores: &DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]>,
22    attention_mask: &HbmTensor<i32, Chip, m![1, S, T]>,
23) -> DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]> {
24    let mask_dm: DmTensor<i32, Chip, Cluster, m![S / 8, N], m![S % 8, T]> = attention_mask.to_dm(&mut ctx.tdma, 0x3e00); // Load attention mask to DM.
25
26    // Build mask branches in VRF for masked softmax.
27    let _mask_vrf: VrfTensor<i32, Chip, Cluster, m![S / 8, N], m![S % 8, T]> = ctx
28        .sub
29        .begin(mask_dm.view())
30        .fetch::<i32, m![S % 8], m![T]>()
31        .collect::<m![S % 8], m![T]>()
32        .vector_init()
33        .vector_intra_slice_branch(BranchMode::Comparison([
34            InputCmp::I32(InputCmpI32::Equal { boundary: 0 }),
35            InputCmp::I32(InputCmpI32::True),
36            InputCmp::I32(InputCmpI32::True),
37            InputCmp::I32(InputCmpI32::True),
38        ]))
39        .vector_final()
40        .to_vrf(0);
41    // BranchMode::Vrf reads this branch state implicitly.
42
43    // Pre-allocate full output tensor. Each group commits into its tile address.
44    let softmax_out: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]> =
45        unsafe { DmTensor::from_addr(0x8000) };
46
47    // Copy score tensor into a working buffer for per-group processing.
48    let mut qk_scores_buf: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]> =
49        unsafe { DmTensor::from_addr(0x28000) };
50    qk_scores.to_dm_pcopy(&mut ctx.sub, &mut qk_scores_buf);
51
52    // Run masked softmax independently for each GQA group.
53
54    // Base SRAM addresses for each GQA group workspace.
55    const GROUP_SRAM_ADDRS: [u64; 7] = [0x8000, 0xc000, 0x10000, 0x14000, 0x18000, 0x1c000, 0x20000];
56    for (g, addr) in GROUP_SRAM_ADDRS.iter().enumerate() {
57        // Extract one group tile into its workspace.
58        let mut group: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, 1, T]> =
59            unsafe { DmTensor::from_addr(*addr) };
60        qk_scores_buf
61            .view()
62            .tile::<m![G], 1, m![S % 8, 1, T]>(g)
63            .to_dm_view_pcopy(&mut ctx.sub, group.view_mut());
64
65        // Apply the mask, filling masked positions with a large negative value.
66        let masked: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, T]> = ctx
67            .main
68            .begin(group.view())
69            .fetch::<f32, m![S % 8], m![T]>()
70            .collect::<m![S % 8], m![T]>()
71            .vector_init()
72            .vector_intra_slice_branch(BranchMode::Vrf)
73            .vector_logic(LogicBinaryOpF32::BitAnd, -3.3895314e38f32)
74            .vector_final()
75            .cast::<bf16, m![T % 8 # 16]>()
76            .commit(*addr);
77
78        // Compute per-row max for numerical stability.
79        let max_vrf: VrfTensor<f32, Chip, Cluster, m![S / 8, N], m![S % 8]> = ctx
80            .main
81            .begin(masked.view())
82            .fetch::<f32, m![S % 8, T / 8 % 128], m![T % 8]>()
83            .collect::<m![S % 8, T / 8 % 128], m![T % 8]>()
84            .vector_init()
85            .vector_intra_slice_branch(BranchMode::Unconditional)
86            .vector_split::<m![S % 8, T / 8 % 128, T / 4 % 2], m![T % 4]>()
87            .vector_intra_slice_reduce::<T, m![S % 8], m![1 # 4]>(IntraSliceReduceOpF32::Max)
88            .vector_pad_way8::<m![1 # 8]>()
89            .vector_final()
90            .to_vrf(0);
91
92        // Compute sum(exp(x - max)) per row.
93        let sum_exp_vrf: VrfTensor<f32, Chip, Cluster, m![S / 8, N], m![S % 8]> = ctx
94            .main
95            .begin(masked.view())
96            .fetch::<f32, m![S % 8, T / 8 % 128], m![T % 8]>()
97            .collect::<m![S % 8, T / 8 % 128], m![T % 8]>()
98            .vector_init()
99            .vector_intra_slice_branch(BranchMode::Unconditional)
100            .vector_split::<m![S % 8, T / 8 % 128, T / 4 % 2], m![T % 4]>()
101            .vector_fp_binary(FpBinaryOp::SubF, &max_vrf)
102            .vector_fp_unary(FpUnaryOp::Exp)
103            .vector_intra_slice_reduce::<T, m![S % 8], m![1 # 4]>(IntraSliceReduceOpF32::Add)
104            .vector_pad_way8::<m![1 # 8]>() // TODO: use filter to move Time -> Packet
105            .vector_final()
106            .to_vrf(1);
107
108        // Normalize each row to produce probabilities.
109        let _softmax: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, T]> = ctx
110            .main
111            .begin(masked.view())
112            .fetch::<f32, m![S % 8, T / 8 % 128], m![T % 8]>()
113            .collect::<m![S % 8, T / 8 % 128], m![T % 8]>()
114            .vector_init()
115            .vector_intra_slice_branch(BranchMode::Unconditional)
116            .vector_split::<m![S % 8, T / 8 % 128, T / 4 % 2], m![T % 4]>()
117            .vector_fp_binary(FpBinaryOp::SubF, &max_vrf)
118            .vector_fp_unary(FpUnaryOp::Exp)
119            .vector_fp_div(
120                FpDivBinaryOp::DivF.with_mode(BinaryArgMode::Mode01),
121                ArrayVec::new([VeBranchOperand::always(VeRhs::vrf(&sum_exp_vrf))]),
122            )
123            .vector_concat::<m![S % 8, T / 8 % 128], m![T % 8]>()
124            .vector_final()
125            .cast::<bf16, m![T % 8 # 16]>()
126            .commit(*addr);
127    }
128
129    // Return the concatenated softmax output view across all groups.
130    softmax_out
131}
