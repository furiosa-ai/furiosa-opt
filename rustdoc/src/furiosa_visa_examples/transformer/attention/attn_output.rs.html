1//! Score × V matmul + output reshape.
2
3use crate::transformer::axes::{
4    D,              // = 64, head_dim
5    G,              // = 7, gqa_ratio = 14 / 2
6    H,              // = 896, hidden_size = 14 * 64
7    I,              // = 2, interleave (ClipAdd dual-input)
8    K,              // = 128, kv_proj_size = N * D
9    N,              // = 2, num_kv_heads
10    S_prefill as S, // = 1024, query sequence length (prefill)
11    T,              // = 1024, key/value sequence length
12    Z,              // = 2, broadcast
13};
14use furiosa_visa_std::prelude::*;
15
16type Chip = m![1];
17type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
18
19pub(super) fn attn_output(
20    ctx: &mut Context,
21    softmax_scores: &DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T]>,
22    attn_v: &HbmTensor<bf16, Chip, m![T, K]>,
23    out_attn: &mut HbmTensor<bf16, Chip, m![S, H]>,
24) {
25    // Load value tensor from HBM to DM for the score x V matmul.
26    let v_dm: DmTensor<bf16, Chip, Cluster, m![Z, T / 8 % 64, T / 512], m![T % 8, K]> =
27        attn_v.to_dm(&mut ctx.tdma, 0x200);
28
29    // Reorder value tiles to prepare V for TRF loading.
30    let v_it: DmTensor<bf16, Chip, Cluster, m![Z, T / 8 % 64, K / 64], m![T / 512, T % 8, K % 64]> = ctx
31        .main
32        .begin(v_dm.view())
33        .fetch::<bf16, m![T % 8, K / 64], m![K % 64]>()
34        .switch::<m![Z, T / 8 % 64, K / 64], m![T % 8, T / 512]>(SwitchConfig::InterTranspose {
35            slice1: 2,
36            slice0: 1,
37            time0: 1,
38        })
39        .collect::<m![T / 512, T % 8], m![K % 64]>()
40        .commit(0xa00);
41
42    // Transpose V packet layout for TRF consumption.
43    let v_transpose: DmTensor<bf16, Chip, Cluster, m![Z, T / 8 % 64, K / 64], m![T / 512, K % 64, T % 8]> = ctx
44        .main
45        .begin(v_it.view())
46        .fetch::<bf16, m![T / 512, K / 64], m![T % 8, K % 64]>()
47        .collect::<m![T / 512, K / 64], m![K % 64, T % 8]>()
48        .commit(0x1200);
49
50    // Copy the first T half of V.
51    let mut v_half_0: DmTensor<bf16, Chip, Cluster, m![Z, T / 8 % 64, K / 64], m![K % 64, T % 8]> =
52        unsafe { DmTensor::from_addr(0x1a00) };
53    v_transpose
54        .view()
55        .tile::<m![T / 512], 2, m![K % 64, T % 8]>(0)
56        .to_dm_view_pcopy(&mut ctx.sub, v_half_0.view_mut());
57
58    // Copy the second T half of V.
59    let mut v_half_1: DmTensor<bf16, Chip, Cluster, m![Z, T / 8 % 64, K / 64], m![K % 64, T % 8]> =
60        unsafe { DmTensor::from_addr(0x1200) };
61    v_transpose
62        .view()
63        .tile::<m![T / 512], 2, m![K % 64, T % 8]>(1)
64        .to_dm_view_pcopy(&mut ctx.sub, v_half_1.view_mut());
65
66    // Load first V half into TRF first half.
67    let v_trf_first: TrfTensor<bf16, Chip, Cluster, m![S / 8, N], m![T % 8], m![K % 64]> = ctx
68        .sub
69        .begin(v_half_0.view())
70        .fetch::<bf16, m![T % 8], m![K % 64]>()
71        .switch::<m![S / 8, N], m![T % 8]>(SwitchConfig::Broadcast1 { slice1: 128, slice0: 2 })
72        .collect::<m![T % 8], m![K % 64]>()
73        .to_trf(TrfAddress::FirstHalf);
74
75    // Load second V half into TRF second half.
76    let v_trf_second: TrfTensor<bf16, Chip, Cluster, m![S / 8, N], m![T % 8], m![K % 64]> = ctx
77        .sub
78        .begin(v_half_1.view())
79        .fetch::<bf16, m![T % 8], m![K % 64]>()
80        .switch::<m![S / 8, N], m![T % 8]>(SwitchConfig::Broadcast1 { slice1: 128, slice0: 2 })
81        .collect::<m![T % 8], m![K % 64]>()
82        .to_trf(TrfAddress::SecondHalf);
83
84    // Split the score T axis into two T halves for cascaded matmul.
85    let scores_reshaped: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![T / 512, S % 8, G, T % 512]> = ctx
86        .main
87        .begin(softmax_scores.view())
88        .fetch::<bf16, m![S % 8, G], m![T]>()
89        .collect::<m![S % 8, G, T / 8], m![T % 8]>()
90        .commit(0x28000);
91    let mut scores_half_0: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T % 512]> =
92        unsafe { DmTensor::from_addr(0x28000) };
93    scores_reshaped
94        .view()
95        .tile::<m![T / 512], 2, m![S % 8, G, T % 512]>(0)
96        .to_dm_view_pcopy(&mut ctx.sub, scores_half_0.view_mut());
97    let mut scores_half_1: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, T % 512]> =
98        unsafe { DmTensor::from_addr(0x36000) };
99    scores_reshaped
100        .view()
101        .tile::<m![T / 512], 2, m![S % 8, G, T % 512]>(1)
102        .to_dm_view_pcopy(&mut ctx.sub, scores_half_1.view_mut());
103
104    // Multiply first score half with first TRF-loaded V half.
105    let _matmul_97: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, D]> = ctx
106        .main
107        .begin(scores_half_0.view())
108        .fetch::<bf16, m![S % 8, G], m![T % 512]>()
109        .collect::<m![S % 8, G, T / 8 % 64], m![T % 8]>()
110        .align::<m![S % 8, G, T / 16 % 32], m![T % 16], _, _>(&v_trf_first)
111        .contract::<m![1]>()
112        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
113        .vector_init()
114        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
115        .vector_final()
116        .cast::<bf16, m![D % 8 # 16]>()
117        .commit(0x2200);
118
119    // Accumulate first score half with second TRF-loaded V half.
120    let matmul_98: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, D]> = ctx
121        .main
122        .begin(scores_half_0.view())
123        .fetch::<bf16, m![S % 8, G], m![T % 512]>()
124        .collect::<m![S % 8, G, T / 8 % 64], m![T % 8]>()
125        .align::<m![S % 8, G, T / 16 % 32], m![T % 16], _, _>(&v_trf_second)
126        .contract::<m![1]>()
127        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
128        .vector_init()
129        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
130        .vector_final()
131        .cast::<bf16, m![D % 8 # 16]>()
132        .commit(0x2200);
133
134    // Multiply second score half with first TRF-loaded V half.
135    let _matmul_100: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, D]> = ctx
136        .main
137        .begin(scores_half_1.view())
138        .fetch::<bf16, m![S % 8, G], m![T % 512]>()
139        .collect::<m![S % 8, G, T / 8 % 64], m![T % 8]>()
140        .align::<m![S % 8, G, T / 16 % 32], m![T % 16], _, _>(&v_trf_first)
141        .contract::<m![1]>()
142        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
143        .vector_init()
144        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
145        .vector_final()
146        .cast::<bf16, m![D % 8 # 16]>()
147        .commit(0x3e00);
148
149    // Accumulate second score half with second TRF-loaded V half.
150    let matmul_102: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, D]> = ctx
151        .main
152        .begin(scores_half_1.view())
153        .fetch::<bf16, m![S % 8, G], m![T % 512]>()
154        .collect::<m![S % 8, G, T / 8 % 64], m![T % 8]>()
155        .align::<m![S % 8, G, T / 16 % 32], m![T % 16], _, _>(&v_trf_second)
156        .contract::<m![1]>()
157        .accumulate::<m![S % 8, G], m![T % 8]>(AccumulationKind::Interleaved)
158        .vector_init()
159        .vector_inter_slice_reduce::<m![S / 8, N], m![S % 8, G]>(InterSliceReduceOpF32::Add)
160        .vector_final()
161        .cast::<bf16, m![D % 8 # 16]>()
162        .commit(0x3e00);
163
164    // Add both T-half matmul outputs to form the final attention output block.
165    let matmul_out: DmTensor<bf16, Chip, Cluster, m![S / 8, N], m![S % 8, G, D]> = ctx
166        .main
167        .begin_interleaved::<I, _, _, _, _, _>(matmul_98.view(), matmul_102.view())
168        .fetch::<f32, m![I, S % 8, G], m![D]>()
169        .collect::<m![I, S % 8, G], m![D]>()
170        .vector_init()
171        .vector_intra_slice_unzip::<I, m![1 # 2], m![1]>()
172        .vector_clip_zip(ClipBinaryOpF32::Add)
173        .vector_final()
174        .cast::<bf16, m![D]>()
175        .commit(0x2200);
176
177    // Reshape output from [N, G, D] back to hidden-size layout H.
178    let output_reshaped: DmTensor<bf16, Chip, Cluster, m![S / 8, S / 4 % 2], m![S % 4, H]> = ctx
179        .main
180        .begin(matmul_out.view())
181        .fetch::<bf16, m![S % 4, S / 4 % 2], m![H % 448]>()
182        .switch::<m![S / 8, S / 4 % 2], m![S % 4, H / 448]>(SwitchConfig::InterTranspose {
183            slice1: 2,
184            slice0: 1,
185            time0: 1,
186        })
187        .collect::<m![S % 4, H / 448], m![H % 448]>()
188        .commit(0x0);
189
190    // Write reshaped attention output back to HBM.
191    output_reshaped.view().to_hbm_view(&mut ctx.tdma, out_attn.view_mut());
192}
