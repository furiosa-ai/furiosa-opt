1//! Composable kernel: `transformer_0:output_projection..transformer_1:qkv_projection`
2//!
3//! Pipeline: O-Projection → Residual Add → RMS Norm → MLP (gate+up+SiLU+down)
4//!           → Residual Add → RMS Norm → Q/K/V Projection → RoPE
5//!
6//! Qwen 2.5 0.5B, 4 PE, W16A16KV16 (bf16), tokenwise seq_len=128.
7
8use crate::transformer::common::{mlp, norm, o_proj, rope};
9use crate::transformer::embedding::proj;
10
11use crate::transformer::axes::{
12    D,             // = 64, head_dim
13    H,             // = 896, hidden_size
14    K,             // = 128, k_proj
15    M,             // = 4864, mlp_intermediate
16    P,             // = 32768, max_position
17    Q,             // = 896, q_proj
18    R,             // = 2, rope_rot
19    S_decode as S, // = 128, sequence length
20    V,             // = 128, v_proj
21};
22use furiosa_visa_std::prelude::*;
23
24type Chip = m![1];
25
26#[device(chip = 1)]
27#[allow(clippy::too_many_arguments)]
28pub fn forward(
29    ctx: &mut Context,
30    // Inputs (arg0..arg15)
31    matmul_score_v: &HbmTensor<bf16, Chip, m![S, H]>,       // arg0
32    o_proj_weight: &HbmTensor<bf16, Chip, m![H, H]>,        // arg1
33    hidden_states: &HbmTensor<bf16, Chip, m![S, H]>,        // arg2
34    norm_weight_0: &HbmTensor<bf16, Chip, m![H]>,           // arg3
35    gate_weight: &HbmTensor<bf16, Chip, m![M, H]>,          // arg4
36    up_weight: &HbmTensor<bf16, Chip, m![M, H]>,            // arg5
37    down_weight: &HbmTensor<bf16, Chip, m![H, M]>,          // arg6
38    norm_weight_1: &HbmTensor<bf16, Chip, m![H]>,           // arg7
39    q_weight: &HbmTensor<bf16, Chip, m![Q, H]>,             // arg8
40    q_bias: &HbmTensor<bf16, Chip, m![Q]>,                  // arg9
41    k_weight: &HbmTensor<bf16, Chip, m![K, H]>,             // arg10
42    _k_bias: &HbmTensor<bf16, Chip, m![K]>,                 // arg11
43    v_weight: &HbmTensor<bf16, Chip, m![V, H]>,             // arg12
44    _v_bias: &HbmTensor<bf16, Chip, m![V]>,                 // arg13
45    rope_table: &HbmTensor<bf16, Chip, m![P, D / 2, R, R]>, // arg14
46    position_ids: &HbmTensor<i32, Chip, m![S]>,             // arg15
47    // Outputs (out#0..out#3, pre-allocated by caller)
48    out_q: &mut HbmTensor<bf16, Chip, m![S, Q]>,      // out#0
49    out_k: &mut HbmTensor<bf16, Chip, m![S, K]>,      // out#1
50    out_v: &mut HbmTensor<bf16, Chip, m![S, V]>,      // out#2
51    out_hidden: &mut HbmTensor<bf16, Chip, m![S, H]>, // out#3
52) {
53    // PyTorch Qwen2Decoder.forward() equivalent:
54    //   residual = hidden_states
55    //   hidden_states = input_layernorm(hidden_states)
56    //   hidden_states = self_attn(hidden_states, ...)
57    //   hidden_states = residual + hidden_states
58    //   residual = hidden_states
59    //   hidden_states = post_attention_layernorm(hidden_states)
60    //   hidden_states = mlp(hidden_states)
61    //   hidden_states = hidden_states + residual
62    //
63    // This kernel spans two decoder layers: it finishes layer N (o_proj → MLP)
64    // and starts layer N+1 (QKV projection → RoPE).
65
66    // Phase 1: O-projection — output = o_proj(attn_output)  [bias=False]
67    let o_proj_out = o_proj::o_proj(ctx, matmul_score_v, o_proj_weight);
68    // Phase 2: residual + post_attention_layernorm (fused)
69    //   hidden = residual + o_proj_out; hidden = RMSNorm(hidden)
70    let (normalized_0, residual_0) = norm::residual_norm(ctx, hidden_states, &o_proj_out, norm_weight_0);
71    // Phase 3: MLP — silu(gate_proj(x)) * up_proj(x), then down_proj
72    let mlp_out = mlp::mlp(ctx, &normalized_0, gate_weight, up_weight, down_weight);
73    // Phase 4: residual + input_layernorm (fused, starts next layer)
74    //   hidden = residual + mlp_out; hidden = RMSNorm(hidden)
75    let normalized_1 = norm::residual_norm_post(ctx, &residual_0, &mlp_out, norm_weight_1, out_hidden);
76    // Phase 5: QKV projections
77    //   PyTorch: q = q_proj(x) [bias=True], k = k_proj(x) [bias=True], v = v_proj(x) [bias=True]
78    //   K/V bias is omitted in this kernel path
79    proj::v_proj(ctx, &normalized_1, v_weight, out_v);
80    let q = proj::q_proj(ctx, &normalized_1, q_weight, q_bias);
81    let k = proj::k_proj(ctx, &normalized_1, k_weight);
82    // Phase 6: RoPE — q, k = rotary_emb(position_ids, q, k)
83    rope::rope(ctx, &q, &k, rope_table, position_ids, out_q, out_k);
84}
