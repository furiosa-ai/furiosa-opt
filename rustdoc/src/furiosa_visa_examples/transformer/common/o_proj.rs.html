1//! Attention output projection.
2//!
3//! PyTorch: output = o_proj(attn_output)  — Linear [H=896 → H=896, bias=False]
4//!
5//! DPE: input [S=128, H=896] is smaller → TRF FirstHalf; weight [896, 896] streams from DM.
6//! Uses 112-width tiling (H/112=8, GReduceAdd ratio=8), vs embedding’s 56-width.
7
8use crate::transformer::axes::{
9    H,             // = 896, hidden_size = 14 * 64
10    S_decode as S, // = 128, sequence length
11    X,             // = 16, broadcast
12    Z,             // = 2, broadcast
13};
14use furiosa_visa_std::prelude::*;
15
16type Chip = m![1];
17type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
18
19/// O-projection: `attn_output × O_proj_weight → [S, H]`.
20///
21/// Reversed matmul: O_proj weight [896,896] is too large for TRF FirstHalf.
22/// The kernel places input in TRF FirstHalf and streams weight from SRAM.
23pub(crate) fn o_proj(
24    ctx: &mut Context,
25    input: &HbmTensor<bf16, Chip, m![S, H]>,
26    weight: &HbmTensor<bf16, Chip, m![H, H]>,
27) -> HbmTensor<bf16, Chip, m![S, H]> {
28    // Load input to SRAM with 112-wide hidden tiling.
29    let input_dm: DmTensor<bf16, Chip, Cluster, m![Z, H / 112, S / 8], m![S % 8, H % 112]> =
30        input.to_dm(&mut ctx.tdma, 0xbb00);
31
32    // Load projection weights to SRAM.
33    let weight_dm: DmTensor<bf16, Chip, Cluster, m![H / 28, H / 448, H / 7 % 4], m![H % 7, H % 448]> =
34        weight.to_dm(&mut ctx.tdma, 0x8900);
35
36    // Reorder weight tiles for contraction over 112 hidden channels.
37    let weight_it: DmTensor<bf16, Chip, Cluster, m![H / 28, H / 448, H / 112 % 4], m![H / 7 % 4, H % 7, H % 112]> = ctx
38        .main
39        .begin(weight_dm.view())
40        .fetch::<bf16, m![H % 7, H / 112 % 4], m![H % 112]>()
41        .switch::<m![H / 28, H / 448, H / 112 % 4], m![H / 7 % 4, H % 7]>(SwitchConfig::InterTranspose {
42            slice1: 4,
43            slice0: 1,
44            time0: 1,
45        })
46        .collect::<m![H / 7 % 4, H % 7], m![H % 112]>()
47        .commit(0xa200);
48
49    // Stage input in TRF FirstHalf for reversed matmul ordering.
50    let input_trf: TrfTensor<
51        bf16,
52        Chip,
53        Cluster,
54        m![H / 448, X, H / 112],
55        m![S % 8],
56        m![H / 16 % 7, S / 8, S % 8 # 16],
57    > = ctx
58        .sub
59        .begin(input_dm.view())
60        .fetch::<bf16, m![S % 8], m![H % 112]>()
61        .switch::<m![H / 448, X, H / 112], m![S % 8]>(SwitchConfig::Broadcast1 { slice1: 16, slice0: 8 })
62        .collect::<m![S % 8], m![H / 16 % 7, S / 8, S % 8 # 16]>()
63        .to_trf(TrfAddress::FirstHalf);
64
65    // Reshape weight Slice to match TRF Slice for alignment.
66    let weight_it: DmTensor<bf16, Chip, Cluster, m![H / 448, X, H / 112], m![H / 7 % 4, H % 7, H % 112]> =
67        unsafe { weight_it.reshape() };
68
69    // Contract weights with TRF-staged input and reduce across hidden tiles.
70    let result: DmTensor<bf16, Chip, Cluster, m![H / 7, S / 64], m![H % 7, S % 64]> = ctx
71        .main
72        .begin(weight_it.view())
73        .fetch::<bf16, m![H / 7 % 4, H % 7], m![H / 16 % 7, H % 16]>()
74        .collect::<m![H / 7 % 4, H % 7], m![H / 16 % 7, H % 16]>()
75        .align::<m![H / 7 % 4, H % 7], m![H / 16 % 7, H % 16], _, _>(&input_trf)
76        .contract::<m![1]>()
77        .accumulate::<m![H % 7, S / 8 % 8], m![S % 8]>(AccumulationKind::Interleaved)
78        .vector_init()
79        .vector_inter_slice_reduce::<m![H / 7, S / 64], m![H % 7, S % 64]>(InterSliceReduceOpF32::Add)
80        .vector_final()
81        .cast::<bf16, m![S % 64]>()
82        .commit(0x8400);
83
84    // Store projected output to HBM.
85    result.to_hbm(&mut ctx.tdma, 0x10e36000)
86}
