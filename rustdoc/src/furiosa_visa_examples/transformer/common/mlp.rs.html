1//! MLP block (SwiGLU activation).
2//!
3//! PyTorch Qwen2MLP.forward():
4//!   x = silu(gate_proj(x)) * up_proj(x)    # SiLU(x) = x * sigmoid(x)
5//!   x = down_proj(x)
6//!
7//! This kernel computes gate and up projections, applies SwiGLU, then runs
8//! down projection back to hidden size.
9
10use crate::transformer::axes::{
11    H,             // = 896, hidden_size = 14 * 64
12    M,             // = 4864, mlp_intermediate_size
13    S_decode as S, // = 128, sequence length
14    X,             // = 16, broadcast
15    Y,             // = 4, broadcast
16};
17use furiosa_visa_std::prelude::*;
18
19type Chip = m![1];
20type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
21
22/// MLP (SwiGLU): normalized [S, H] → mlp_output [S, H] in DRAM.
23pub(crate) fn mlp(
24    ctx: &mut Context,
25    input: &DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]>,
26    gate_weight: &HbmTensor<bf16, Chip, m![M, H]>,
27    up_weight: &HbmTensor<bf16, Chip, m![M, H]>,
28    down_weight: &HbmTensor<bf16, Chip, m![H, M]>,
29) -> HbmTensor<bf16, Chip, m![S, H]> {
30    let gate_dm: DmTensor<bf16, Chip, Cluster, m![M / 76, M / 19 % 4], m![M % 19, H]> =
31        gate_weight.to_dm(&mut ctx.tdma, 0x0);
32
33    let gate_it: DmTensor<bf16, Chip, Cluster, m![M / 76, H / 224], m![M / 19 % 4, M % 19, H % 224]> = ctx
34        .main
35        .begin(gate_dm.view())
36        .fetch::<bf16, m![M % 19, H / 224], m![H % 224]>()
37        .switch::<m![M / 76, H / 224], m![M / 19 % 4, M % 19]>(SwitchConfig::InterTranspose {
38            slice1: 4,
39            slice0: 1,
40            time0: 1,
41        })
42        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
43        .commit(0x10000);
44
45    // Reshape gate_it Slice to match TRF Slice for alignment.
46    let gate_it: DmTensor<bf16, Chip, Cluster, m![M / 1216, X, H / 224], m![M / 19 % 4, M % 19, H % 224]> =
47        unsafe { gate_it.reshape() };
48
49    let input_trf_second: TrfTensor<
50        bf16,
51        Chip,
52        Cluster,
53        m![M / 1216, X, H / 224],
54        m![S % 8],
55        m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16],
56    > = ctx
57        .sub
58        .begin(input.view())
59        .fetch::<bf16, m![S % 8], m![H / 8 % 7, H % 8]>()
60        .switch::<m![M / 1216, X, H / 224], m![S % 8]>(SwitchConfig::Broadcast1 { slice1: 16, slice0: 8 })
61        .collect::<m![S % 8], m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16]>()
62        .to_trf(TrfAddress::SecondHalf);
63
64    let input_trf_first: TrfTensor<
65        bf16,
66        Chip,
67        Cluster,
68        m![M / 1216, X, H / 224],
69        m![S % 8],
70        m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16],
71    > = ctx
72        .sub
73        .begin(input.view())
74        .fetch::<bf16, m![S % 8], m![H / 8 % 7, H % 8]>()
75        .switch::<m![M / 1216, X, H / 224], m![S % 8]>(SwitchConfig::Broadcast1 { slice1: 16, slice0: 8 })
76        .collect::<m![S % 8], m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16]>()
77        .to_trf(TrfAddress::FirstHalf);
78
79    let gate_second: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
80        .main
81        .begin(gate_it.view())
82        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
83        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
84        .align::<m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16], _, _>(&input_trf_second)
85        .contract::<m![1]>()
86        .accumulate::<m![M % 19, S / 8 % 16], m![S % 8]>(AccumulationKind::Interleaved)
87        .vector_init()
88        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
89        .vector_final()
90        .cast::<bf16, m![S % 128]>()
91        .commit(0x0);
92
93    let gate_first_raw: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
94        .main
95        .begin(gate_it.view())
96        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
97        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
98        .align::<m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16], _, _>(&input_trf_first)
99        .contract::<m![1]>()
100        .accumulate::<m![M % 19, S / 8 % 16], m![S % 8]>(AccumulationKind::Interleaved)
101        .vector_init()
102        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
103        .vector_final()
104        .cast::<bf16, m![S % 128]>()
105        .commit(0x10000);
106
107    let gate_second_vrf: VrfTensor<f32, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
108        .sub
109        .begin(gate_second.view())
110        .fetch::<f32, m![M % 19], m![S % 128]>()
111        .collect::<m![M % 19], m![S % 128]>()
112        .to_vrf(0);
113
114    let gate_sum: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
115        .main
116        .begin(gate_first_raw.view())
117        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
118        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
119        .vector_init()
120        .vector_intra_slice_branch(BranchMode::Unconditional)
121        .vector_split::<m![M % 19, S / 8 % 16, S / 4 % 2], m![S % 4]>()
122        .vector_fp_binary(FpBinaryOp::AddF, &gate_second_vrf)
123        .vector_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
124        .vector_final()
125        .cast::<bf16, m![S % 8 # 16]>()
126        .commit(0x10000);
127
128    let gate_sigmoid: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
129        .main
130        .begin(gate_sum.view())
131        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
132        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
133        .vector_init()
134        .vector_intra_slice_branch(BranchMode::Unconditional)
135        .vector_split::<m![S / 4 % 2, M % 19, S / 8 % 16], m![S % 4]>()
136        .vector_fp_unary(FpUnaryOp::Sigmoid)
137        .vector_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
138        .vector_final()
139        .cast::<bf16, m![S % 8 # 16]>()
140        .commit(0x18000);
141
142    let up_dm: DmTensor<bf16, Chip, Cluster, m![M / 76, M / 19 % 4], m![M % 19, H]> =
143        up_weight.to_dm(&mut ctx.tdma, 0x20000);
144
145    let up_it: DmTensor<bf16, Chip, Cluster, m![M / 76, H / 224], m![M / 19 % 4, M % 19, H % 224]> = ctx
146        .main
147        .begin(up_dm.view())
148        .fetch::<bf16, m![M % 19, H / 224], m![H % 224]>()
149        .switch::<m![M / 76, H / 224], m![M / 19 % 4, M % 19]>(SwitchConfig::InterTranspose {
150            slice1: 4,
151            slice0: 1,
152            time0: 1,
153        })
154        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
155        .commit(0x28000);
156
157    // Reshape up_it Slice to match TRF Slice for alignment.
158    let up_it: DmTensor<bf16, Chip, Cluster, m![M / 1216, X, H / 224], m![M / 19 % 4, M % 19, H % 224]> =
159        unsafe { up_it.reshape() };
160
161    let up_second: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
162        .main
163        .begin(up_it.view())
164        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
165        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
166        .align::<m![M / 19 % 4, M % 19, H / 32 % 7], m![H % 32], _, _>(&input_trf_second)
167        .contract::<m![1]>()
168        .accumulate::<m![M % 19, S / 8 % 16], m![S % 8]>(AccumulationKind::Interleaved)
169        .vector_init()
170        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
171        .vector_final()
172        .cast::<bf16, m![S % 128]>()
173        .commit(0x20000);
174
175    let up_first_raw: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
176        .main
177        .begin(up_it.view())
178        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
179        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
180        .align::<m![M / 19 % 4, M % 19, H / 32 % 7], m![H % 32], _, _>(&input_trf_first)
181        .contract::<m![1]>()
182        .accumulate::<m![M % 19, S / 8 % 16], m![S % 8]>(AccumulationKind::Interleaved)
183        .vector_init()
184        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
185        .vector_final()
186        .cast::<bf16, m![S % 128]>()
187        .commit(0x30000);
188
189    let up_second_vrf: VrfTensor<f32, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
190        .sub
191        .begin(up_second.view())
192        .fetch::<f32, m![M % 19], m![S % 128]>()
193        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
194        .to_vrf(0);
195
196    let up_sum: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
197        .main
198        .begin(up_first_raw.view())
199        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
200        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
201        .vector_init()
202        .vector_intra_slice_branch(BranchMode::Unconditional)
203        .vector_split::<m![M % 19, S / 8 % 16, S / 4 % 2], m![S % 4]>()
204        .vector_fp_binary(FpBinaryOp::AddF, &up_second_vrf)
205        .vector_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
206        .vector_final()
207        .cast::<bf16, m![S % 8 # 16]>()
208        .commit(0x38000);
209
210    let up_vrf: VrfTensor<f32, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
211        .sub
212        .begin(up_sum.view())
213        .fetch::<f32, m![M % 19], m![S % 128]>()
214        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
215        .to_vrf(0);
216
217    let gated: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
218        .main
219        .begin(gate_sigmoid.view())
220        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
221        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
222        .vector_init()
223        .vector_intra_slice_branch(BranchMode::Unconditional)
224        .vector_split::<m![M % 19, S / 8 % 16, S / 4 % 2], m![S % 4]>()
225        .vector_fp_binary(FpBinaryOp::MulF(FpMulAlu::Mul0), &up_vrf)
226        .vector_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
227        .vector_final()
228        .cast::<bf16, m![S % 8 # 16]>()
229        .commit(0x0);
230
231    let gated_hbm: HbmTensor<bf16, Chip, m![M, S]> = gated.to_hbm(&mut ctx.tdma, 0x256d000);
232
233    let gated_retiled: DmTensor<bf16, Chip, Cluster, m![Y, M / 76], m![M % 76, S]> =
234        gated_hbm.to_dm(&mut ctx.tdma, 0x9000);
235
236    // Reshape gated activation Slice for down-projection TRF alignment.
237    let gated_retiled: DmTensor<bf16, Chip, Cluster, m![H / 224, M / 152, M / 76 % 2], m![M % 76, S]> =
238        unsafe { gated_retiled.reshape() };
239
240    // Load gated activation into TRF for down projection.
241    let gated_trf: TrfTensor<bf16, Chip, Cluster, m![H / 224, M / 152, M / 76 % 2], m![M % 76], m![S]> = ctx
242        .sub
243        .begin(gated_retiled.view())
244        .fetch::<bf16, m![M % 76], m![S]>()
245        .collect::<m![M % 76], m![S]>()
246        .to_trf(TrfAddress::Full);
247    let down_dm: DmTensor<bf16, Chip, Cluster, m![H / 224, M / 152, H / 112 % 2], m![H % 112, M % 152]> =
248        down_weight.to_dm(&mut ctx.tdma, 0x30000);
249    let down_it: DmTensor<
250        bf16,
251        Chip,
252        Cluster,
253        m![H / 224, M / 152, M / 76 % 2],
254        m![H / 112 % 2, H % 112, M % 76 # 80],
255    > = ctx
256        .main
257        .begin(down_dm.view())
258        .fetch::<bf16, m![H % 112, M / 76 % 2], m![M % 76]>()
259        .switch::<m![H / 224, M / 152, M / 76 % 2], m![H / 112 % 2, H % 112]>(SwitchConfig::InterTranspose {
260            slice1: 2,
261            slice0: 1,
262            time0: 1,
263        })
264        .collect::<m![H / 112 % 2, H % 112], m![M % 76 # 80]>()
265        .commit(0x34000);
266    // Reorder down-projection weight tiles for contraction alignment.
267    let down_transposed: DmTensor<bf16, Chip, Cluster, m![H / 224, M / 152, M / 76 % 2], m![M % 76, H % 224]> = ctx
268        .main
269        .begin(down_it.view())
270        .fetch::<bf16, m![H / 112 % 2, H % 112], m![M % 76]>()
271        .collect::<m![M % 76], m![H % 224]>()
272        .commit(0x20000);
273
274    // Compute down projection with gated activation staged in TRF.
275    let down_raw: DmTensor<bf16, Chip, Cluster, m![H / 56, S / 16], m![H % 56, S % 16]> = ctx
276        .main
277        .begin(down_transposed.view())
278        .fetch::<bf16, m![M % 76], m![H % 224]>()
279        .collect::<m![M % 76], m![H % 224]>()
280        .align::<m![M % 76], m![H % 224], _, _>(&gated_trf)
281        .contract::<m![1]>()
282        .accumulate::<m![M % 76], m![1 # 8]>(AccumulationKind::Sequential)
283        .vector_init()
284        .vector_inter_slice_reduce::<m![H / 56, S / 16], m![H % 56, S % 16]>(InterSliceReduceOpF32::Add)
285        .vector_final()
286        .cast::<bf16, m![S % 16]>()
287        .commit(0x8000);
288
289    let down_result: DmTensor<bf16, Chip, Cluster, m![H / 56, S / 16], m![S % 32, H % 56]> = ctx
290        .main
291        .begin(down_raw.view())
292        .fetch::<bf16, m![H % 56, S / 16 % 2], m![S % 16]>()
293        .collect::<m![S % 32], m![H % 56]>()
294        .commit(0x0);
295
296    // Write final MLP output tensor to HBM.
297    down_result.to_hbm(&mut ctx.tdma, 0x10e50000)
298}
