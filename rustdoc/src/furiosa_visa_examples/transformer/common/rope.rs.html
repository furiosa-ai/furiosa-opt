1//! Rotary Position Embedding (RoPE).
2//!
3//! This kernel applies RoPE to both K and Q projections in four stages:
4//! 1. Load `position_ids`, scale them to byte offsets, and gather per-token
5//!    2x2 rotation coefficients from `rope_table`.
6//! 2. Reshape gathered coefficients into an execution-friendly layout and load
7//!    them into TRF for reuse.
8//! 3. Apply the same RoPE contraction pattern to K: multiply each 2-element
9//!    rotation pair by its token-specific coefficient matrix, then flatten back
10//!    to contiguous K layout.
11//! 4. Apply the same contraction to Q and flatten to contiguous Q layout.
12
13use crate::transformer::axes::{
14    D,             // = 64, head_dim
15    K,             // = 128, k_proj
16    P,             // = 32768, max_position
17    Q,             // = 896, q_proj
18    R,             // = 2, rope_rot
19    S_decode as S, // = 128, sequence length
20};
21use furiosa_visa_std::prelude::*;
22
23type Chip = m![1];
24type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
25
26/// Gather rope coefficients and apply RoPE to Q and K tensors.
27///
28/// The flow is:
29/// - position IDs -> scaled offsets -> rope table gather,
30/// - rope coefficient reshape and TRF load,
31/// - K RoPE contraction and flatten,
32/// - Q RoPE contraction and flatten.
33pub(crate) fn rope(
34    ctx: &mut Context,
35    q: &HbmTensor<bf16, Chip, m![S, Q]>,
36    k: &HbmTensor<bf16, Chip, m![S, K]>,
37    rope_table: &HbmTensor<bf16, Chip, m![P, D / 2, R, R]>,
38    position_ids: &HbmTensor<i32, Chip, m![S]>,
39    out_q: &mut HbmTensor<bf16, Chip, m![S, Q]>,
40    out_k: &mut HbmTensor<bf16, Chip, m![S, K]>,
41) {
42    // Position IDs -> rope-table gather.
43
44    // Load position IDs into DM tiles used by the offset-scaling step.
45    let pos_dm: DmTensor<i32, Chip, Cluster, m![1 # 128, S / 64], m![S % 64]> =
46        position_ids.to_dm(&mut ctx.tdma, 0x2300);
47
48    // Convert each position ID into a byte offset for rope-table row access.
49    let pos_scaled: DmTensor<i32, Chip, Cluster, m![1 # 128, S / 64], m![S % 64]> = ctx
50        .main
51        .begin(pos_dm.view())
52        .fetch::<i32, m![S / 8 % 8], m![S % 8]>()
53        .collect::<m![S / 8 % 8], m![S % 8]>()
54        .vector_init()
55        .vector_intra_slice_branch(BranchMode::Unconditional)
56        .vector_fxp(FxpBinaryOp::MulInt, 256)
57        .vector_final()
58        .commit::<m![S % 64]>(0x2300);
59
60    // Reshape offsets into the layout expected by `dma_gather` indexing.
61    let pos_reshaped: DmTensor<i32, Chip, Cluster, m![1 # 128, S / 64], m![S / 32 % 2, S % 16, S / 16 % 2]> = ctx
62        .main
63        .begin(pos_scaled.view())
64        .fetch::<i32, m![S / 32 % 2, S / 16 % 2], m![S % 16]>()
65        .collect::<m![S / 32 % 2, S / 16 % 2], m![S % 16]>()
66        .commit(0x2300);
67
68    // Spill reshaped offsets to HBM and use them as gather indices.
69    let pos_hbm: HbmTensor<i32, Chip, m![S]> = pos_reshaped.to_hbm(&mut ctx.tdma, 0x10e36000);
70
71    // Gather per-position RoPE coefficients from `rope_table`.
72    let rope_dm: DmTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![D % 16, R, R]> =
73        rope_table.dma_gather(&pos_hbm, 0x500);
74
75    // RoPE coefficient reshape + TRF load.
76
77    // Move the rotation dimensions into the layout used for TRF-backed contraction.
78    let rope_reshaped: DmTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![R, R, D % 16]> = ctx
79        .main
80        .begin(rope_dm.view())
81        .fetch::<bf16, m![R, R], m![D % 16]>()
82        .collect::<m![R, R], m![D % 16]>()
83        .commit(0x700);
84
85    // Load rope coefficients into TRF FirstHalf for reuse across Q and K.
86    let rope_trf: TrfTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![R], m![R, D % 16]> = ctx
87        .sub
88        .begin(rope_reshaped.view())
89        .fetch::<bf16, m![R], m![R, D % 16]>()
90        .collect::<m![R], m![R, D % 16]>()
91        .to_trf(TrfAddress::FirstHalf);
92
93    // K RoPE: load K in rotation-pair layout and place it in TRF SecondHalf.
94    let k_dm: DmTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![R, R, D % 16]> = k.to_dm(&mut ctx.tdma, 0x0);
95    let k_trf: TrfTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![R], m![R, D % 16]> = ctx
96        .sub
97        .begin(k_dm.view())
98        .fetch::<bf16, m![R], m![R, D % 16]>()
99        .collect::<m![R], m![R, D % 16]>()
100        .to_trf(TrfAddress::SecondHalf);
101
102    // Apply RoPE to K by contracting each 2x2 coefficient matrix with K pairs.
103    let k_rotated: DmTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![R, R, D % 16]> = ctx
104        .main
105        .begin(rope_reshaped.view())
106        .fetch::<bf16, m![R, R], m![D % 16]>()
107        .collect::<m![R, R], m![D % 16]>()
108        .align::<m![R], m![R, D % 16], _, _>(&k_trf)
109        .contract::<m![1]>()
110        .accumulate::<m![R, R], m![D % 16]>(AccumulationKind::Sequential)
111        .vector_init()
112        .vector_inter_slice_reduce::<m![S, D / 16 % 2], m![R, R, D % 16]>(InterSliceReduceOpF32::Add)
113        .vector_final()
114        .cast::<bf16, m![D % 16]>()
115        .commit(0x0);
116
117    // Flatten rotated K from decomposed tiles back to contiguous K layout.
118    let k_d0b: DmTensor<bf16, Chip, Cluster, m![S, 1 # 2], m![K]> = ctx
119        .main
120        .begin(k_rotated.view())
121        .fetch::<bf16, m![R, R], m![D % 16]>()
122        .switch::<m![S, 1 # 2], m![R, R]>(SwitchConfig::Broadcast01 {
123            slice1: 2, // D/16%2 = dim0_volume
124            slice0: 1,
125            time0: 1,
126        })
127        .collect::<m![R, R], m![D % 16]>()
128        .commit(0x100);
129    // Write K result to output buffer.
130    k_d0b.view().to_hbm_view(&mut ctx.tdma, out_k.view_mut());
131
132    // Q RoPE.
133
134    // Reload Q into the layout used by the RoPE contraction path.
135    let q_dm_reload: DmTensor<bf16, Chip, Cluster, m![S / 16, D % 16, D / 16 % 2], m![Q / 64, R, S % 16]> =
136        q.to_dm(&mut ctx.tdma, 0x200);
137
138    // Reorder Q so the sequence sub-axis and head-group axis match downstream access.
139    let q_te: DmTensor<bf16, Chip, Cluster, m![S / 16, D % 16, D / 16 % 2], m![R, S % 16, Q / 64 # 16]> = ctx
140        .main
141        .begin(q_dm_reload.view())
142        .fetch::<bf16, m![R, Q / 64], m![S % 16]>()
143        .collect::<m![R, Q / 64], m![S % 16]>()
144        .commit(0x600);
145
146    // Move Q into [Q-head-group, rotation, packet] form for contraction.
147    let q_it: DmTensor<bf16, Chip, Cluster, m![S / 16, S % 16, D / 16 % 2], m![R, D % 16, Q / 64]> = ctx
148        .main
149        .begin(q_te.view())
150        .fetch::<bf16, m![R, S % 16], m![Q / 64 # 16]>()
151        .switch::<m![S / 16, S % 16, D / 16 % 2], m![R, D % 16]>(SwitchConfig::InterTranspose {
152            slice1: 16,
153            slice0: 2,
154            time0: 1,
155        })
156        .collect::<m![R, D % 16], m![Q / 64]>()
157        .commit(0x200);
158
159    // Reshape q_it Slice to match rope_trf Slice for alignment.
160    let q_it: DmTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![R, D % 16, Q / 64]> = unsafe { q_it.reshape() };
161
162    // Apply RoPE to Q with the same contraction pattern used for K.
163    let q_rotated: DmTensor<bf16, Chip, Cluster, m![S, D / 16 % 2], m![Q / 64, R, D % 16]> = ctx
164        .main
165        .begin(q_it.view())
166        .fetch::<bf16, m![Q / 64, R], m![D % 16]>()
167        .collect::<m![Q / 64, R], m![D % 16]>()
168        .align::<m![R], m![R, D % 16], _, _>(&rope_trf)
169        .contract::<m![1]>()
170        .accumulate::<m![Q / 64, R], m![D % 16]>(AccumulationKind::Sequential)
171        .vector_init()
172        .vector_inter_slice_reduce::<m![S, D / 16 % 2], m![Q / 64, R, D % 16]>(InterSliceReduceOpF32::Add)
173        .vector_final()
174        .cast::<bf16, m![D % 16]>()
175        .commit(0x600);
176
177    // Flatten rotated Q from decomposed tiles back to contiguous Q layout.
178    let q_d0b: DmTensor<bf16, Chip, Cluster, m![S, 1 # 2], m![Q]> = ctx
179        .main
180        .begin(q_rotated.view())
181        .fetch::<bf16, m![Q / 64, R], m![D % 16]>()
182        .switch::<m![S, 1 # 2], m![Q / 64, R]>(SwitchConfig::Broadcast01 {
183            slice1: 2,
184            slice0: 1,
185            time0: 1,
186        })
187        .collect::<m![Q / 64, R], m![D % 16]>()
188        .commit(0xa00);
189    // Write Q result to output buffer.
190    q_d0b.view().to_hbm_view(&mut ctx.tdma, out_q.view_mut());
191}
