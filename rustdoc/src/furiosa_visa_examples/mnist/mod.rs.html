8fn fc1_matmul(
9    ctx: &mut Context,
10    input: &HbmTensor<bf16, Chip, m![X]>,
11    weight: &HbmTensor<bf16, Chip, m![H, X]>,
12) -> DmTensor<bf16, Chip, Cluster, m![H], m![1 # 16]> {
13    let input_dm: DmTensor<bf16, Chip, Cluster, m![H], m![X]> = input.to_dm(&mut ctx.tdma, 0);
14    let weight_dm: DmTensor<bf16, Chip, Cluster, m![H], m![X]> = weight.to_dm(&mut ctx.tdma, 0x10000);
15
16    let input_trf: TrfTensor<bf16, Chip, Cluster, m![H], m![1], m![X]> = ctx
17        .sub
18        .begin(input_dm.view())
19        .fetch::<bf16, m![1], m![X]>()
20        .collect::<m![X / 16], m![X % 16]>()
21        .to_trf(TrfAddress::Full);
22
23    ctx.main
24        .begin(weight_dm.view())
25        .fetch::<bf16, m![X / 16], m![X % 16]>()
26        .collect::<m![X / 16], m![X % 16]>()
27        .align::<m![X / 32], m![X % 32], _, _>(&input_trf)
28        .contract::<m![1]>()
29        .accumulate::<m![1], m![1 # 8]>(AccumulationKind::Interleaved)
30        .cast::<bf16, m![1 # 16]>()
31        .commit(0x20000)
32}

34fn fc1_bias_prepared(
35    ctx: &mut Context,
36    bias: &HbmTensor<bf16, Chip, m![H]>,
37) -> DmTensor<bf16, Chip, Cluster, m![H], m![1 # 16]> {
38    let bias_dm_0: DmTensor<bf16, Chip, Cluster, m![H / 8, 1 # 8], m![H % 8]> = bias.to_dm(&mut ctx.tdma, 0x30000);
39    let bias_dm_1: DmTensor<bf16, Chip, Cluster, m![H / 8, 1 # 8], m![H % 8, 1 # 8]> = ctx
40        .main
41        .begin(bias_dm_0.view())
42        .fetch::<bf16, m![1], m![H % 8]>()
43        .collect::<m![1], m![H % 8 # 16]>()
44        .transpose::<m![H % 8], m![1 # 16]>()
45        .commit(0x30000);
46    let bias_dm_2: DmTensor<bf16, Chip, Cluster, m![H / 8, Dummy8], m![H % 8, 1 # 8]> = unsafe { bias_dm_1.reshape() };
47    let bias_dm_3: DmTensor<bf16, Chip, Cluster, m![H], m![Dummy8 # 16]> = ctx
48        .main
49        .begin(bias_dm_2.view())
50        .fetch::<bf16, m![H % 8], m![1 # 8]>()
51        .switch::<m![H], m![Dummy8]>(SwitchConfig::InterTranspose {
52            slice1: 8,
53            slice0: 1,
54            time0: 1,
55        })
56        .collect::<m![Dummy8], m![1 # 16]>()
57        .transpose::<m![1], m![Dummy8 # 16]>()
58        .commit(0x20000);
59
60    unsafe { bias_dm_3.reshape() }
61}

63fn fc1_relu(
64    ctx: &mut Context,
65    input: &HbmTensor<bf16, Chip, m![X]>,
66    weight: &HbmTensor<bf16, Chip, m![H, X]>,
67    bias: &HbmTensor<bf16, Chip, m![H]>,
68) -> HbmTensor<bf16, Chip, m![H]> {
69    let matmul = fc1_matmul(ctx, input, weight);
70    let bias_dm_4 = fc1_bias_prepared(ctx, bias);
71
72    let activated: DmTensor<bf16, Chip, Cluster, m![H], m![1 # 4]> = ctx
73        .main
74        .begin_interleaved::<I, _, _, _, _, _>(matmul.view(), bias_dm_4.view())
75        .fetch::<f32, m![I], m![1 # 2]>()
76        .collect::<m![I], m![1 # 8]>()
77        .vector_init()
78        .vector_intra_slice_unzip::<I, m![1 # 2], m![1]>()
79        .vector_clip_zip(ClipBinaryOpF32::Add)
80        .vector_clip(ClipBinaryOpF32::Max, 0.0f32)
81        .vector_final()
82        .cast::<bf16, m![1 # 16]>()
83        .commit(0x20000);
84
85    activated.to_hbm(&mut ctx.tdma, 0x1000_0000)
86}

135fn fc2(
136    ctx: &mut Context,
137    input: &HbmTensor<bf16, Chip, m![H]>,
138    weight: &HbmTensor<bf16, Chip, m![C, H]>,
139    bias: &HbmTensor<bf16, Chip, m![C]>,
140) -> HbmTensor<bf16, Chip, m![C]> {
141    let matmul = fc2_matmul(ctx, input, weight);
142    let bias_dm = fc2_bias_prepared(ctx, bias);
143
144    let logits: DmTensor<bf16, Chip, Cluster, m![C, 1 # 16], m![1 # 16]> = ctx
145        .main
146        .begin_interleaved::<I, _, _, _, _, _>(matmul.view(), bias_dm.view())
147        .fetch::<f32, m![I], m![1 # 2]>()
148        .collect::<m![I], m![1 # 8]>()
149        .vector_init()
150        .vector_intra_slice_unzip::<I, m![1 # 2], m![1]>()
151        .vector_clip_zip(ClipBinaryOpF32::Add)
152        .vector_final()
153        .cast::<bf16, m![1 # 16]>()
154        .commit(0x60000);
155
156    logits.to_hbm(&mut ctx.tdma, 0x1100_0000)
157}

159#[device(chip = 1)]
160pub fn forward(
161    ctx: &mut Context,
162    input: &HbmTensor<bf16, Chip, m![X]>,
163    fc1_weight: &HbmTensor<bf16, Chip, m![H, X]>,
164    fc1_bias: &HbmTensor<bf16, Chip, m![H]>,
165    fc2_weight: &HbmTensor<bf16, Chip, m![C, H]>,
166    fc2_bias: &HbmTensor<bf16, Chip, m![C]>,
167) -> HbmTensor<bf16, Chip, m![C]> {
168    let hidden = fc1_relu(ctx, input, fc1_weight, fc1_bias);
169    fc2(ctx, &hidden, fc2_weight, fc2_bias)
