1//! Matrix multiplication with chip reduce: (1024x4) * (4x1024) -> (1024x1024)
2//! This example demonstrates reduction over the chip dimension.
3
4use furiosa_visa_std::prelude::*;
5
6axes![A = 1024, B = 4, C = 1024, I = 2];
7
8type Chip = m![B]; // Chip dimension = 4
9type Cluster = m![1 # 2];
10
11/// Multiply matrices: [A, B] * [B, C] -> [A, C]
12/// where B=4 is mapped to Chip dimension and needs to be reduced
13#[device(chip = 4)]
14pub fn matmul_chip_reduce(
15    ctx: &mut Context,
16    lhs: &HbmTensor<i8, Chip, m![A]>,
17    rhs: &HbmTensor<i8, Chip, m![C]>,
18) -> HbmTensor<i8, m![C / 2 % 4], m![A, C / 8, C % 2]> {
19    let lhs = lhs.to_dm::<Cluster, m![A / 4], m![A % 4]>(&mut ctx.tdma, 0);
20
21    let rhs = rhs.to_dm::<Cluster, m![C / 4], m![C % 4]>(&mut ctx.tdma, 128 * 1024);
22
23    // Load rhs into VRF
24    let rhs_broadcasted: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4, C % 4 # 8]> = ctx
25        .main
26        .begin(rhs.view())
27        .fetch::<i8, m![1], m![C % 4 # 8]>()
28        .switch::<m![A / 4], m![C / 4]>(SwitchConfig::Broadcast01 {
29            slice1: 256,
30            slice0: 1,
31            time0: 1,
32        })
33        .collect::<m![C / 4], m![C % 4 # 32]>()
34        .commit(0);
35    let rhs_vrf: VrfTensor<i32, Chip, Cluster, m![A / 4], m![C / 4, C % 4 # 8]> = ctx
36        .sub
37        .begin(rhs_broadcasted.view())
38        .fetch::<i32, m![C / 4], m![C % 4 # 8]>()
39        .collect::<m![C / 4], m![C % 4 # 8]>()
40        .to_vrf(0);
41
42    // Perform elementwise mul
43    // The B dimension is in Chip (=4), which will be reduced later via reduce_over_chip
44    // Result shape: [Chip=4, Cluster=2, Slice=A/4, Element=[C, A%4 # 8]]
45    let mul_result: DmTensor<i32, Chip, Cluster, m![A / 4], m![C, A % 4 # 8]> = ctx
46        .main
47        .begin(lhs.view())
48        .fetch::<i32, m![C / 4, C % 4], m![A % 4]>()
49        .collect::<m![C / 4, C % 4], m![A % 4 # 8]>()
50        .vector_init()
51        .vector_intra_slice_branch(BranchMode::Unconditional)
52        .vector_fxp(FxpBinaryOp::MulInt, &rhs_vrf)
53        .vector_final()
54        .commit(0);
55
56    // Now reduce over the Chip dimension using ReduceScatter pattern (for 4 chips)
57    let reduced = reduce_over_chip(ctx, &mul_result);
58
59    // Write back to HBM
60    reduced.to_hbm(&mut ctx.tdma, 0x3000)
61}

63/// Reduce over chip axis using ReduceScatter pattern for Chip=4.
64fn reduce_over_chip(
65    ctx: &mut Context,
66    tensor: &DmTensor<i32, Chip, Cluster, m![A / 4], m![C, A % 4 # 8]>,
67) -> DmTensor<i8, m![C / 2 % 4], Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> {
68    // ReduceScatter for Chip=4
69    // Input:  Chip=[B=4], Cluster=2, Slice=[A/4], Element=[C, A%4 # 8]
70    // Output: Chip=[C/2%4], Cluster=2, Slice=[A/4], Element=[C/8, C%2, A%4 # 8]
71    //
72    // parallel_copy_chip_slice will:
73    // - Take Element's first dimension (C) and slice it into 4 parts
74    // - C splits as: C = (C/8) × 4 × (C%2), each chip picks 1 of 4
75    // - Each chip ends up with C/4 = (C/8 × C%2) elements
76
77    // Step 1: Create T0 with Slice(0,1,2,3)
78    let sliced0: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
79        .sub
80        .parallel_copy_chip_slice::<4, m![C / 2 % 4], m![C / 8, 1 # 4, C % 2, A % 4 # 8], _, _, _, _, _, _>(
81            tensor.view(),
82            &[0, 1, 2, 3],
83        );
84
85    // Step 2: Create T1 with Slice(3,0,1,2) + ChipShuffle(1,2,3,0)
86    let sliced1: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
87        .sub
88        .parallel_copy_chip_slice::<4, m![C / 2 % 4], m![C / 8, 1 # 4, C % 2, A % 4 # 8], _, _, _, _, _, _>(
89            tensor.view(),
90            &[3, 0, 1, 2],
91        );
92
93    let shuffled1: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
94        .tdma
95        .dm_chip_shuffle::<_, 4, _, _, _, _>(sliced1.view(), &[1, 2, 3, 0]);
96
97    // Step 3: Create T2 with Slice(2,3,0,1) + ChipShuffle(2,3,0,1)
98    let sliced2: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
99        .sub
100        .parallel_copy_chip_slice::<4, m![C / 2 % 4], m![C / 8, 1 # 4, C % 2, A % 4 # 8], _, _, _, _, _, _>(
101            tensor.view(),
102            &[2, 3, 0, 1],
103        );
104
105    let shuffled2: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
106        .tdma
107        .dm_chip_shuffle::<_, 4, _, _, _, _>(sliced2.view(), &[2, 3, 0, 1]);
108
109    // Step 4: Create T3 with Slice(1,2,3,0) + ChipShuffle(3,0,1,2)
110    let sliced3: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
111        .sub
112        .parallel_copy_chip_slice::<4, m![C / 2 % 4], m![C / 8, 1 # 4, C % 2, A % 4 # 8], _, _, _, _, _, _>(
113            tensor.view(),
114            &[1, 2, 3, 0],
115        );
116
117    let shuffled3: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> = ctx
118        .tdma
119        .dm_chip_shuffle::<_, 4, _, _, _, _>(sliced3.view(), &[3, 0, 1, 2]);
120
121    // Step 5: Add T0 + T1 + T2 + T3 using two-stage binary addition
122    let mut sum01: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> =
123        unsafe { DmTensor::from_addr(0) };
124    let mut sum012: DmTensor<i32, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> =
125        unsafe { DmTensor::from_addr(0) };
126
127    // Add T0 + T1
128    ctx.main
129        .begin_interleaved::<I, _, _, _, _, _>(sliced0.view(), shuffled1.view())
130        .fetch::<i32, m![C / 8, C % 2, I], m![A % 4 # 8]>()
131        .collect::<m![C / 8, C % 2, I], m![A % 4 # 8]>()
132        .vector_init()
133        .vector_intra_slice_unzip::<I, m![C / 8, C % 2, 1 # 2], m![C / 8, C % 2]>()
134        .vector_clip_zip(ClipBinaryOpI32::AddFxp)
135        // After vector_clip_zip, result is implicitly filtered to Group 1 only
136        .vector_final()
137        .commit_view(sum01.view_mut());
138
139    // Add (TO + T1) + T2
140    ctx.main
141        .begin_interleaved::<I, _, _, _, _, _>(sum01.view(), shuffled2.view())
142        .fetch::<i32, m![C / 8, C % 2, I], m![A % 4 # 8]>()
143        .collect::<m![C / 8, C % 2, I], m![A % 4 # 8]>()
144        .vector_init()
145        .vector_intra_slice_unzip::<I, m![C / 8, C % 2, 1 # 2], m![C / 8, C % 2]>()
146        .vector_clip_zip(ClipBinaryOpI32::AddFxp)
147        // After vector_clip_zip, result is implicitly filtered to Group 1 only
148        .vector_final()
149        .commit_view(sum012.view_mut());
150
151    // Stage 2: Add ((T0 + T1) + T2) + T3 to get final result
152    let mut reduced: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> =
153        unsafe { DmTensor::from_addr(0) };
154
155    ctx.main
156        .begin_interleaved::<I, _, _, _, _, _>(sum012.view(), shuffled3.view())
157        .fetch::<i32, m![C / 8, C % 2, I], m![A % 4 # 8]>()
158        .collect::<m![C / 8, C % 2, I], m![A % 4 # 8]>()
159        .vector_init()
160        .vector_intra_slice_unzip::<I, m![C / 8, C % 2, 1 # 2], m![C / 8, C % 2]>()
161        .vector_clip_zip(ClipBinaryOpI32::AddFxp)
162        // After vector_clip_zip, result is implicitly filtered to Group 1 only
163        .vector_final()
164        .cast::<i8, m![A % 4 # 32]>()
165        .commit_view(reduced.view_mut());
166
167    // Reshape to output format
168    // Input:  Chip=[B=4], Cluster, Slice=[A/4], Element=[C/8, C%2, A%4#8]
169    // Output: Chip=[C/2%4], Cluster, Slice=[A/4], Element=[C/8, C%2, A%4#8]
170    let reshaped: DmTensor<i8, m![C / 2 % 4], Cluster, m![A / 4], m![C / 8, C % 2, A % 4 # 8]> =
171        unsafe { reduced.reshape() };
172
173    reshaped
174}
