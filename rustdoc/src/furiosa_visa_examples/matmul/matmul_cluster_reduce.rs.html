1//! Matrix multiplication with cluster reduce: (1024x2) * (2x1024) -> (1024x1024)
2//! This example demonstrates reduction over the cluster dimension.
3
4use furiosa_visa_std::prelude::*;
5
6axes![A = 1024, B = 2, C = 1024, X = 32, I = 2];
7
8type Chip = m![1];
9type Cluster = m![B]; // Cluster dimension = 2
10
11/// Multiply matrices: [A, B] * [B, C] -> [A, C]
12/// where B=2 is mapped to Cluster dimension and needs to be reduced
13#[device(chip = 1)]
14pub fn matmul_cluster_reduce(
15    ctx: &mut Context,
16    lhs: &HbmTensor<i8, Chip, m![A, B]>,
17    rhs: &HbmTensor<i8, Chip, m![B, C]>,
18) -> HbmTensor<i8, Chip, m![A, C]> {
19    // Load lhs: [A=1024, B=2] with B mapped to Cluster
20    let lhs = lhs.to_dm::<Cluster, m![A / 4], m![A % 4]>(&mut ctx.tdma, 0);
21
22    // Load rhs: [B=2, C=1024] with B mapped to Cluster
23    let rhs = rhs.to_dm::<Cluster, m![C / 4], m![C % 4]>(&mut ctx.tdma, 128 * 1024);
24
25    // Load rhs into VRF
26    let rhs_broadcasted: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4, C % 4 # 8]> = ctx
27        .main
28        .begin(rhs.view())
29        .fetch::<i8, m![1], m![C % 4 # 8]>()
30        .switch::<m![A / 4], m![C / 4]>(SwitchConfig::Broadcast01 {
31            slice1: 256,
32            slice0: 1,
33            time0: 1,
34        })
35        .collect::<m![C / 4], m![C % 4 # 32]>()
36        .commit(0);
37    let rhs_vrf: VrfTensor<i32, Chip, Cluster, m![A / 4], m![C / 4, C % 4 # 8]> = ctx
38        .sub
39        .begin(rhs_broadcasted.view())
40        .fetch::<i32, m![C / 4], m![C % 4 # 8]>()
41        .collect::<m![C / 4], m![C % 4 # 8]>()
42        .to_vrf(0);
43
44    // Perform elementwise mul
45    // The B dimension is in Cluster, so after contraction we still have Cluster=2
46    // Result will have broadcast dimension X from GAT reduction
47    let mul_result: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4, C % 4, A % 4 # 8]> = ctx
48        .main
49        .begin(lhs.view())
50        .fetch::<i32, m![C / 4, C % 4], m![A % 4]>()
51        .collect::<m![C / 4, C % 4], m![A % 4 # 8]>()
52        .vector_init()
53        .vector_intra_slice_branch(BranchMode::Unconditional)
54        .vector_fxp(FxpBinaryOp::MulInt, &rhs_vrf)
55        .vector_final()
56        .cast::<i8, m![A % 4 # 32]>()
57        .commit(0);
58
59    // Now reduce over the Cluster dimension using ReduceScatter pattern
60    let reduced = reduce_over_cluster(ctx, &mul_result);
61
62    // Write back to HBM
63    reduced.to_hbm(&mut ctx.tdma, 0x3000)
64}

66/// Reduce over cluster axis using ReduceScatter pattern.
67fn reduce_over_cluster(
68    ctx: &mut Context,
69    tensor: &DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4, C % 4, A % 4 # 8]>,
70) -> DmTensor<i8, Chip, m![C / 512], m![A / 4], m![C / 4 % 128, C % 4, A % 4 # 8]> {
71    // ReduceScatter for Cluster=2
72    // Input: Cluster dimension B=2
73    // Output: C / 512 promoted to Cluster dimension
74    //
75    // For 2-cluster ReduceScatter:
76    // Step 1: Create T0 (with slice (0, 1)) and T1 (with slice(1,0) + shuffle(1,0))
77    // Step 2: Add T0 + T1 to reduce cluster axis
78
79    // Step 1a: Use ParallelCopy (via sub context with stos) to create sliced version
80    // This performs asymmetric cluster slice operation
81    let sliced0: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4 % 128, C % 4, A % 4 # 8]> = ctx
82        .sub
83        .parallel_copy_cluster_slice::<2, m![C / 512], m![1 # 2, C / 4 % 128, C % 4, A % 4 # 8], _, _, _, _, _, _>(
84            tensor.view(),
85            &[0, 1],
86        );
87
88    // Step 1b: Use DmaCommand to shuffle the sliced data across clusters
89    // This swaps data between Cluster 0 and Cluster 1
90    let sliced1: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4 % 128, C % 4, A % 4 # 8]> = ctx
91        .sub
92        .parallel_copy_cluster_slice::<2, m![C / 512], m![1 # 2, C / 4 % 128, C % 4, A % 4 # 8], _, _, _, _, _, _>(
93            tensor.view(),
94            &[1, 0],
95        );
96
97    let shuffled1: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4 % 128, C % 4, A % 4 # 8]> =
98        ctx.tdma.dm_cluster_shuffle::<_, 2, _, _, _, _>(sliced1.view(), &[1, 0]);
99
100    // Step 2: Binary add T0 + T1 to reduce the cluster dimension
101    // Use interleaved fetch with I=2 to read both tensors
102    let mut reduced: DmTensor<i8, Chip, Cluster, m![A / 4], m![C / 4 % 128, C % 4, A % 4 # 8]> =
103        unsafe { DmTensor::from_addr(0) };
104
105    ctx.main
106        .begin_interleaved::<I, _, _, _, _, _>(sliced0.view(), shuffled1.view())
107        .fetch::<i32, m![C / 4 % 128, C % 4, I], m![A % 4 # 8]>()
108        .collect::<m![C / 4 % 128, C % 4, I], m![A % 4 # 8]>()
109        .vector_init()
110        .vector_intra_slice_unzip::<I, m![C / 4 % 128, C % 4, 1 # 2], m![C / 4 % 128, C % 4]>()
111        .vector_clip_zip(ClipBinaryOpI32::AddFxp)
112        // After vector_clip_zip, result is implicitly filtered to Group 1 only
113        .vector_final()
114        .cast::<i8, m![A % 4 # 32]>()
115        .commit_view(reduced.view_mut());
116
117    let reshaped: DmTensor<i8, Chip, m![C / 512], m![A / 4], m![C / 4 % 128, C % 4, A % 4 # 8]> =
118        unsafe { reduced.reshape() };
119
120    reshaped
121}
