1//! Matrix multiplication of size [64, 1024] x [1024, 128] -> [64, 128] with split accumulation pattern.
2//! This demonstrates the `test_compile_dfg_with_loop_and_index_access_ref` example in virtual ISA.
3
4use furiosa_visa_std::prelude::*;
5
6axes![M = 64, K = 1024, N = 128, I = 2];
7
8#[device(chip = 1)]
9pub fn matmul_with_split_reduce2(
10    ctx: &mut Context,
11    a: &HbmTensor<bf16, m![1], m![M, K]>,
12    b: &HbmTensor<bf16, m![1], m![K, N]>,
13    acc_zero: &HbmTensor<bf16, m![1], m![M, N]>,
14) -> HbmTensor<bf16, m![1], m![M, N]> {
15    type Chip = m![1];
16    type Cluster = m![1 # 2];
17
18    let mut acc: DmTensor<bf16, Chip, Cluster, m![1 # 4, M], m![N]> = acc_zero.to_dm(&mut ctx.tdma, 0x0000_0000);
19
20    for i in 0..4 {
21        let t87: DmTensor<bf16, Chip, Cluster, m![1 # 4, M], m![K % 128]> = a
22            .view()
23            .tile::<m![K / 128], 1, m![M, 1 # 8, K % 128]>(i * 2)
24            .to_dm(&mut ctx.tdma, 0x0000_0100);
25
26        let t95: DmTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, M % 8], m![K % 128 / 16, K % 128 % 16]> =
27            unsafe { t87.reshape() };
28        let t96: DmTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, K % 128 / 16], m![M % 8, K % 128 % 16]> = ctx
29            .main
30            .begin(t95.view())
31            .fetch::<bf16, m![K % 128 / 16], m![K % 128 % 16]>()
32            .switch::<m![1 # 4, M / 8, K % 128 / 16], m![M % 8]>(SwitchConfig::InterTranspose {
33                slice1: 8,
34                slice0: 1,
35                time0: 1,
36            })
37            .collect::<m![M % 8], m![K % 128 % 16]>()
38            .commit(0x0000_0200);
39        let t72: DmTensor<bf16, Chip, Cluster, m![1 # 4, K % 128 / 2], m![K % 128 % 2, N]> = b
40            .view()
41            .tile::<m![K / 128], 1, m![1 # 8, K % 128, N]>(i * 2)
42            .to_dm(&mut ctx.tdma, 0x0000_0300);
43        let t89: DmTensor<
44            bf16,
45            Chip,
46            Cluster,
47            m![1 # 4, M / 8, K % 128 / 16],
48            m![M % 8, K % 128 % 16 / 2, K % 128 % 16 % 2],
49        > = unsafe { t96.reshape() };
50        let t94: TrfTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, K % 128 / 16], m![M % 8], m![K % 128 % 16]> = ctx
51            .sub
52            .begin(t89.view())
53            .fetch::<bf16, m![M % 8], m![K % 128 % 16]>()
54            .collect::<m![M % 8], m![K % 128 % 16]>()
55            .to_trf(TrfAddress::FirstHalf);
56        let t88: DmTensor<
57            bf16,
58            Chip,
59            Cluster,
60            m![1 # 4, K % 128 / 2 / 8, K % 128 / 2 % 8],
61            m![K % 128 % 2, N / 32, N % 32],
62        > = unsafe { t72.reshape() };
63        let t90: DmTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, M % 8], m![N / 32, N % 32]> = ctx
64            .main
65            .begin(t88.view())
66            .fetch::<bf16, m![N / 32, K % 128 % 2], m![N % 32]>()
67            .switch::<m![1 # 4, M / 8, K % 128 / 16], m![N / 32, K % 128 % 2, K % 128 / 2 % 8]>(
68                SwitchConfig::TransposedBroadcast1 { slice1: 8, slice0: 8 },
69            )
70            .collect::<m![N / 32, K % 128 % 2, K % 128 / 2 % 8, N % 32 / 16], m![N % 32 % 16]>()
71            .align::<m![N / 32, K % 128 % 2, K % 128 / 2 % 8], m![N % 32], _, _>(&t94)
72            .contract::<m![N % 32]>()
73            .accumulate::<m![N / 32, M % 8, N % 32 / 8], m![N % 32 % 8]>(AccumulationKind::Sequential)
74            .vector_init()
75            .vector_inter_slice_reduce::<m![1 # 4, M / 8, M % 8], m![N / 32, N % 32 / 8]>(InterSliceReduceOpF32::Add)
76            .vector_final()
77            .cast::<bf16, m![N % 32 % 8 # 16]>()
78            .commit(0x0000_0100);
79
80        let t108: DmTensor<bf16, Chip, Cluster, m![1 # 4, M], m![K % 128]> = a
81            .view()
82            .tile::<m![K / 128], 1, m![M, 1 # 8, K % 128]>(1 + i * 2)
83            .to_dm(&mut ctx.tdma, 0x0000_0200);
84        let t116: DmTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, M % 8], m![K % 128 / 16, K % 128 % 16]> =
85            unsafe { t108.reshape() };
86        let t103: DmTensor<bf16, Chip, Cluster, m![1 # 4, K % 128 / 2], m![K % 128 % 2, N]> = b
87            .view()
88            .tile::<m![K / 128], 1, m![1 # 8, K % 128, N]>(1 + i * 2)
89            .to_dm(&mut ctx.tdma, 0x0000_0500);
90        let t109: DmTensor<
91            bf16,
92            Chip,
93            Cluster,
94            m![1 # 4, K % 128 / 2 / 8, K % 128 / 2 % 8],
95            m![K % 128 % 2, N / 32, N % 32],
96        > = unsafe { t103.reshape() };
97
98        let t80: DmTensor<bf16, Chip, Cluster, m![1 # 4, M], m![N]> = unsafe { t90.reshape() };
99
100        let t117: DmTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, K % 128 / 16], m![M % 8, K % 128 % 16]> = ctx
101            .main
102            .begin(t116.view())
103            .fetch::<bf16, m![K % 128 / 16], m![K % 128 % 16]>()
104            .switch::<m![1 # 4, M / 8, K % 128 / 16], m![M % 8]>(SwitchConfig::InterTranspose {
105                slice1: 8,
106                slice0: 1,
107                time0: 1,
108            })
109            .collect::<m![M % 8], m![K % 128 % 16]>()
110            .commit(0x0000_0300);
111        let t110: DmTensor<
112            bf16,
113            Chip,
114            Cluster,
115            m![1 # 4, M / 8, K % 128 / 16],
116            m![M % 8, K % 128 % 16 / 2, K % 128 % 16 % 2],
117        > = unsafe { t117.reshape() };
118
119        let t81: DmTensor<bf16, Chip, Cluster, m![1 # 4, M], m![N]> = ctx
120            .main
121            .begin_interleaved::<I, _, _, _, _, _>(acc.view(), t80.view())
122            .fetch::<f32, m![I, N / 8], m![N % 8]>()
123            .collect::<m![I, N / 8], m![N % 8]>()
124            .vector_init()
125            .vector_intra_slice_unzip::<I, m![1 # 2, N / 8], m![N / 8]>()
126            .vector_clip_zip(ClipBinaryOpF32::Add)
127            .vector_final()
128            .cast::<bf16, m![N % 8 # 16]>()
129            .commit(0x0000_0200);
130
131        let t115: TrfTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, K % 128 / 16], m![M % 8], m![K % 128 % 16]> = ctx
132            .sub
133            .begin(t110.view())
134            .fetch::<bf16, m![M % 8], m![K % 128 % 16]>()
135            .collect::<m![M % 8], m![K % 128 % 16]>()
136            .to_trf(TrfAddress::SecondHalf);
137
138        let t111: DmTensor<bf16, Chip, Cluster, m![1 # 4, M / 8, M % 8], m![N / 32, N % 32]> = ctx
139            .main
140            .begin(t109.view())
141            .fetch::<bf16, m![N / 32, K % 128 % 2], m![N % 32]>()
142            .switch::<m![1 # 4, M / 8, K % 128 / 16], m![N / 32, K % 128 % 2, K % 128 / 2 % 8]>(
143                SwitchConfig::TransposedBroadcast1 { slice1: 8, slice0: 8 },
144            )
145            .collect::<m![N / 32, K % 128 % 2, K % 128 / 2 % 8, N % 32 / 16], m![N % 32 % 16]>()
146            .align::<m![N / 32, K % 128 % 2, K % 128 / 2 % 8], m![N % 32], _, _>(&t115)
147            .contract::<m![N % 32]>()
148            .accumulate::<m![N / 32, M % 8, N % 32 / 8], m![N % 32 % 8]>(AccumulationKind::Sequential)
149            .vector_init()
150            .vector_inter_slice_reduce::<m![1 # 4, M / 8, M % 8], m![N / 32, N % 32 / 8]>(InterSliceReduceOpF32::Add)
151            .vector_final()
152            .cast::<bf16, m![N % 32 % 8 # 16]>()
153            .commit(0x0000_0100);
154        let t105: DmTensor<bf16, Chip, Cluster, m![1 # 4, M], m![N]> = unsafe { t111.reshape() };
155
156        acc = ctx
157            .main
158            .begin_interleaved::<I, _, _, _, _, _>(t81.view(), t105.view())
159            .fetch::<f32, m![I, N / 8], m![N % 8]>()
160            .collect::<m![I, N / 8], m![N % 8]>()
161            .vector_init()
162            .vector_intra_slice_unzip::<I, m![1 # 2, N / 8], m![N / 8]>()
163            .vector_clip_zip(ClipBinaryOpF32::Add)
164            .vector_final()
165            .cast::<bf16, m![N % 8 # 16]>()
166            .commit(0x0000_0000);
167    }
168
169    acc.to_hbm(&mut ctx.tdma, 0x3000)
170}
