1//! Matrix multiplication of size [1024, 2048] x [2048] -> [1024] with split accumulation pattern.
2//! This demonstrates the same split/accumulate pattern as matmul_16384 but with smaller size for testing.
3
4use furiosa_visa_std::prelude::*;
5
6axes![A = 1024, B = 2048, X = 32, I = 2];
7
8type Chip = m![1];
9type Cluster = m![A / 512 % 2];
10
11// TODO: Generics in a user-defined function is not supported yet. (LOW-76)
12fn add_split_contractions_i32(
13    ctx: &mut Context,
14    lhs: &DmTensor<i32, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>,
15    rhs: &DmTensor<i32, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>,
16    out: DmTensorViewMut<i32, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>,
17) {
18    ctx.main
19        .begin_interleaved::<I, _, _, _, _, _>(rhs.view(), lhs.view())
20        .fetch::<i32, m![A / 8 % 8, I], m![A % 8]>()
21        .collect::<m![A / 8 % 8, I], m![A % 8]>()
22        .vector_init()
23        .vector_intra_slice_unzip::<I, m![A / 8 % 8, 1 # 2], m![A / 8 % 8]>()
24        .vector_clip_zip(ClipBinaryOpI32::AddFxp)
25        .vector_final()
26        .commit_view(out)
27}

30fn add_split_contractions_i8(
31    ctx: &mut Context,
32    lhs: &DmTensor<i32, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>,
33    rhs: &DmTensor<i32, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>,
34    out: DmTensorViewMut<i8, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>,
35) {
36    ctx.main
37        .begin_interleaved::<I, _, _, _, _, _>(rhs.view(), lhs.view())
38        .fetch::<i32, m![A / 8 % 8, I], m![A % 8]>()
39        .collect::<m![A / 8 % 8, I], m![A % 8]>()
40        .vector_init()
41        .vector_intra_slice_unzip::<I, m![A / 8 % 8, 1 # 2], m![A / 8 % 8]>()
42        .vector_clip_zip(ClipBinaryOpI32::AddFxp)
43        .vector_final()
44        .cast::<i8, m![A % 8 # 32]>()
45        .commit_view(out)
46}

48/// Multiply matrices of size \[1024, 2048\] x \[2048\] -> \[1024\]
49/// Demonstrates split/accumulate pattern: B axis is split into 4 tiles (B=512 each)
50/// and accumulated by stashing one group inside VE and later merging with the other group.
51#[device(chip = 1)]
52pub fn matmul_with_split_reduce(
53    ctx: &mut Context,
54    lhs: &HbmTensor<i8, m![1], m![A, B]>,
55    rhs: &HbmTensor<i8, m![1], m![B]>,
56) -> HbmTensor<i8, m![1], m![A]> {
57    type AccDmTensor = DmTensor<i32, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>;
58    let mut result0 = unsafe { AccDmTensor::from_addr(0) };
59    let mut result1 = unsafe { AccDmTensor::from_addr(0) };
60    let mut temp = unsafe { AccDmTensor::from_addr(0) };
61    let mut output = unsafe { DmTensor::<i8, Chip, Cluster, m![A / 64 % 8, X], m![A % 64]>::from_addr(0) };
62
63    // Split B axis into 4 tiles (B=512 each) and accumulate using vector engine
64    for j in 0..4 {
65        // Load lhs and rhs for this B tile
66        let lhs_tile = lhs
67            .view()
68            .tile::<m![B / 512], 1, m![A, 1 # 4, B % 512]>(j)
69            .to_dm::<Cluster, m![A / 64 % 8, B / 16 % 32], m![A % 64, B % 16]>(&mut ctx.tdma, 0);
70        let rhs_tile = rhs
71            .view()
72            .tile::<m![B / 512], 1, m![1 # 4, B % 512]>(j)
73            .to_dm::<Cluster, m![A / 64 % 8, B / 16 % 32], m![B % 16]>(&mut ctx.tdma, 256 * 1024);
74        let rhs_trf: TrfTensor<i8, Chip, Cluster, m![A / 64 % 8, B / 16 % 32], m![1], m![B % 16 # 32]> = ctx
75            .sub
76            .begin(rhs_tile.view())
77            .fetch::<i8, m![1], m![B % 16]>()
78            .collect::<m![1], m![B % 16 # 32]>()
79            .to_trf(TrfAddress::Full);
80
81        // Perform contraction for this tile
82        if j == 0 {
83            // First iteration: store result directly in result0
84            ctx.main
85                .begin(lhs_tile.view())
86                .fetch::<i8, m![A / 2 % 32], m![A % 2, B % 16]>()
87                .collect::<m![A / 2 % 32], m![A % 2, B % 16]>()
88                .align::<m![A / 4 % 16], m![A % 4, B % 16], _, _>(&rhs_trf)
89                .contract::<m![A % 4]>()
90                .accumulate::<m![A / 4 % 16], m![A % 4 # 8]>(AccumulationKind::Sequential)
91                .vector_init()
92                .vector_inter_slice_reduce::<m![A / 64 % 8, X], m![A / 4 % 16]>(InterSliceReduceOpI32::Add)
93                .vector_final()
94                .commit_view(result0.view_mut());
95        } else if j == 3 {
96            // Final iteration: store in output, conversion to i8 happens
97            ctx.main
98                .begin(lhs_tile.view())
99                .fetch::<i8, m![A / 2 % 32], m![A % 2, B % 16]>()
100                .collect::<m![A / 2 % 32], m![A % 2, B % 16]>()
101                .align::<m![A / 4 % 16], m![A % 4, B % 16], _, _>(&rhs_trf)
102                .contract::<m![A % 4]>()
103                .accumulate::<m![A / 4 % 16], m![A % 4 # 8]>(AccumulationKind::Sequential)
104                .vector_init()
105                .vector_inter_slice_reduce::<m![A / 64 % 8, X], m![A / 4 % 16]>(InterSliceReduceOpI32::Add)
106                .vector_final()
107                .commit_view(temp.view_mut());
108
109            // Accumulate using vector engine: stash previous partial sums and zip with new ones
110            add_split_contractions_i8(ctx, &result0, &temp, output.view_mut());
111        } else {
112            // Subsequent iterations: store in temp and accumulate
113            ctx.main
114                .begin(lhs_tile.view())
115                .fetch::<i8, m![A / 2 % 32], m![A % 2, B % 16]>()
116                .collect::<m![A / 2 % 32], m![A % 2, B % 16]>()
117                .align::<m![A / 4 % 16], m![A % 4, B % 16], _, _>(&rhs_trf)
118                .contract::<m![A % 4]>()
119                .accumulate::<m![A / 4 % 16], m![A % 4 # 8]>(AccumulationKind::Sequential)
120                .vector_init()
121                .vector_inter_slice_reduce::<m![A / 64 % 8, X], m![A / 4 % 16]>(InterSliceReduceOpI32::Add)
122                .vector_final()
123                .commit_view(temp.view_mut());
124
125            // Accumulate using vector engine: stash the older tile internally and zipped with the new tile
126            if j % 2 == 1 {
127                add_split_contractions_i32(ctx, &result0, &temp, result1.view_mut());
128            } else {
129                add_split_contractions_i32(ctx, &result1, &temp, result0.view_mut());
130            }
131        }
132    }
133
134    // Write back to HBM
135    let mut out = unsafe { HbmTensor::<i8, m![1], m![A]>::from_addr(0x3000) };
136    output
137        .view()
138        .slice_tile::<m![X], 1, m![A / 64 % 8, 1 # 32]>(0)
139        .to_hbm_view(&mut ctx.tdma, out.view_mut());
140    out
141}
