30#[expect(missing_docs)]
31impl Ident {
32    /// Creates a new identifier.
33    ///
34    /// The identifier must start with an uppercase ASCII letter and contain
35    /// only ASCII alphanumeric characters or underscores.
36    pub const fn new(s: &'static str) -> Self {
37        let b = s.as_bytes();
38        assert!(!b.is_empty(), "Ident must not be empty");
39        assert!(
40            b[0].is_ascii_uppercase(),
41            "Ident must start with an uppercase ASCII letter"
42        );
43        let mut i = 1;
44        while i < b.len() {
45            assert!(
46                b[i].is_ascii_alphanumeric() || b[i] == b'_',
47                "Ident must contain only ASCII alphanumeric or underscore characters"
48            );
49            i += 1;
50        }
51        Self(s)
52    }

57    }

119impl<'a> TryFrom<&'a str> for Ident {
120    type Error = &'a str;

122    fn try_from(value: &'a str) -> std::result::Result<Self, Self::Error> {
123        use lasso::ThreadedRodeo;
124        use std::sync::LazyLock;
125        static INTERNER: LazyLock<ThreadedRodeo> = LazyLock::new(ThreadedRodeo::new);
126
127        let key = INTERNER.get_or_intern(value);
128        let interned: &'static str = INTERNER.resolve(&key);
129        std::panic::catch_unwind(|| Self::new(interned)).map_err(|_| value)
130    }

133/// Mapping expression enum.
134#[repr(C)]
135#[derive(StableAbi, Debug, Clone, PartialEq, Eq, Hash)]
136pub enum Mapping {
137    /// Identity mapping.
138    Identity,
139    /// Symbol mapping.
140    Symbol {
141        /// Symbol.
142        symbol: Ident,
143        /// Size.
144        size: usize,
145    },
146    /// Stride mapping.
147    Stride {
148        /// Inner mapping.
149        inner: RBox<Mapping>,
150        /// Stride size.
151        stride: usize,
152    },
153    /// Modulo mapping.
154    Modulo {
155        /// Inner mapping.
156        inner: RBox<Mapping>,
157        /// Stride size.
158        modulo: usize,
159    },
160    /// Resize mapping.
161    Resize {
162        /// Inner mapping.
163        inner: RBox<Mapping>,
164        /// Truncate size.
165        resize: usize,
166    },
167    /// Padding mapping.
168    Padding {
169        /// Inner mapping.
170        inner: RBox<Mapping>,
171        /// Size after padding.
172        padding: usize,
173        /// Accessibility of this padding region.
174        kind: PaddingKind,
175    },
176    /// Pair mapping.
177    Pair {
178        /// Left mapping.
179        left: RBox<Mapping>,
180        /// Right mapping.
181        right: RBox<Mapping>,
182    },
183}
184
185impl Display for Mapping {
186    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
187        fn flatten_pair<'a>(acc: &mut Vec<&'a Mapping>, m: &'a Mapping) {
188            match m {
189                Mapping::Pair { left, right } => {
190                    flatten_pair(acc, left);
191                    flatten_pair(acc, right);
192                }
193                _ => acc.push(m),
194            }
195        }
196
197        match self {
198            Self::Identity => write!(f, "1"),
199            Self::Symbol { symbol, size: _ } => {
200                // We hide the size just for readability.
201                write!(f, "{symbol}")
202            }
203            Self::Stride { inner, stride } => write!(f, "{inner} / {stride}"),
204            Self::Modulo { inner, modulo } => write!(f, "{inner} % {modulo}"),
205            Self::Resize { inner, resize } => write!(f, "{inner} = {resize}"),
206            Self::Padding {
207                inner,
208                padding,
209                kind: PaddingKind::Top,
210            } => write!(f, "{inner} # {padding}"),
211            Self::Padding {
212                inner,
213                padding,
214                kind: PaddingKind::Bottom,
215            } => write!(f, "{inner} #_ {padding}"),
216            Self::Pair { left, right } => {
217                // Collect all nested pairs and print them as flattened.
218                let mut elements = vec![];
219                flatten_pair(&mut elements, left);
220                flatten_pair(&mut elements, right);
221                write!(f, "({})", elements.iter().join(", "))
222            }
223        }
224    }
225}

352/// `inner / stride % modulo`.
353#[repr(C)]
354#[derive(StableAbi, Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
355pub struct Term {
356    /// Inner mapping expression.
357    pub inner: Atom,
358    /// Stride of the mapping.
359    pub stride: usize,
360    /// Modulo of the mapping.
361    pub modulo: usize,
362}
363
364impl Display for Term {
365    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
366        match &self.inner {
367            Atom::Symbol { symbol, size } => {
368                if self.stride == 1 && self.modulo == *size {
369                    write!(f, "{}", symbol)
370                } else if self.stride == 1 {
371                    write!(f, "{}%{}", symbol, self.modulo)
372                } else if self.modulo == *size {
373                    write!(f, "{}//{}", symbol, self.stride)
374                } else {
375                    write!(f, "({}//{})%{}", symbol, self.stride, self.modulo)
376                }
377            }
378            Atom::Composite(inner) => {
379                if self.stride == 1 && self.modulo == inner.size() {
380                    write!(f, "({})", inner)
381                } else if self.stride == 1 {
382                    write!(f, "({})%{}", inner, self.modulo)
383                } else if self.modulo == inner.size() {
384                    write!(f, "({})//{}", inner, self.stride)
385                } else {
386                    write!(f, "(({})//{})%{}", inner, self.stride, self.modulo)
387                }
388            }
389        }
390    }
391}

436/// Factor mapping expression.
437///
438/// Factors are ordered from innermost (index 0) to outermost (last index).
439/// `push`/`pop`/`last` operate on the outermost factor.
440#[repr(C)]
441#[derive(StableAbi, Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
442pub struct FMapping(pub RVec<Factor>);
443
444impl Default for FMapping {
445    fn default() -> Self {
446        Self::new()
447    }
448}
449
450impl FMapping {
451    /// Creates a new empty factor mapping.
452    pub fn new() -> Self {
453        Self(RVec::new())
454    }
455
456    /// Returns the size of the factor mapping.
457    pub fn size(&self) -> usize {
458        let mut x = 1;
459        for term in self.0.iter().rev() {
460            match term {
461                Factor::Padding { size, .. } => {
462                    return x * *size;
463                }
464                Factor::Term { resize, .. } => {
465                    x *= *resize;
466                }
467            }
468        }
469        x
470    }
471}
472
473impl Display for FMapping {
474    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
475        let terms: Vec<String> = self
476            .0
477            .iter()
478            .map(|term| match term {
479                Factor::Padding {
480                    size,
481                    kind: PaddingKind::Top,
482                } => format!("pad({})", size),
483                Factor::Padding {
484                    size,
485                    kind: PaddingKind::Bottom,
486                } => format!("bottom_pad({})", size),
487                Factor::Term {
488                    inner: Term { inner, stride, modulo },
489                    resize,
490                } => {
491                    let atom_str = match inner {
492                        Atom::Symbol { symbol, size } => format!("{}:{}", symbol, size),
493                        Atom::Composite(inner) => format!("({})", inner),
494                    };
495                    format!("term({} / {} % {} = {})", atom_str, stride, modulo, resize)
496                }
497            })
498            .collect();
499        write!(f, "FMapping[{}]", terms.join(" * "))
500    }
501}

584/// Result of dividing two factor mappings.
585#[repr(C)]
586#[derive(StableAbi, Debug, Clone, PartialEq, Eq, Default)]
587pub struct Division {
588    /// Information about each matched divisor term.
589    pub division_terms: RVec<DivisionTerm>,
590    /// Original dividend before matching.
591    pub dividend: FMapping,
592    /// Dividend residue. Internal encoding: matched-term positions appear as
593    /// `PaddingKind::Bottom` factors. Consume via `DivisionExt::{extents,
594    /// relaxed_residues, tile_residues, remainder}` rather than walking
595    /// directly.
596    pub dividend_residue: FMapping,
597    /// Original divisor before matching.
598    pub divisor: FMapping,
599    /// Divisor residue. Internal encoding: matched-term positions appear as
600    /// `PaddingKind::Bottom` factors. Consume via `DivisionExt::{extents,
601    /// relaxed_residues, tile_residues, remainder}` rather than walking
602    /// directly.
603    pub divisor_residue: FMapping,
604}

606impl Division {
607    /// Creates a new division result.
608    pub fn new(
609        dividend: FMapping,
610        dividend_residue: FMapping,
611        mut division_terms: Vec<DivisionTerm>,
612        divisor: FMapping,
613        divisor_residue: FMapping,
614    ) -> Self {
615        division_terms.sort_by_key(|b| std::cmp::Reverse(b.dividend_stride));
616
617        Self {
618            division_terms: division_terms.into(),
619            dividend,
620            dividend_residue,
621            divisor,
622            divisor_residue,
623        }
624    }
625
626    /// Returns the original mapping for the selected side.
627    pub fn mapping(&self, side: DivisionSide) -> &FMapping {
628        match side {
629            DivisionSide::Dividend => &self.dividend,
630            DivisionSide::Divisor => &self.divisor,
631        }
632    }
633
634    /// Returns matched division terms in dividend-order.
635    pub fn division_terms(&self) -> &[DivisionTerm] {
636        &self.division_terms
637    }

725///
726/// Returned by `Extents::slots` (see `furiosa-mapping`'s `ExtentsExt`).
727#[repr(C)]
728#[derive(StableAbi, Debug, Clone, PartialEq, Eq)]
729pub enum Slot {
730    /// A matched term with its compact bounds on both sides.
731    Matched(TermBounds),
732    /// A surviving (unmatched) term with its original stride.
733    Unmatched(TermPosition),
734    /// A Top padding range on this side.
735    Padding(Range),
736}
737
738impl Extents {
739    /// Returns the side-local extent for the selected side.
740    pub fn side(&self, side: DivisionSide) -> &SideExtent {
741        match side {
742            DivisionSide::Dividend => &self.dividend,
743            DivisionSide::Divisor => &self.divisor,
744        }
745    }
746}

748/// Index mapping for tensor operations.
749#[repr(C)]
750#[derive(StableAbi, Debug, Clone, PartialEq, Eq)]
751pub struct Index(pub RResult<RSortedMap<Term, usize>, ()>);
752
753/// Error returned when querying ident contributions from an [`Index`].
754#[repr(C)]
755#[derive(StableAbi, Debug, Clone, Copy, PartialEq, Eq)]
756pub enum IndexValueError {
757    /// The index is invalid, typically because it points into padding.
758    Invalid,
759    /// The index still contains Composite terms and is not evaluable per ident.
760    NonFlattened,
761}
762
763impl Default for Index {
764    fn default() -> Self {
765        Self(RResult::ROk(RSortedMap::new()))
766    }
767}
768
769impl Display for Index {
770    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
771        match &self.0 {
772            RResult::ROk(map) => {
773                let terms = map.iter().map(|(k, v)| format!("{k} = {v}")).join(", ");
774                write!(f, "Index[{}]", terms)
775            }
776            RResult::RErr(_) => {
777                write!(f, "Invalid Index")
778            }
779        }
780    }
781}
