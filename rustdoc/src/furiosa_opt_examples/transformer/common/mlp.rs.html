1//! MLP block (SwiGLU activation).
2//!
3//! PyTorch Qwen2MLP.forward():
4//!   x = silu(gate_proj(x)) * up_proj(x)    # SiLU(x) = x * sigmoid(x)
5//!   x = down_proj(x)
6//!
7//! This kernel computes gate and up projections, applies SwiGLU, then runs
8//! down projection back to hidden size.
9
10use crate::transformer::axes::{
11    H,             // = 896, hidden_size = 14 * 64
12    M,             // = 4864, mlp_intermediate_size
13    S_decode as S, // = 128, sequence length
14    X,             // = 16, broadcast
15    Y,             // = 4, broadcast
16};
17use furiosa_opt_std::prelude::*;
18
19type Chip = m![1];
20type Cluster = m![1 # 2]; // 1 logical cluster, padded to 2 (hardware has 2 clusters/chip)
21
22/// MLP (SwiGLU): normalized [S, H] → mlp_output [S, H] in DRAM.
23pub(crate) fn mlp(
24    ctx: &mut Context,
25    input: &DmTensor<bf16, Chip, Cluster, m![Y, S / 32, H / 56], m![S % 32, H % 56]>,
26    gate_weight: &HbmTensor<bf16, Chip, m![M, H]>,
27    up_weight: &HbmTensor<bf16, Chip, m![M, H]>,
28    down_weight: &HbmTensor<bf16, Chip, m![H, M]>,
29) -> HbmTensor<bf16, Chip, m![S, H]> {
30    let gate_dm: DmTensor<bf16, Chip, Cluster, m![M / 76, M / 19 % 4], m![M % 19, H]> =
31        gate_weight.to_dm(&mut ctx.tdma, 0x0);
32
33    let gate_it: DmTensor<bf16, Chip, Cluster, m![M / 76, H / 224], m![M / 19 % 4, M % 19, H % 224]> = ctx
34        .main
35        .begin(gate_dm.view())
36        .fetch::<bf16, m![M % 19, H / 224], m![H % 224]>()
37        .switch::<m![M / 76, H / 224], m![M / 19 % 4, M % 19]>(SwitchConfig::InterTranspose {
38            slice1: 4,
39            slice0: 1,
40            time0: 1,
41        })
42        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
43        .commit(0x10000);
44
45    // Reshape gate_it Slice to match TRF Slice for alignment.
46    let gate_it: DmTensor<bf16, Chip, Cluster, m![M / 1216, X, H / 224], m![M / 19 % 4, M % 19, H % 224]> =
47        unsafe { gate_it.reshape() };
48
49    let input_trf_second: TrfTensor<
50        bf16,
51        Chip,
52        Cluster,
53        m![M / 1216, X, H / 224],
54        m![S % 8],
55        m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16],
56    > = ctx
57        .sub
58        .begin(input.view())
59        .fetch::<bf16, m![S % 8], m![H / 8 % 7, H % 8]>()
60        .switch::<m![M / 1216, X, H / 224], m![S % 8]>(SwitchConfig::Broadcast1 { slice1: 16, slice0: 8 })
61        .collect::<m![S % 8], m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16]>()
62        .to_trf(TrfAddress::SecondHalf);
63
64    let input_trf_first: TrfTensor<
65        bf16,
66        Chip,
67        Cluster,
68        m![M / 1216, X, H / 224],
69        m![S % 8],
70        m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16],
71    > = ctx
72        .sub
73        .begin(input.view())
74        .fetch::<bf16, m![S % 8], m![H / 8 % 7, H % 8]>()
75        .switch::<m![M / 1216, X, H / 224], m![S % 8]>(SwitchConfig::Broadcast1 { slice1: 16, slice0: 8 })
76        .collect::<m![S % 8], m![H / 8 % 7, H / 56 % 4, S / 32, S % 8 # 16]>()
77        .to_trf(TrfAddress::FirstHalf);
78
79    let gate_second: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
80        .main
81        .begin(gate_it.view())
82        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
83        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
84        .contract_outer::<m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16], _, _>(&input_trf_second)
85        .contract_packet::<m![1]>()
86        .contract_time::<m![M % 19, S / 8 % 16]>()
87        .contract_lane::<m![M % 19, S / 8 % 16], m![S % 8]>(LaneMode::Interleaved)
88        .vector_init()
89        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
90        .vector_final()
91        .cast::<bf16, m![S % 128]>()
92        .commit(0x0);
93
94    let gate_first_raw: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
95        .main
96        .begin(gate_it.view())
97        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
98        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
99        .contract_outer::<m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16], _, _>(&input_trf_first)
100        .contract_packet::<m![1]>()
101        .contract_time::<m![M % 19, S / 8 % 16]>()
102        .contract_lane::<m![M % 19, S / 8 % 16], m![S % 8]>(LaneMode::Interleaved)
103        .vector_init()
104        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
105        .vector_final()
106        .cast::<bf16, m![S % 128]>()
107        .commit(0x10000);
108
109    let gate_second_vrf: VrfTensor<f32, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
110        .sub
111        .begin(gate_second.view())
112        .fetch::<f32, m![M % 19], m![S % 128]>()
113        .collect::<m![M % 19], m![S % 128]>()
114        .to_vrf(0);
115
116    let gate_sum: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
117        .main
118        .begin(gate_first_raw.view())
119        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
120        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
121        .vector_init()
122        .vector_intra_slice_tag(TagMode::Zero)
123        .vector_narrow_split::<m![M % 19, S / 8 % 16, S / 4 % 2], m![S % 4]>()
124        .vector_fp_binary(FpBinaryOp::AddF, &gate_second_vrf)
125        .vector_widen_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
126        .vector_final()
127        .cast::<bf16, m![S % 8 # 16]>()
128        .commit(0x10000);
129
130    let gate_sigmoid: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
131        .main
132        .begin(gate_sum.view())
133        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
134        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
135        .vector_init()
136        .vector_intra_slice_tag(TagMode::Zero)
137        .vector_narrow_split::<m![S / 4 % 2, M % 19, S / 8 % 16], m![S % 4]>()
138        .vector_fp_unary(FpUnaryOp::Sigmoid)
139        .vector_widen_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
140        .vector_final()
141        .cast::<bf16, m![S % 8 # 16]>()
142        .commit(0x18000);
143
144    let up_dm: DmTensor<bf16, Chip, Cluster, m![M / 76, M / 19 % 4], m![M % 19, H]> =
145        up_weight.to_dm(&mut ctx.tdma, 0x20000);
146
147    let up_it: DmTensor<bf16, Chip, Cluster, m![M / 76, H / 224], m![M / 19 % 4, M % 19, H % 224]> = ctx
148        .main
149        .begin(up_dm.view())
150        .fetch::<bf16, m![M % 19, H / 224], m![H % 224]>()
151        .switch::<m![M / 76, H / 224], m![M / 19 % 4, M % 19]>(SwitchConfig::InterTranspose {
152            slice1: 4,
153            slice0: 1,
154            time0: 1,
155        })
156        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
157        .commit(0x28000);
158
159    // Reshape up_it Slice to match TRF Slice for alignment.
160    let up_it: DmTensor<bf16, Chip, Cluster, m![M / 1216, X, H / 224], m![M / 19 % 4, M % 19, H % 224]> =
161        unsafe { up_it.reshape() };
162
163    let up_second: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
164        .main
165        .begin(up_it.view())
166        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
167        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
168        .contract_outer::<m![M / 19 % 4, M % 19, H / 32 % 7], m![H % 32], _, _>(&input_trf_second)
169        .contract_packet::<m![1]>()
170        .contract_time::<m![M % 19, S / 8 % 16]>()
171        .contract_lane::<m![M % 19, S / 8 % 16], m![S % 8]>(LaneMode::Interleaved)
172        .vector_init()
173        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
174        .vector_final()
175        .cast::<bf16, m![S % 128]>()
176        .commit(0x20000);
177
178    let up_first_raw: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
179        .main
180        .begin(up_it.view())
181        .fetch::<bf16, m![M / 19 % 4, M % 19], m![H / 16 % 14, H % 16]>()
182        .collect::<m![M / 19 % 4, M % 19, H / 16 % 14], m![H % 16]>()
183        .contract_outer::<m![M / 19 % 4, M % 19, H / 32 % 7], m![H % 32], _, _>(&input_trf_first)
184        .contract_packet::<m![1]>()
185        .contract_time::<m![M % 19, S / 8 % 16]>()
186        .contract_lane::<m![M % 19, S / 8 % 16], m![S % 8]>(LaneMode::Interleaved)
187        .vector_init()
188        .vector_inter_slice_reduce::<m![M / 19, S / 128], m![M % 19, S % 128]>(InterSliceReduceOpF32::Add)
189        .vector_final()
190        .cast::<bf16, m![S % 128]>()
191        .commit(0x30000);
192
193    let up_second_vrf: VrfTensor<f32, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
194        .sub
195        .begin(up_second.view())
196        .fetch::<f32, m![M % 19], m![S % 128]>()
197        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
198        .to_vrf(0);
199
200    let up_sum: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
201        .main
202        .begin(up_first_raw.view())
203        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
204        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
205        .vector_init()
206        .vector_intra_slice_tag(TagMode::Zero)
207        .vector_narrow_split::<m![M % 19, S / 8 % 16, S / 4 % 2], m![S % 4]>()
208        .vector_fp_binary(FpBinaryOp::AddF, &up_second_vrf)
209        .vector_widen_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
210        .vector_final()
211        .cast::<bf16, m![S % 8 # 16]>()
212        .commit(0x38000);
213
214    let up_vrf: VrfTensor<f32, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
215        .sub
216        .begin(up_sum.view())
217        .fetch::<f32, m![M % 19], m![S % 128]>()
218        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
219        .to_vrf(0);
220
221    let gated: DmTensor<bf16, Chip, Cluster, m![M / 19, S / 128], m![M % 19, S % 128]> = ctx
222        .main
223        .begin(gate_sigmoid.view())
224        .fetch::<f32, m![M % 19, S / 8 % 16], m![S % 8]>()
225        .collect::<m![M % 19, S / 8 % 16], m![S % 8]>()
226        .vector_init()
227        .vector_intra_slice_tag(TagMode::Zero)
228        .vector_narrow_split::<m![M % 19, S / 8 % 16, S / 4 % 2], m![S % 4]>()
229        .vector_fp_binary(FpBinaryOp::MulF(FpMulAlu::Mul0), &up_vrf)
230        .vector_widen_concat::<m![M % 19, S / 8 % 16], m![S % 8]>()
231        .vector_final()
232        .cast::<bf16, m![S % 8 # 16]>()
233        .commit(0x0);
234
235    let gated_hbm: HbmTensor<bf16, Chip, m![M, S]> = gated.to_hbm(&mut ctx.tdma, 0x256d000);
236
237    let gated_retiled: DmTensor<bf16, Chip, Cluster, m![Y, M / 76], m![M % 76, S]> =
238        gated_hbm.to_dm(&mut ctx.tdma, 0x9000);
239
240    // Reshape gated activation Slice for down-projection TRF alignment.
241    let gated_retiled: DmTensor<bf16, Chip, Cluster, m![H / 224, M / 152, M / 76 % 2], m![M % 76, S]> =
242        unsafe { gated_retiled.reshape() };
243
244    // Load gated activation into TRF for down projection.
245    let gated_trf: TrfTensor<bf16, Chip, Cluster, m![H / 224, M / 152, M / 76 % 2], m![M % 76], m![S]> = ctx
246        .sub
247        .begin(gated_retiled.view())
248        .fetch::<bf16, m![M % 76], m![S]>()
249        .collect::<m![M % 76], m![S]>()
250        .to_trf(TrfAddress::Full);
251    let down_dm: DmTensor<bf16, Chip, Cluster, m![H / 224, M / 152, H / 112 % 2], m![H % 112, M % 152]> =
252        down_weight.to_dm(&mut ctx.tdma, 0x30000);
253    let down_it: DmTensor<
254        bf16,
255        Chip,
256        Cluster,
257        m![H / 224, M / 152, M / 76 % 2],
258        m![H / 112 % 2, H % 112, M % 76 # 80],
259    > = ctx
260        .main
261        .begin(down_dm.view())
262        .fetch::<bf16, m![H % 112, M / 76 % 2], m![M % 76]>()
263        .switch::<m![H / 224, M / 152, M / 76 % 2], m![H / 112 % 2, H % 112]>(SwitchConfig::InterTranspose {
264            slice1: 2,
265            slice0: 1,
266            time0: 1,
267        })
268        .collect::<m![H / 112 % 2, H % 112], m![M % 76 # 80]>()
269        .commit(0x34000);
270    // Reorder down-projection weight tiles for contraction alignment.
271    let down_transposed: DmTensor<bf16, Chip, Cluster, m![H / 224, M / 152, M / 76 % 2], m![M % 76, H % 224]> = ctx
272        .main
273        .begin(down_it.view())
274        .fetch::<bf16, m![H / 112 % 2, H % 112], m![M % 76]>()
275        .collect::<m![M % 76], m![H % 224]>()
276        .commit(0x20000);
277
278    // Compute down projection with gated activation staged in TRF.
279    let down_raw: DmTensor<bf16, Chip, Cluster, m![H / 56, S / 16], m![H % 56, S % 16]> = ctx
280        .main
281        .begin(down_transposed.view())
282        .fetch::<bf16, m![M % 76], m![H % 224]>()
283        .collect::<m![M % 76], m![H % 224]>()
284        .contract_outer::<m![M % 76], m![H % 224], _, _>(&gated_trf)
285        .contract_packet::<m![1]>()
286        .contract_time::<m![1]>()
287        .contract_lane::<m![M % 76], m![1 # 8]>(LaneMode::Sequential)
288        .vector_init()
289        .vector_inter_slice_reduce::<m![H / 56, S / 16], m![H % 56, S % 16]>(InterSliceReduceOpF32::Add)
290        .vector_final()
291        .cast::<bf16, m![S % 16]>()
292        .commit(0x8000);
293
294    let down_result: DmTensor<bf16, Chip, Cluster, m![H / 56, S / 16], m![S % 32, H % 56]> = ctx
295        .main
296        .begin(down_raw.view())
297        .fetch::<bf16, m![H % 56, S / 16 % 2], m![S % 16]>()
298        .collect::<m![S % 32], m![H % 56]>()
299        .commit(0x0);
300
301    // Write final MLP output tensor to HBM.
302    down_result.to_hbm(&mut ctx.tdma, 0x10e50000)
